### Overview

* `hash_generator`: A module responsible for generating hashes for the URLs.
* `url_validator`: A module that validates the input URLs.
* `storage`: A module that stores the mapping between short and original URLs.

### Interfaces

The library provides the following interfaces:

* `shorten_url(long_url)`: Returns a shortened version of the input URL.
* `get_original_url(short_url)`: Returns the original URL corresponding to the input short URL.

### Data Flow

The data flow of the library is as follows:

2. The `url_validator` module checks if the input URL is valid.
3. If the URL is valid, the `hash_generator` module generates a hash for the URL.
4. The `storage` module checks if the generated hash already exists.
5. If the hash does not exist, the `storage` module stores the mapping between the short URL and the original URL.
6. The `shorten_url(long_url)` function returns the shortened URL.
7. The user can call the `get_original_url(short_url)` function to retrieve the original URL corresponding to a short URL.

```python
import hashlib
import urllib.parse
import threading

class URLShortener:
    def __init__(self):
        self.storage = {}
        self.lock = threading.Lock()

    def shorten_url(self, long_url):
        if not self.is_valid_url(long_url):
            raise ValueError("Invalid URL")

        with self.lock:
            hash_object = hashlib.sha256(long_url.encode())
            short_url = hash_object.hexdigest()[:6]

            if short_url in self.storage:
                # Collision detection: append a counter to the short URL
                counter = 1
                while f"{short_url}{counter}" in self.storage:
                    counter += 1
                short_url = f"{short_url}{counter}"

            self.storage[short_url] = long_url

        return short_url

    def get_original_url(self, short_url):
        with self.lock:
            return self.storage.get(short_url)

    def is_valid_url(self, url):
        try:
            result = urllib.parse.urlparse(url)
            return all([result.scheme, result.netloc])
        except ValueError:
            return False

# Example usage:
url_shortener = URLShortener()
long_url = "https://www.example.com/very/long/url"
short_url = url_shortener.shorten_url(long_url)
print(f"Short URL: {short_url}")

import urllib.parse
import threading
import sqlite3

class URLShortener:
    def __init__(self, db_name):
        self.db_name = db_name
        self.conn = sqlite3.connect(db_name)
        self.cursor = self.conn.cursor()
        self.lock = threading.Lock()

        self.cursor.execute("""
            CREATE TABLE IF NOT EXISTS urls
            (short_url TEXT PRIMARY KEY, long_url TEXT)
        """)

        self.cursor.execute("""
            CREATE TABLE IF NOT EXISTS hashes
            (hash TEXT PRIMARY KEY, short_url TEXT)
        """)

        self.conn.commit()

    def shorten_url(self, long_url):
        if not self.is_valid_url(long_url):
            raise ValueError("Invalid URL")

        with self.lock:
            hash_object = hashlib.sha256(long_url.encode())
            short_url = hash_object.hexdigest()[:6]

            if self.cursor.execute("SELECT * FROM urls WHERE short_url = ?", (short_url,)).fetchone():

                counter = 1
                while self.cursor.execute("SELECT * FROM urls WHERE short_url = ?", (f"{short_url}{counter}",)).fetchone():
                    counter += 1
                short_url = f"{short_url}{counter}"

            self.cursor.execute("INSERT INTO urls VALUES (?, ?)", (short_url, long_url))
            self.cursor.execute("INSERT INTO hashes VALUES (?, ?)", (hash_object.hexdigest(), short_url))
            self.conn.commit()

            return self.cursor.execute("SELECT long_url FROM urls WHERE short_url = ?", (short_url,)).fetchone()[0]

        if not self.is_valid_url(new_long_url):
            raise ValueError("Invalid URL")

        with self.lock:
            self.cursor.execute("UPDATE urls SET long_url = ? WHERE short_url = ?", (new_long_url, short_url))
            self.conn.commit()

class URLShortener:
    def __init__(self, db_name):

                short_url = hash_object.hexdigest()[:6]

                if self.cursor.execute("SELECT * FROM urls WHERE short_url = ?", (short_url,)).fetchone():
                    # Collision detection: append a counter to the short URL
                    counter = 1
                    while self.cursor.execute("SELECT * FROM urls WHERE short_url = ?", (f"{short_url}{counter}",)).fetchone():
                        counter += 1
                    short_url = f"{short_url}{counter}"

                self.cursor.execute("INSERT INTO urls VALUES (?, ?)", (short_url, long_url))

                self.conn.commit()

            return short_url
        except Exception as e:
            self.logger.error(f"Error shortening URL: {e}")
            raise

        try:

        except Exception as e:
