import threading

from typing import Callable

class TokenBucket:
    def __init__(self, rate: int, burst_size: int):
        self.rate = rate
        self.burst_size = burst_size
        self.available_tokens = burst_size
        self.last_replenish_time = time.time()

    def consume_token(self) -> bool:
        if self.available_tokens > 0:
            self.available_tokens -= 1
            return True
        return False

    def replenish_tokens(self):
        current_time = time.time()
        elapsed_time = current_time - self.last_replenish_time
        self.last_replenish_time = current_time
        self.available_tokens = min(self.burst_size, self.available_tokens + elapsed_time * self.rate)
        if self.available_tokens < 0:
            self.available_tokens = 0

class RateLimiter:
    def __init__(self, rate: int, burst_size: int):
        self.rate = rate
        self.burst_size = burst_size
        self.token_bucket = TokenBucket(rate, burst_size)
        self.lock = threading.Lock()

    def allow_request(self) -> bool:
        with self.lock:
            self.token_bucket.replenish_tokens()
            if self.token_bucket.consume_token():
                return True
            return False

    def decorator(self, func: Callable):
        def wrapper(*args, **kwargs):
            if self.allow_request():
                return func(*args, **kwargs)
            else:
                raise Exception("Rate limit exceeded")
        return wrapper

### Overview

The rate limiter module is designed to provide a flexible and thread-safe way to limit the rate of requests to a system. It uses the token bucket algorithm to manage the rate of requests and supports multiple named rate limiters.

*   `RateLimiter`: The main class responsible for managing the rate of requests. It uses a token bucket algorithm to track the available tokens and replenish them at a specified rate.
*   `TokenBucket`: A class that implements the token bucket algorithm. It tracks the available tokens and replenishes them at a specified rate.
*   `RateLimiterDecorator`: A decorator that can be used to easily apply rate limiting to functions.
*   `RateLimiterFactory`: A factory class that creates and manages multiple named rate limiters.
