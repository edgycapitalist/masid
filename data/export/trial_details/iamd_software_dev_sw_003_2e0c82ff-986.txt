    def __init__(self):

    def apply_transformations(self, data):

            data = transformation.apply(data)
        return data

class PluginSystem:

import pandas as pd
import json
import os

class DataIngestion:
    """
    Responsible for reading CSV files and performing schema validation.
    """
    def __init__(self, schema: list):
        """
        Initializes the DataIngestion class.

        Args:
        schema (list): A list of column names to validate against.
        """
        self.schema = schema

    def read_csv(self, file_path: str) -> pd.DataFrame:
        """
        Reads a CSV file and performs schema validation.

        Args:
        file_path (str): The path to the CSV file.

        Returns:
        pd.DataFrame: The ingested data if schema validation succeeds, otherwise None.
        """
        try:
            data = pd.read_csv(file_path)
            if self.validate_schema(data):
                return data
            else:
                print("Error: Schema validation failed")
                return None
        except pd.errors.EmptyDataError:
            print("Error: Empty data")
            return None
        except pd.errors.ParserError as e:
            print(f"Error: {e}")
            return None

    def validate_schema(self, data: pd.DataFrame) -> bool:
        """
        Validates the schema of the ingested data.

        Args:
        data (pd.DataFrame): The ingested data.

        Returns:
        bool: True if schema validation succeeds, otherwise False.
        """
        for column in self.schema:
            if column not in data.columns:
                print(f"Error: Column '{column}' not found in data")
                return False
        return True

class TransformationEngine:
    """
    Applies a chain of transformations to the ingested data.
    """
    def __init__(self):
        """
        Initializes the TransformationEngine class.
        """
        self.transformations = []

    def add_transformation(self, transformation):
        """
        Adds a transformation to the chain.

        Args:
        transformation: A function that takes a pd.DataFrame as input and returns a transformed pd.DataFrame.
        """
        self.transformations.append(transformation)

    def apply_transformations(self, data: pd.DataFrame) -> pd.DataFrame:
        """
        Applies the chain of transformations to the ingested data.

        Args:
        data (pd.DataFrame): The ingested data.

        Returns:
        pd.DataFrame: The transformed data.
        """
        for transformation in self.transformations:
            data = transformation(data)
        return data

class PluginSystem:
    """
    Manages the loading and execution of custom transformations.
    """
    def __init__(self):
        """
        Initializes the PluginSystem class.
        """
        self.plugins = []

    def add_plugin(self, plugin):
        """
        Adds a plugin to the system.

        Args:
        plugin: A function that takes no arguments and returns a transformation function.
        """
        self.plugins.append(plugin)

    def load_plugins(self):
        """
        Loads all plugins in the system.
        """
        for plugin in self.plugins:
            plugin()

class DataOutput:

    Writes the transformed data to either CSV or JSON.

        """
        Initializes the DataOutput class.

    def write_data(self, data: pd.DataFrame):
        """
