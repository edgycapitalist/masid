import time
import functools

class TokenBucket:
    def __init__(self, rate, burst_size):
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()

    def add_tokens(self, num_tokens):
        self.tokens = min(self.burst_size, self.tokens + num_tokens)

    def remove_tokens(self, num_tokens):
        self.tokens -= num_tokens
        if self.tokens < 0:
            self.tokens = 0

    def get_tokens(self):
        return self.tokens

    def update(self):
        now = time.time()
        elapsed_time = now - self.last_update
        num_tokens = int(elapsed_time * self.rate)
        self.add_tokens(num_tokens)
        self.last_update = now

class RateLimiter:
    def __init__(self, rate, burst_size):
        self.rate = rate
        self.burst_size = burst_size
        self.token_bucket = TokenBucket(rate, burst_size)
        self.lock = threading.Lock()

    def allow_request(self):
        with self.lock:
            self.token_bucket.update()
            if self.token_bucket.get_tokens() >= 1:
                self.token_bucket.remove_tokens(1)
                return True
            return False

import threading
import time
import functools

class TokenBucket:
    def __init__(self, rate, burst_size):
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()

    def add_tokens(self, num_tokens):
        self.tokens = min(self.burst_size, self.tokens + num_tokens)

    def remove_tokens(self, num_tokens):
        self.tokens -= num_tokens
        if self.tokens < 0:
            self.tokens = 0

    def get_tokens(self):
        return self.tokens

    def update(self):
        now = time.time()
        elapsed_time = now - self.last_update
        num_tokens = int(elapsed_time * self.rate)
        self.add_tokens(num_tokens)
        self.last_update = now

class RateLimiter:
    def __init__(self, rate, burst_size):
        self.rate = rate
        self.burst_size = burst_size
        self.token_bucket = TokenBucket(rate, burst_size)
        self.lock = threading.Lock()

    def allow_request(self):
        with self.lock:
            self.token_bucket.update()
            if self.token_bucket.get_tokens() >= 1:
                self.token_bucket.remove_tokens(1)
                return True
            return False

    def get_rate(self):
        return self.rate

    def get_burst_size(self):
        return self.burst_size

class RateLimiterManager:
    def __init__(self):
        self.rate_limiters = {}
        self.lock = threading.Lock()

    def add_rate_limiter(self, name, rate, burst_size):
        with self.lock:
            if name in self.rate_limiters:
                raise Exception("Rate limiter already exists")
            self.rate_limiters[name] = RateLimiter(rate, burst_size)

    def get_rate_limiter(self, name):
        with self.lock:
            if name not in self.rate_limiters:
                raise Exception("Rate limiter does not exist")
            return self.rate_limiters[name]

    def remove_rate_limiter(self, name):
        with self.lock:
            if name not in self.rate_limiters:
                raise Exception("Rate limiter does not exist")
            del self.rate_limiters[name]

def rate_limiter_decorator(rate_limiter_manager, name):
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            rate_limiter = rate_limiter_manager.get_rate_limiter(name)
            if rate_limiter.allow_request():
                return func(*args, **kwargs)
            else:
                raise Exception("Rate limit exceeded")
        return wrapper
    return decorator

    def __init__(self, rate: int, burst_size: int):
        """
        Initialize the token bucket.

        Args:
        num_tokens (int): The number of tokens to add.
        """
        self.tokens = min(self.burst_size, self.tokens + num_tokens)

    def remove_tokens(self, num_tokens: int):
        """
        Remove tokens from the bucket.

        Args:
        num_tokens (int): The number of tokens to remove.
        """
        self.tokens -= num_tokens
        if self.tokens < 0:
            self.tokens = 0

        Returns:
        int: The number of tokens in the bucket.
        """
        return self.tokens
