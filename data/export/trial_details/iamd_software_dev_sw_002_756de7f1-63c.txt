import threading
import time
from functools import wraps

class RateLimiter:
    def __init__(self, rate, burst_size):
        """
        Initialize the rate limiter.

        :param rate: The rate at which tokens are added to the bucket (requests per second).
        :param burst_size: The maximum number of tokens that can be in the bucket at any given time.
        """
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()
        self.lock = threading.Lock()

    def allow_request(self):
        """
        Check if a request is allowed.

        :return: True if the request is allowed, False otherwise.
        """
        with self.lock:
            current_time = time.time()
            elapsed_time = current_time - self.last_update
            self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)
            self.last_update = current_time
            if self.tokens < 1:
                return False
            self.tokens -= 1
            return True

def rate_limited(rate, burst_size):
    """
    Decorator to apply the rate limiter to a function.

    :param rate: The rate at which tokens are added to the bucket (requests per second).
    :param burst_size: The maximum number of tokens that can be in the bucket at any given time.
    :return: The decorated function.
    """
    limiter = RateLimiter(rate, burst_size)

    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            if not limiter.allow_request():
                raise Exception("Rate limit exceeded")
            return func(*args, **kwargs)
        return wrapper
    return decorator

@rate_limited(5, 10)
def example_function():

import threading
import time
from functools import wraps

class RateLimiter:
    """
    A rate limiter class that uses the token bucket algorithm.

    Attributes:
        rate (float): The rate at which tokens are added to the bucket (requests per second).
        burst_size (int): The maximum number of tokens that can be in the bucket at any given time.
        tokens (float): The current number of tokens in the bucket.
        last_update (float): The last time the bucket was updated.
        lock (threading.Lock): A lock to prevent concurrent access to the bucket.
    """

    def __init__(self, rate, burst_size):
        """
        Initialize the rate limiter.

        Args:
            rate (float): The rate at which tokens are added to the bucket (requests per second).
            burst_size (int): The maximum number of tokens that can be in the bucket at any given time.

        Raises:
            ValueError: If the rate or burst size is not positive.
        """
        if rate <= 0 or burst_size <= 0:
            raise ValueError("Rate and burst size must be positive")
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()
        self.lock = threading.Lock()

    def allow_request(self):
        """
        Check if a request is allowed.

        Returns:
            bool: True if the request is allowed, False otherwise.
        """
        with self.lock:
            current_time = time.time()
            elapsed_time = current_time - self.last_update
            self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)
            self.last_update = current_time
            if self.tokens < 1:
                return False
            self.tokens -= 1
            return True

    def reset(self):
        """
        Reset the rate limiter to its initial state.
        """
        with self.lock:
            self.tokens = self.burst_size
            self.last_update = time.time()

def rate_limited(rate, burst_size):
    """
    Decorator to apply the rate limiter to a function.

    Args:
        rate (float): The rate at which tokens are added to the bucket (requests per second).
        burst_size (int): The maximum number of tokens that can be in the bucket at any given time.

    Returns:
        function: The decorated function.
    """
    limiter = RateLimiter(rate, burst_size)

    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            if limiter.allow_request():
                return func(*args, **kwargs)
            else:
                raise Exception("Rate limit exceeded")
        return wrapper
    return decorator

            burst_size (int): The maximum number of tokens that can be in the bucket at any given time.

        Raises:
            ValueError: If the rate or burst size is not positive.
        """
        if rate is None or burst_size is None:
            raise ValueError("Rate and burst size cannot be None")
        if rate <= 0 or burst_size <= 0:
            raise ValueError("Rate and burst size must be positive")
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()
        self.lock = threading.Lock()

    def allow_request(self):
        """
        Check if a request is allowed.

        Returns:
            bool: True if the request is allowed, False otherwise.
        """
        with self.lock:
            current_time = time.time()
            elapsed_time = current_time - self.last_update
            self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)
            self.last_update = current_time
            if self.tokens < 1:
                return False
            self.tokens -= 1
            return True

    def reset(self):
        """
        Reset the rate limiter to its initial state.
        """
        with self.lock:
            self.tokens = self.burst_size
            self.last_update = time.time()

def rate_limited(rate, burst_size):
    """
    Decorator to apply the rate limiter to a function.

    Args:
        rate (float): The rate at which tokens are added to the bucket (requests per second).
        burst_size (int): The maximum number of tokens that can be in the bucket at any given time.

    Returns:
        function: The decorated function.
    """
    if rate is None or burst_size is None:
        raise ValueError("Rate and burst size cannot be None")
    limiter = RateLimiter(rate, burst_size)

    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            if limiter.allow_request():
                return func(*args, **kwargs)
            else:
                raise Exception("Rate limit exceeded")
        return wrapper
    return decorator
