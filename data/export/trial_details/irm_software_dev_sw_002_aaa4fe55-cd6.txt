import threading
import time
from functools import wraps

class TokenBucket:
    """Implements the token bucket algorithm."""

    def __init__(self, rate, burst_size):
        """
        Initializes a token bucket with a given rate and burst size.

        Args:
            rate (int): The rate at which tokens are added to the bucket (tokens per second).
            burst_size (int): The maximum number of tokens in the bucket.
        """
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()

    def add_tokens(self, amount):
        """
        Adds tokens to the bucket.

        Args:
            amount (int): The number of tokens to add.
        """
        self.tokens = min(self.tokens + amount, self.burst_size)

    def consume_tokens(self, amount):
        """
        Consumes tokens from the bucket if available.

        Args:
            amount (int): The number of tokens to consume.

        Returns:
            bool: True if tokens were consumed, False otherwise.
        """
        if self.tokens >= amount:
            self.tokens -= amount
            return True
        return False

    def update_tokens(self):
        """
        Updates the token bucket state based on the elapsed time since the last update.
        """
        now = time.time()
        elapsed_time = now - self.last_update
        tokens_to_add = int(elapsed_time * self.rate)
        self.add_tokens(tokens_to_add)
        self.last_update = now


class RateLimiter:
    """Manages rate limiters."""

    def __init__(self, name, rate, burst_size):
        """
        Initializes a rate limiter with a given name, rate, and burst size.

        Args:
            name (str): The name of the rate limiter.
            rate (int): The rate at which tokens are added to the bucket (tokens per second).
            burst_size (int): The maximum number of tokens in the bucket.
        """
        self.name = name
        self.rate = rate
        self.burst_size = burst_size
        self.token_bucket = TokenBucket(rate, burst_size)
        self.lock = threading.Lock()

    def allow_request(self):
        """
        Checks if a request is allowed based on the current token bucket state.

        Returns:
            bool: True if the request is allowed, False otherwise.
        """
        with self.lock:
            self.token_bucket.update_tokens()
            return self.token_bucket.consume_tokens(1)


class RateLimiterDecorator:
    """Applies rate limiting to functions."""

    def __init__(self, rate_limiter):
        """
        Initializes a decorator with a given rate limiter.

        Args:
            rate_limiter (RateLimiter): The rate limiter to apply.
        """
        self.rate_limiter = rate_limiter

    def __call__(self, func):
        """
        Applies rate limiting to a function using the given rate limiter.

        Args:
            func (callable): The function to decorate.

        Returns:
            callable: The decorated function.
        """
        @wraps(func)
        def wrapper(*args, **kwargs):
            if self.rate_limiter.allow_request():
                return func(*args, **kwargs)
            else:
                raise Exception("Rate limit exceeded")
        return wrapper

import threading
import time
from functools import wraps

class TokenBucket:
    """Implements the token bucket algorithm."""

    def __init__(self, rate, burst_size):
        """
        Initializes a token bucket with a given rate and burst size.

        Args:
            rate (int): The rate at which tokens are added to the bucket (tokens per second).
            burst_size (int): The maximum number of tokens in the bucket.
        """
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()
        self.lock = threading.Lock()

    def add_tokens(self, amount):
        """
        Adds tokens to the bucket.

        Args:
            amount (int): The number of tokens to add.
        """
        with self.lock:
            self.tokens = min(self.tokens + amount, self.burst_size)

    def consume_tokens(self, amount):
        """
        Consumes tokens from the bucket if available.

        Args:
            amount (int): The number of tokens to consume.

        Returns:
            bool: True if tokens were consumed, False otherwise.
        """
        with self.lock:
            if self.tokens >= amount:
                self.tokens -= amount
                return True
            return False

    def update_tokens(self):
        """
        Updates the token bucket state based on the elapsed time since the last update.
        """
        with self.lock:
            now = time.time()
            elapsed_time = now - self.last_update
            tokens_to_add = int(elapsed_time * self.rate)
            self.add_tokens(tokens_to_add)
            self.last_update = now

import threading
import time
from functools import wraps

class TokenBucket:
    """Implements the token bucket algorithm."""

    def __init__(self, rate, burst_size):
        """
        Initializes a token bucket with a given rate and burst size.

        Args:
            rate (int): The rate at which tokens are added to the bucket (tokens per second).
            burst_size (int): The maximum number of tokens in the bucket.
        """
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()
        self.lock = threading.Lock()

    def add_tokens(self, amount):
        """
        Adds tokens to the bucket.

        Args:
            amount (int): The number of tokens to add.
        """
        with self.lock:
            self.tokens = min(self.tokens + amount, self.burst_size)

    def consume_tokens(self, amount):
        """
        Consumes tokens from the bucket if available.

        Args:
            amount (int): The number of tokens to consume.

        Returns:
            bool: True if tokens were consumed, False otherwise.
        """
        with self.lock:
            if self.tokens >= amount:
                self.tokens -= amount
                return True
            return False

    def update_tokens(self):
        """
        Updates the token bucket state based on the elapsed time since the last update.
        """
        with self.lock:
            now = time.time()
            elapsed_time = now - self.last_update
            tokens_to_add = int(elapsed_time * self.rate)
            self.add_tokens(tokens_to_add)
            self.last_update = now

    """Manages rate limiters."""
