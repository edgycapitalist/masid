import threading
import time

class TokenBucket:
    def __init__(self, rate, burst_size):
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()

    def get_tokens(self):
        return self.tokens

    def consume_tokens(self, num_tokens):
        self.tokens -= num_tokens
        if self.tokens < 0:
            self.tokens = 0

    def update(self):
        current_time = time.time()
        elapsed_time = current_time - self.last_update
        self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)
        self.last_update = current_time

class RateLimiter:
    def __init__(self, rate, burst_size):
        self.token_bucket = TokenBucket(rate, burst_size)
        self.lock = threading.Lock()

    def allow(self):
        with self.lock:
            self.token_bucket.update()
            if self.token_bucket.get_tokens() >= 1:
                self.token_bucket.consume_tokens(1)
                return True
            return False

class RateLimiterDecorator:
    def __init__(self, rate_limiter):
        self.rate_limiter = rate_limiter

    def __call__(self, func):
        def wrapper(*args, **kwargs):
            if self.rate_limiter.allow():
                return func(*args, **kwargs)
            raise Exception("Rate limit exceeded")
        return wrapper

class RateLimiterRegistry:
    def __init__(self):
        self.registry = {}
        self.lock = threading.Lock()

    def get_rate_limiter(self, name):
        with self.lock:
            return self.registry.get(name)

    def add_rate_limiter(self, name, rate_limiter):
        with self.lock:
            self.registry[name] = rate_limiter

# Example usage:
registry = RateLimiterRegistry()

registry.add_rate_limiter("example", rate_limiter)

@RateLimiterDecorator(rate_limiter)

import threading
import time
from functools import wraps

class TokenBucket:
    def __init__(self, rate, burst_size):
        """
        Initialize the token bucket with a given rate and burst size.

        :param rate: The rate at which tokens are added to the bucket (tokens per second)
        :param burst_size: The maximum number of tokens in the bucket
        """
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()

    def get_tokens(self):
        """
        Return the current number of tokens in the bucket.
        """
        return self.tokens

    def consume_tokens(self, num_tokens):
        """
        Consume a specified number of tokens from the bucket.

        :param num_tokens: The number of tokens to consume
        """
        self.tokens -= num_tokens
        if self.tokens < 0:
            self.tokens = 0

    def update(self):
        """
        Update the token bucket state based on the elapsed time since the last update.
        """
        current_time = time.time()
        elapsed_time = current_time - self.last_update
        self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)
        self.last_update = current_time

class RateLimiter:
    def __init__(self, rate, burst_size):
        """
        Initialize the rate limiter with a given rate and burst size.

        :param rate: The rate at which requests are allowed (requests per second)
        :param burst_size: The maximum number of requests allowed in a burst
        """
        self.token_bucket = TokenBucket(rate, burst_size)
        self.lock = threading.Lock()

    def allow(self):
        """
        Check if a request is allowed based on the current token bucket state.

        :return: True if the request is allowed, False otherwise
        """
        with self.lock:
            self.token_bucket.update()
            if self.token_bucket.get_tokens() >= 1:
                self.token_bucket.consume_tokens(1)
                return True
            return False

class RateLimiterDecorator:
    def __init__(self, rate_limiter):
        """
        Initialize the decorator with a given rate limiter instance.

        :param rate_limiter: The rate limiter instance to use
        """
        self.rate_limiter = rate_limiter

    def __call__(self, func):
        """
        Apply the rate limiter to a given function.

        :param func: The function to decorate
        :return: The decorated function
        """
        @wraps(func)
        def wrapper(*args, **kwargs):
            if self.rate_limiter.allow():
                return func(*args, **kwargs)
            raise Exception("Rate limit exceeded")
        return wrapper

class RateLimiterRegistry:
    def __init__(self):
        """
        Initialize the registry.
        """
        self.registry = {}
        self.lock = threading.Lock()

    def get_rate_limiter(self, name):
        """
        Return a rate limiter instance with the given name.

        :param name: The name of the rate limiter
        :return: The rate limiter instance, or None if not found
        """
        with self.lock:
            return self.registry.get(name)

    def add_rate_limiter(self, name, rate_limiter):
        """
        Add a rate limiter instance to the registry.

        :param name: The name of the rate limiter
        :param rate_limiter: The rate limiter instance to add
        """
        with self.lock:
            self.registry[name] = rate_limiter

import threading
import time

class TokenBucket:
    def __init__(self, rate, burst_size):
        if rate <= 0 or burst_size <= 0:
            raise ValueError("Rate and burst size must be positive")
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()

    def get_tokens(self):
        return self.tokens

    def consume_tokens(self, num_tokens):
        if num_tokens < 0:
            raise ValueError("Number of tokens to consume must be non-negative")
        self.tokens -= num_tokens
        if self.tokens < 0:
            self.tokens = 0

    def update(self):
        current_time = time.time()
        elapsed_time = current_time - self.last_update
        self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)
        self.last_update = current_time

class RateLimiter:
    def __init__(self, rate, burst_size):
        self.token_bucket = TokenBucket(rate, burst_size)
        self.lock = threading.Lock()

    def allow(self):
        with self.lock:
            self.token_bucket.update()
            if self.token_bucket.get_tokens() >= 1:
                self.token_bucket.consume_tokens(1)
                return True
            return False

    def reset(self):
        with self.lock:
            self.token_bucket.tokens = self.token_bucket.burst_size

class RateLimiterDecorator:
    def __init__(self, rate_limiter):
        self.rate_limiter = rate_limiter

    def __call__(self, func):
        def wrapper(*args, **kwargs):
            if self.rate_limiter.allow():
                return func(*args, **kwargs)
            raise Exception("Rate limit exceeded")
        return wrapper

class RateLimiterRegistry:
    def __init__(self):
        self.registry = {}
        self.lock = threading.Lock()

    def get_rate_limiter(self, name):

    def remove_rate_limiter(self, name):
        with self.lock:
            if name not in self.registry:
                raise ValueError("Rate limiter with this name does not exist")
            del self.registry[name]

import threading
import time

class TokenBucket:
    def __init__(self, rate, burst_size):
        if rate <= 0 or burst_size <= 0:
            raise ValueError("Rate and burst size must be positive")
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()

    def get_tokens(self):
        return self.tokens

    def consume_tokens(self, num_tokens):
        if num_tokens < 0:
            raise ValueError("Number of tokens to consume must be non-negative")
        self.tokens -= num_tokens
        if self.tokens < 0:
            self.tokens = 0

    def refill(self):
        current_time = time.time()
        elapsed_time = current_time - self.last_update
        self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)
        self.last_update = current_time

    def update(self):
        self.refill()

class RateLimiter:
    def __init__(self, rate, burst_size):
        self.token_bucket = TokenBucket(rate, burst_size)
        self.lock = threading.Lock()

    def allow(self):
        with self.lock:
            self.token_bucket.update()
            if self.token_bucket.get_tokens() >= 1:
                self.token_bucket.consume_tokens(1)
                return True
            return False

    def reset(self):
        with self.lock:
            self.token_bucket.tokens = self.token_bucket.burst_size

class RateLimiterDecorator:
    def __init__(self, rate_limiter):
        self.rate_limiter = rate_limiter

    def __call__(self, func):
        def wrapper(*args, **kwargs):
            if self.rate_limiter.allow():
                return func(*args, **kwargs)
            raise Exception("Rate limit exceeded")
        return wrapper

class RateLimiterRegistry:
    def __init__(self):
        self.registry = {}
        self.lock = threading.Lock()

    def get_rate_limiter(self, name):
        with self.lock:
            return self.registry.get(name)

    def add_rate_limiter(self, name, rate_limiter):
        with self.lock:
            if name in self.registry:
                raise ValueError("Rate limiter with this name already exists")
            self.registry[name] = rate_limiter

    def remove_rate_limiter(self, name):
        with self.lock:
            if name not in self.registry:
                raise ValueError("Rate limiter with this name does not exist")
            del self.registry[name]
