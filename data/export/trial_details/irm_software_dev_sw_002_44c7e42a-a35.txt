import threading
import time
from functools import wraps

class TokenBucket:
    def __init__(self, rate, burst_size):
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_refill = time.time()

    def consume_token(self):
        if self.tokens > 0:
            self.tokens -= 1
            return True
        return False

    def refill_bucket(self):
        current_time = time.time()
        elapsed_time = current_time - self.last_refill
        self.last_refill = current_time
        self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)

class RateLimiter:
    def __init__(self, name, rate, burst_size):
        self.name = name
        self.rate = rate
        self.burst_size = burst_size
        self.token_bucket = TokenBucket(rate, burst_size)
        self.lock = threading.Lock()

    def allow_request(self):
        with self.lock:
            self.token_bucket.refill_bucket()
            return self.token_bucket.consume_token()

    def reset(self):
        with self.lock:
            self.token_bucket.tokens = self.burst_size

class RateLimiterDecorator:
    def __init__(self, rate_limiter):
        self.rate_limiter = rate_limiter

        @wraps(func)
        def wrapper(*args, **kwargs):
            if self.rate_limiter.allow_request():
                return func(*args, **kwargs)
            else:
                raise Exception("Rate limit exceeded")
        return wrapper

class TokenBucket:

    A token bucket is a simple, widely used algorithm for rate limiting.
    It's based on the idea of a bucket that can hold a certain number of tokens,
    which are added to the bucket at a constant rate.
    """
    def __init__(self, rate, burst_size):
        """
        Initializes a new TokenBucket instance with the given rate and burst size.

        Args:
            rate (float): The rate at which tokens are added to the bucket per second.
            burst_size (int): The maximum number of tokens that the bucket can hold.
        """
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_refill = time.time()

    def consume_token(self):
        """
        Consumes a token from the bucket, returning True if the token was consumed successfully and False otherwise.

        Returns:
            bool: Whether the token was consumed successfully.
        """
        if self.tokens > 0:
            self.tokens -= 1
            return True
        return False

    def refill_bucket(self):
        """
        Refills the bucket with new tokens based on the current time and the rate.
        """
        current_time = time.time()
        elapsed_time = current_time - self.last_refill
        self.last_refill = current_time
        self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)

class RateLimiter:
    """
    A rate limiter is used to limit the rate at which certain actions can be performed.
    It uses a token bucket to manage the rate limiting.
    """
    def __init__(self, name, rate, burst_size):
        """
        Initializes a new RateLimiter instance with the given name, rate, and burst size.

        Args:
            name (str): The name of the rate limiter.
            rate (float): The rate at which tokens are added to the bucket per second.
            burst_size (int): The maximum number of tokens that the bucket can hold.

import threading
import time
from functools import wraps

class TokenBucket:
    """
    A token bucket is a simple, widely used algorithm for rate limiting.
    It's based on the idea of a bucket that can hold a certain number of tokens,
    which are added to the bucket at a constant rate.
    """
    def __init__(self, rate, burst_size):
        """
        Initializes a new TokenBucket instance with the given rate and burst size.

        Args:
            rate (float): The rate at which tokens are added to the bucket per second.
            burst_size (int): The maximum number of tokens that the bucket can hold.
        """
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_refill = time.time()

    def consume_token(self):
        """
        Consumes a token from the bucket, returning True if the token was consumed successfully and False otherwise.

        Returns:
            bool: Whether the token was consumed successfully.
        """
        if self.tokens > 0:
            self.tokens -= 1
            return True
        return False

    def refill_bucket(self):
        """
        Refills the bucket with new tokens based on the current time and the rate.
        """
        current_time = time.time()
        elapsed_time = current_time - self.last_refill
        self.last_refill = current_time
        self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)

class RateLimiter:
    """
    A rate limiter is used to limit the rate at which certain actions can be performed.
    It uses a token bucket to manage the rate limiting.
    """
    def __init__(self, name, rate, burst_size):
        """
        Initializes a new RateLimiter instance with the given name, rate, and burst size.

        Args:
            name (str): The name of the rate limiter.
            rate (float): The rate at which tokens are added to the bucket per second.
            burst_size (int): The maximum number of tokens that the bucket can hold.
        """
        self.name = name
        self.rate = rate
        self.burst_size = burst_size
        self.token_bucket = TokenBucket(rate, burst_size)
        self.lock = threading.Lock()

    def allow_request(self):
        """
        Checks if a request is allowed based on the current state of the token bucket.

        Returns:
            bool: Whether the request is allowed.
        """
        with self.lock:
            self.token_bucket.refill_bucket()
            return self.token_bucket.consume_token()

    def reset(self):
        """
        Resets the token bucket to its initial state.
        """
        with self.lock:
            self.token_bucket.tokens = self.burst_size

class RateLimiterDecorator:
    """
    A decorator that can be used to easily apply rate limiting to functions.
    """
    def __init__(self, rate_limiter):
        """
        Initializes a new RateLimiterDecorator instance with the given RateLimiter instance.

        Args:
            rate_limiter (RateLimiter): The rate limiter to use.

            func (callable): The function to decorate.

            callable: The decorated function.

import threading
import time
from functools import wraps

class TokenBucket:
    """
    A token bucket is a simple, widely used algorithm for rate limiting.
    It's based on the idea of a bucket that can hold a certain number of tokens,
    which are added to the bucket at a constant rate.
    """
    def __init__(self, rate, burst_size):
        """
        Initializes a new TokenBucket instance with the given rate and burst size.

        Args:
            rate (float): The rate at which tokens are added to the bucket per second.
            burst_size (int): The maximum number of tokens that the bucket can hold.
        """
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_refill = time.time()

    def consume_token(self):
        """
        Consumes a token from the bucket, returning True if the token was consumed successfully and False otherwise.

        Returns:
            bool: Whether the token was consumed successfully.
        """
        if self.tokens > 0:
            self.tokens -= 1
            return True
        return False

    def refill_bucket(self):
        """
        Refills the bucket with new tokens based on the current time and the rate.
        """
        current_time = time.time()
        elapsed_time = current_time - self.last_refill
        self.last_refill = current_time
        self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)

class RateLimiter:
    """
    A rate limiter is used to limit the rate at which certain actions can be performed.
    It uses a token bucket to manage the rate limiting.
    """
    def __init__(self, name, rate, burst_size):
        """
        Initializes a new RateLimiter instance with the given name, rate, and burst size.

        Args:
            name (str): The name of the rate limiter.
            rate (float): The rate at which tokens are added to the bucket per second.
            burst_size (int): The maximum number of tokens that the bucket can hold.
        """
        self.name = name
        self.rate = rate
        self.burst_size = burst_size
        self.token_bucket = TokenBucket(rate, burst_size)
        self.lock = threading.Lock()

    def allow_request(self):
        """
        Checks if a request is allowed based on the current state of the token bucket.

        Returns:
            bool: Whether the request is allowed.
        """
        with self.lock:
            self.token_bucket.refill_bucket()
            return self.token_bucket.consume_token()

    def reset(self):
        """
        Resets the token bucket to its initial state.
        """
        with self.lock:
            self.token_bucket.tokens = self.burst_size

class RateLimiterDecorator:
    """
    A decorator that can be used to easily apply rate limiting to functions.
    """
    def __init__(self, rate_limiter):
        """
        Initializes a new RateLimiterDecorator instance with the given RateLimiter instance.

        Args:
            rate_limiter (RateLimiter): The rate limiter to use.
        """
        self.rate_limiter = rate_limiter

    def __call__(self, func):
        """
        Decorates the given function with the rate limiter.

        Args:
            func (callable): The function to decorate.

        Returns:
            callable: The decorated function.

        def wrapper(*args, **kwargs):
            if self.rate_limiter.allow_request():
                return func(*args, **kwargs)
            else:
                raise Exception("Rate limit exceeded")
