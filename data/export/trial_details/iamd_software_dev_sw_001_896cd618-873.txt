### Overview

The URL shortening library provides a hash-based approach to shorten URLs and supports reverse lookup from short URL to original URL. The library is designed to be thread-safe for concurrent usage and includes input validation for valid URL format.

*   `hash_generator`: This module generates a unique hash for a given URL.
*   `collision_detector`: This module detects collisions and handles them accordingly.

The library provides the following interfaces:

*   `shorten_url(long_url: str) -> str`: This interface shortens a given long URL and returns the shortened version.
*   `lookup_short_url(short_url: str) -> str`: This interface performs a reverse lookup from a short URL to the original URL.

3.  The `hash_generator` module generates a unique hash for the given URL.
4.  The `collision_detector` module checks if the generated hash already exists. If it does, it generates a new hash until a unique one is found.
5.  The shortened URL is stored in a dictionary with the original URL as the value.
6.  The user can perform a reverse lookup using the `lookup_short_url` interface.
7.  The library returns the original URL if the short URL is found in the dictionary.

import hashlib
import threading
from urllib.parse import urlparse

class URLShortener:
    def __init__(self):
        self.url_map = {}
        self.lock = threading.Lock()

    def shorten_url(self, long_url: str) -> str:
        """
        Shorten a given long URL and return the shortened version.

        Args:
        long_url (str): The long URL to be shortened.

        Returns:
        str: The shortened URL.
        """
        if not self.is_valid_url(long_url):
            raise ValueError("Invalid URL")

        with self.lock:
            hash_object = hashlib.sha256(long_url.encode())
            short_url = hash_object.hexdigest()[:6]

            while short_url in self.url_map:
                hash_object = hashlib.sha256((long_url + str(self.url_map)).encode())
                short_url = hash_object.hexdigest()[:6]

            self.url_map[short_url] = long_url

        return f"http://short.url/{short_url}"

    def lookup_short_url(self, short_url: str) -> str:
        """
        Perform a reverse lookup from a short URL to the original URL.

        Args:
        short_url (str): The short URL to be looked up.

        Returns:
        str: The original URL if found, otherwise None.
        """
        with self.lock:
            short_url = short_url.replace("http://short.url/", "")
            return self.url_map.get(short_url)

    def is_valid_url(self, url: str) -> bool:
        """
        Check if a given URL is valid.

        Args:
        url (str): The URL to be checked.

        Returns:
        bool: True if the URL is valid, otherwise False.
        """
        try:
            result = urlparse(url)
            return all([result.scheme, result.netloc])
        except ValueError:
            return False

# Example usage:

long_url = "https://www.example.com"
short_url = url_shortener.shorten_url(long_url)
print(f"Shortened URL: {short_url}")

import hashlib

class URLShortener:
    def __init__(self):
        self.url_map = {}

    def shorten_url(self, long_url: str) -> str:
        """
        Shorten a given long URL and return the shortened version.

        Args:
        long_url (str): The long URL to be shortened.

        Returns:
        str: The shortened URL.
        """
        if long_url is None:
            raise TypeError("Input URL cannot be None")
        hash_object = hashlib.sha256(long_url.encode())
        short_url = hash_object.hexdigest()[:6]
        while short_url in self.url_map:
            hash_object = hashlib.sha256((long_url + str(self.url_map)).encode())
            short_url = hash_object.hexdigest()[:6]
        self.url_map[short_url] = long_url
        return f"http://short.url/{short_url}"

    def lookup_short_url(self, short_url: str) -> str:
        """
        Perform a reverse lookup from a short URL to the original URL.

        Args:
        short_url (str): The short URL to be looked up.

        Returns:
        str: The original URL if found, otherwise None.
        """
        if short_url is None:
            raise TypeError("Input short URL cannot be None")
        short_url = short_url.replace("http://short.url/", "")
        return self.url_map.get(short_url)
