## Overview

The rate limiter module is designed to provide a flexible and thread-safe way to limit the rate at which functions can be executed. The module uses the token bucket algorithm to manage the rate limiting process.

## Components

*   **RateLimiter**: This is the core class that implements the token bucket algorithm. It has methods to consume tokens and check if a request can be allowed.
*   **RateLimiterFactory**: This class is responsible for creating and managing multiple named rate limiters.
*   **rate_limited**: This is a decorator that can be used to apply rate limiting to functions.

## Interfaces

*   **RateLimiter**:
    *   `__init__(rate, burst_size)`: Initializes the rate limiter with the given rate and burst size.
    *   `consume()`: Consumes a token from the bucket.
    *   `allow_request()`: Checks if a request can be allowed based on the current state of the bucket.
*   **RateLimiterFactory**:
    *   `create_rate_limiter(name, rate, burst_size)`: Creates a new rate limiter with the given name, rate, and burst size.
    *   `get_rate_limiter(name)`: Returns the rate limiter with the given name.
*   **rate_limited**:
    *   `__call__(func)`: Applies rate limiting to the given function.

## Data Flow

1.  The `RateLimiterFactory` class is used to create and manage multiple named rate limiters.
2.  The `rate_limited` decorator is used to apply rate limiting to functions.
3.  When a function is called with the `rate_limited` decorator, it checks with the corresponding rate limiter if the request can be allowed.
4.  If the request can be allowed, the function is executed and a token is consumed from the bucket.
5.  If the request cannot be allowed, the function is not executed and an exception is raised.

import threading
import time
from functools import wraps

class RateLimiter:
    def __init__(self, rate, burst_size):
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()
        self.lock = threading.Lock()

    def consume(self):
        with self.lock:
            current_time = time.time()
            elapsed_time = current_time - self.last_update
            self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)
            self.last_update = current_time
            if self.tokens < 1:
                return False
            self.tokens -= 1
            return True

    def allow_request(self):
        return self.consume()

class RateLimiterFactory:
    def __init__(self):
        self.rate_limiters = {}
        self.lock = threading.Lock()

    def create_rate_limiter(self, name, rate, burst_size):
        with self.lock:
            if name in self.rate_limiters:
                raise ValueError("Rate limiter with the given name already exists")
            self.rate_limiters[name] = RateLimiter(rate, burst_size)

    def get_rate_limiter(self, name):
        with self.lock:
            return self.rate_limiters.get(name)

def rate_limited(rate_limiter_factory, name):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            rate_limiter = rate_limiter_factory.get_rate_limiter(name)
            if rate_limiter is None:
                raise ValueError("Rate limiter with the given name does not exist")
            if not rate_limiter.allow_request():
                raise Exception("Rate limit exceeded")
            return func(*args, **kwargs)
        return wrapper
    return decorator

import time
from functools import wraps

class RateLimiter:
    def __init__(self, rate, burst_size):
        """
        Initializes the rate limiter with the given rate and burst size.

        Args:
            rate (int): The rate at which tokens are added to the bucket per second.
            burst_size (int): The maximum number of tokens in the bucket.
        """
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()
        self.lock = threading.Lock()

    def consume(self):
        """
        Consumes a token from the bucket.

        Returns:
            bool: True if a token was consumed, False otherwise.
        """
        with self.lock:
            current_time = time.time()
            elapsed_time = current_time - self.last_update
            self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)
            self.last_update = current_time
            if self.tokens < 1:
                return False
            self.tokens -= 1
            return True

    def allow_request(self):
        """
        Checks if a request can be allowed based on the current state of the bucket.

        Returns:
            bool: True if the request can be allowed, False otherwise.
        """
        return self.consume()

class RateLimiterFactory:
    def __init__(self):
        """
        Initializes the rate limiter factory.
        """
        self.rate_limiters = {}
        self.lock = threading.Lock()

    def create_rate_limiter(self, name, rate, burst_size):
        """
        Creates a new rate limiter with the given name, rate, and burst size.

        Args:
            name (str): The name of the rate limiter.
            rate (int): The rate at which tokens are added to the bucket per second.
            burst_size (int): The maximum number of tokens in the bucket.

        with self.lock:
            if name in self.rate_limiters:
                raise ValueError("Rate limiter with the given name already exists")
            self.rate_limiters[name] = RateLimiter(rate, burst_size)

        Returns the rate limiter with the given name.

        Args:
            name (str): The name of the rate limiter.

        Returns:

import threading
import time
from functools import wraps

class RateLimiter:
    def __init__(self, rate, burst_size):
        """
        Initializes the rate limiter with the given rate and burst size.

        Args:
            rate (int): The rate at which tokens are added to the bucket per second.
            burst_size (int): The maximum number of tokens in the bucket.
        """
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()
        self.lock = threading.Lock()

    def consume(self):
        """
        Consumes a token from the bucket.

        Returns:
            bool: True if a token was consumed, False otherwise.
        """
        with self.lock:
            current_time = time.time()
            elapsed_time = current_time - self.last_update
            self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)
            self.last_update = current_time
            if self.tokens < 1:
                return False
            self.tokens -= 1
            return True

    def allow_request(self):
        """
        Checks if a request can be allowed based on the current state of the bucket.

        Returns:
            bool: True if the request can be allowed, False otherwise.
        """
        return self.consume()

        """
        Initializes the rate limiter factory.
        """
        self.rate_limiters = {}
        self.lock = threading.Lock()

        """
        Creates a new rate limiter with the given name, rate, and burst size.

        Args:
            name (str): The name of the rate limiter.
            rate (int): The rate at which tokens are added to the bucket per second.
            burst_size (int): The maximum number of tokens in the bucket.

        with self.lock:
            if name in self.rate_limiters:
                raise ValueError("Rate limiter with the given name already exists")
            self.rate_limiters[name] = RateLimiter(rate, burst_size)

        Args:
            name (str): The name of the rate limiter.

        Returns:

*   **RateLimiter**: This is the core class that implements the token bucket algorithm. It has methods to consume tokens and check if a request can be allowed.
*   **RateLimiterFactory**: This class is responsible for creating and managing multiple named rate limiters.
*   **rate_limited**: This is a decorator that can be used to apply rate limiting to functions.

## Interfaces

*   **RateLimiter**:
    *   `__init__(rate, burst_size)`: Initializes the rate limiter with the given rate and burst size.
    *   `consume()`: Consumes a token from the bucket.
    *   `allow_request()`: Checks if a request can be allowed based on the current state of the bucket.
    *   `get_tokens()`: Returns the current number of tokens in the bucket.
*   **RateLimiterFactory**:
    *   `create_rate_limiter(name, rate, burst_size)`: Creates a new rate limiter with the given name, rate, and burst size.
    *   `get_rate_limiter(name)`: Returns the rate limiter with the given name.
    *   `delete_rate_limiter(name)`: Deletes the rate limiter with the given name.
    *   `get_rate_limiters()`: Returns a list of all the rate limiters in the factory.
*   **rate_limited**:
    *   `__call__(func)`: Applies rate limiting to the given function.

## Data Flow

1.  The `RateLimiterFactory` class is used to create and manage multiple named rate limiters.
2.  The `rate_limited` decorator is used to apply rate limiting to functions.
3.  When a function is called with the `rate_limited` decorator, it checks with the corresponding rate limiter if the request can be allowed.
4.  If the request can be allowed, the function is executed and a token is consumed from the bucket.
5.  If the request cannot be allowed, the function is not executed and an exception is raised.

## Technology Choices

*   Python 3.7 or later is required for the implementation.
*   The `threading` module is used to ensure thread-safe operation.

```python
import threading
import time
from functools import wraps

class RateLimiter:
    def __init__(self, rate, burst_size):
        """
        Initializes the rate limiter with the given rate and burst size.

        Args:
            rate (int): The rate at which tokens are added to the bucket per second.
            burst_size (int): The maximum number of tokens in the bucket.
        """
        if rate <= 0:
            raise ValueError("Rate must be greater than zero")
        if burst_size <= 0:
            raise ValueError("Burst size must be greater than zero")
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()
        self.lock = threading.Lock()

    def consume(self):
        """
        Consumes a token from the bucket.

        Returns:
            bool: True if a token was consumed, False otherwise.
        """
        with self.lock:
            current_time = time.time()
            elapsed_time = current_time - self.last_update
            self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)
            self.last_update = current_time
            if self.tokens < 1:
                return False
            self.tokens -= 1
            return True

    def allow_request(self):
        """
        Checks if a request can be allowed based on the current state of the bucket.

        Returns:
            bool: True if the request can be allowed, False otherwise.
        """
        return self.consume()

    def get_tokens(self):
        """
        Returns the current number of tokens in the bucket.

        Returns:
            int: The current number of tokens in the bucket.
        """
        with self.lock:
            return self.tokens

class RateLimiterFactory:
    def __init__(self):
        """
        Initializes the rate limiter factory.
        """
        self.rate_limiters = {}
        self.lock = threading.Lock()

    def create_rate_limiter(self, name, rate, burst_size):
        """
        Creates a new rate limiter with the given name, rate, and burst size.

        Args:
            name (str): The name of the rate limiter.
            rate (int): The rate at which tokens are added to the bucket per second.
            burst_size (int): The maximum number of tokens in the bucket.
        """
        with self.lock:
            if name in self.rate_limiters:
                raise ValueError("Rate limiter with the given name already exists")
            self.rate_limiters[name] = RateLimiter(rate, burst_size)

    def get_rate_limiter(self, name):
        """
        Returns the rate limiter with the given name.

        Args:
            name (str): The name of the rate limiter.

        Returns:
            RateLimiter: The rate limiter with the given name, or None if it does not exist.
        """
        with self.lock:
            return self.rate_limiters.get(name)

    def delete_rate_limiter(self, name):
        """
        Deletes the rate limiter with the given name.

        Args:
            name (str): The name of the rate limiter.

        Returns:
            bool: True if the rate limiter was deleted, False otherwise.
        """
        with self.lock:
            if name in self.rate_limiters:
                del self.rate_limiters[name]
                return True
            return False

    def get_rate_limiters(self):
        """
        Returns a list of all the rate limiters in the factory.

        Returns:
            list: A list of all the rate limiters in the factory.
        """
        with self.lock:
            return list(self.rate_limiters.values())

def rate_limited(rate_limiter_factory, name):
    """
    Applies rate limiting to the given function.

    Args:
        rate_limiter_factory (RateLimiterFactory): The rate limiter factory.
        name (str): The name of the rate limiter.

    Returns:
        callable: The decorated function.
    """
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            rate_limiter = rate_limiter_factory.get_rate_limiter(name)
            if rate_limiter is None:
                raise ValueError("Rate limiter with the given name does not exist")
            if not rate_limiter.allow_request():
                raise Exception("Rate limit exceeded")
            return func(*args, **kwargs)
        return wrapper
    return decorator

# Example usage:
rate_limiter_factory = RateLimiterFactory()
rate_limiter_factory.create_rate_limiter("my_rate_limiter", 5, 10)

@rate_limited(rate_limiter_factory, "my_rate_limiter")
def my_function():
    print("Hello, world!")

my_function()
