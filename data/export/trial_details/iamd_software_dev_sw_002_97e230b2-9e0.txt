import threading
import time
from functools import wraps

class TokenBucket:
    def __init__(self, rate, burst_size):
        """
        Initializes the token bucket with a rate and burst size.

        Args:
            rate (float): The rate at which tokens are added to the bucket per second.
            burst_size (int): The maximum number of tokens in the bucket.
        """
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_refill = time.time()

    def consume_tokens(self, amount):
        """
        Consumes a specified amount of tokens from the bucket.

        Args:
            amount (int): The number of tokens to consume.

        Returns:
            bool: True if the tokens are available, False otherwise.
        """
        if self.tokens >= amount:
            self.tokens -= amount
            return True
        return False

    def refill_bucket(self):
        """
        Refills the bucket with tokens at the specified rate.
        """
        current_time = time.time()
        elapsed_time = current_time - self.last_refill
        self.last_refill = current_time
        self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)


class RateLimiter:
    def __init__(self, name, rate, burst_size):
        """
        Initializes the rate limiter with a name, rate, and burst size.

        Args:
            name (str): The name of the rate limiter.
            rate (float): The rate at which requests are allowed per second.
            burst_size (int): The maximum number of requests allowed in a burst.
        """
        self.name = name
        self.rate = rate
        self.burst_size = burst_size
        self.token_bucket = TokenBucket(rate, burst_size)
        self.lock = threading.Lock()

    def allow_request(self):
        """
        Checks if a request is allowed based on the current token count.

        Returns:
            bool: True if the request is allowed, False otherwise.
        """
        with self.lock:
            self.token_bucket.refill_bucket()
            return self.token_bucket.consume_tokens(1)

    def get_token_count(self):
        """
        Returns the current token count.

        Returns:
            int: The current token count.
        """
        with self.lock:
            return self.token_bucket.tokens


def rate_limiter_decorator(rate_limiter):
    """
    Applies rate limiting to a function.

    Args:
        rate_limiter (RateLimiter): The rate limiter instance.

    Returns:
        function: The decorated function.
    """
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            if rate_limiter.allow_request():
                return func(*args, **kwargs)
            else:
                raise Exception("Rate limit exceeded")
        return wrapper
    return decorator


# Example usage:
if __name__ == "__main__":
    rate_limiter = RateLimiter("example", 5, 10)

    @rate_limiter_decorator(rate_limiter)
    def example_function():
        print("Hello, world!")

    for _ in range(20):
        try:
            example_function()
        except Exception as e:
            print(e)
        time.sleep(0.1)

from functools import wraps

class TokenBucket:
    def __init__(self, rate, burst_size):
        """
        Initializes the token bucket with a rate and burst size.

        Args:
            rate (float): The rate at which tokens are added to the bucket per second.
            burst_size (int): The maximum number of tokens in the bucket.

        Raises:
            ValueError: If the rate or burst size is less than or equal to zero.
        """
        if rate <= 0 or burst_size <= 0:
            raise ValueError("Rate and burst size must be greater than zero")
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_refill = time.time()

    def consume_tokens(self, amount):
        """
        Consumes a specified amount of tokens from the bucket.

        Args:
            amount (int): The number of tokens to consume.

        Returns:
            bool: True if the tokens are available, False otherwise.
        """
        if amount <= 0:
            raise ValueError("Amount must be greater than zero")
        if self.tokens >= amount:
            self.tokens -= amount
            return True
        return False

    def refill_bucket(self):
        """
        Refills the bucket with tokens at the specified rate.
        """
        current_time = time.time()
        elapsed_time = current_time - self.last_refill
        self.last_refill = current_time
        self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)

class TokenBucket:
    def __init__(self, rate, burst_size):
        """
        Initializes the token bucket with a rate and burst size.

        Args:
            rate (float): The rate at which tokens are added to the bucket per second.
            burst_size (int): The maximum number of tokens in the bucket.

        Raises:
            ValueError: If the rate or burst size is less than or equal to zero.
        """
        if rate <= 0 or burst_size <= 0:
            raise ValueError("Rate and burst size must be greater than zero")
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_refill = time.time()

    def consume_tokens(self, amount):
        """
        Consumes a specified amount of tokens from the bucket.

        Args:
            amount (int): The number of tokens to consume.

        Returns:
            bool: True if the tokens are available, False otherwise.
        """
        if amount <= 0:
            raise ValueError("Amount must be greater than zero")
        if self.tokens >= amount:
            self.tokens -= amount
            return True
        return False

    def refill_bucket(self):
        """
        Refills the bucket with tokens at the specified rate.
        """
        current_time = time.time()
        elapsed_time = current_time - self.last_refill
        self.last_refill = current_time
        self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)
