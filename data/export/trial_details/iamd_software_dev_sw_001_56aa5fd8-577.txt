### Implementation Code
```python
import hashlib
import threading
from urllib.parse import urlparse

class URLShortener:
    def __init__(self):
        self.url_map = {}
        self.lock = threading.Lock()

    def shorten_url(self, long_url):
        if not self.is_valid_url(long_url):
            raise ValueError("Invalid URL format")

        with self.lock:
            hash_object = hashlib.sha256(long_url.encode())
            short_url = hash_object.hexdigest()[:6]

            while short_url in self.url_map:
                hash_object = hashlib.sha256((long_url + str(hash_object.hexdigest())).encode())
                short_url = hash_object.hexdigest()[:6]

            self.url_map[short_url] = long_url
            return f"http://short.url/{short_url}"

    def get_original_url(self, short_url):
        with self.lock:
            short_url = short_url.split("/")[-1]
            return self.url_map.get(short_url)

    def is_valid_url(self, url):
        try:
            result = urlparse(url)
            return all([result.scheme, result.netloc])
        except ValueError:
            return False

# Example usage
url_shortener = URLShortener()

```python
import hashlib
import threading
from urllib.parse import urlparse

class URLShortener:
    def __init__(self):
        self.url_map = {}
        self.lock = threading.Lock()

    def shorten_url(self, long_url):
        if not self.is_valid_url(long_url):
            raise ValueError("Invalid URL format")

        with self.lock:
            hash_object = hashlib.sha256(long_url.encode())
            short_url = hash_object.hexdigest()[:6]

            while short_url in self.url_map:
                hash_object = hashlib.sha256((long_url + str(hash_object.hexdigest())).encode())
                short_url = hash_object.hexdigest()[:6]

            self.url_map[short_url] = long_url
            return f"http://short.url/{short_url}"

    def get_original_url(self, short_url):
        with self.lock:
            short_url = short_url.split("/")[-1]
            return self.url_map.get(short_url)

    def is_valid_url(self, url):
        try:
            result = urlparse(url)
            return all([result.scheme, result.netloc])
        except ValueError:
            return False

import hashlib
import threading
from urllib.parse import urlparse

class URLShortener:
    def __init__(self):
        self.url_map = {}
        self.lock = threading.Lock()

    def shorten_url(self, long_url):
        """
        Shortens a given long URL and returns the shortened version.

        Args:
            long_url (str): The long URL to be shortened.

        Returns:
            str: The shortened URL.

        Raises:
            ValueError: If the input URL is invalid.
        """
        if not self.is_valid_url(long_url):
            raise ValueError("Invalid URL format")

        with self.lock:
            hash_object = hashlib.sha256(long_url.encode())
            short_url = hash_object.hexdigest()[:6]

            while short_url in self.url_map:
                hash_object = hashlib.sha256((long_url + str(hash_object.hexdigest())).encode())
                short_url = hash_object.hexdigest()[:6]

            self.url_map[short_url] = long_url
            return f"http://short.url/{short_url}"

    def get_original_url(self, short_url):
        """
        Returns the original URL for a given shortened URL.

        Args:
            short_url (str): The shortened URL.

        Returns:
            str: The original URL, or None if not found.
        """
        with self.lock:
            short_url = short_url.split("/")[-1]
            return self.url_map.get(short_url)

    def is_valid_url(self, url):
        """
        Checks if a given URL is valid.

        Args:
            url (str): The URL to be checked.

        Returns:
            bool: True if the URL is valid, False otherwise.
        """
        try:
            result = urlparse(url)
            return all([result.scheme, result.netloc])
        except ValueError:
            return False
