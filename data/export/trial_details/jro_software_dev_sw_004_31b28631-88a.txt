import queue
import time
from typing import Dict, Callable

class Task:
    def __init__(self, id: int, priority: int, payload: str, max_retries: int = 3):
        self.id = id
        self.priority = priority
        self.payload = payload
        self.max_retries = max_retries
        self.retries = 0
        self.status = "pending"

class TaskQueue:
    def __init__(self):
        self.queue = queue.PriorityQueue()
        self.tasks: Dict[int, Task] = {}

    def add_task(self, task: Task):
        self.queue.put((task.priority, task.id))
        self.tasks[task.id] = task

    def get_task(self) -> Task:
        if self.queue.empty():
            return None
        _, task_id = self.queue.get()
        return self.tasks[task_id]

    def update_task_status(self, task_id: int, status: str):
        if task_id in self.tasks:
            self.tasks[task_id].status = status

class Worker(threading.Thread):
    def __init__(self, task_queue: TaskQueue, task_status_tracker: 'TaskStatusTracker'):
        super().__init__()
        self.task_queue = task_queue
        self.task_status_tracker = task_status_tracker

    def run(self):
        while True:
            task = self.task_queue.get_task()
            if task is None:
                break
            try:
                # Execute task
                print(f"Executing task {task.id} with payload {task.payload}")
                time.sleep(1)  # Simulate task execution
                self.task_queue.update_task_status(task.id, "completed")
                self.task_status_tracker.update_task_status(task.id, "completed")
            except Exception as e:
                self.task_queue.update_task_status(task.id, "failed")
                self.task_status_tracker.update_task_status(task.id, "failed")
                if task.retries < task.max_retries:
                    task.retries += 1
                    self.task_queue.add_task(task)
                else:
                    print(f"Task {task.id} failed after {task.max_retries} retries")

class TaskStatusTracker:
    def __init__(self):
        self.tasks: Dict[int, str] = {}

    def update_task_status(self, task_id: int, status: str):
        self.tasks[task_id] = status

    def get_task_status(self, task_id: int) -> str:

import threading
import queue
import time
from typing import Dict, Callable
import logging

# Set up logging

class Task:
    def __init__(self, id: int, priority: int, payload: str, max_retries: int = 3):
        self.id = id
        self.priority = priority
        self.payload = payload
        self.max_retries = max_retries
        self.retries = 0
        self.status = "pending"

class TaskQueue:
    def __init__(self):
        self.queue = queue.PriorityQueue()
        self.tasks: Dict[int, Task] = {}
        self.lock = threading.Lock()

    def add_task(self, task: Task):
        with self.lock:
            self.queue.put((task.priority, task.id))
            self.tasks[task.id] = task

    def get_task(self) -> Task:
        with self.lock:
            if self.queue.empty():
                return None
            _, task_id = self.queue.get()
            return self.tasks[task_id]

    def update_task_status(self, task_id: int, status: str):
        with self.lock:
            if task_id in self.tasks:
                self.tasks[task_id].status = status

    def remove_task(self, task_id: int):
        with self.lock:
            if task_id in self.tasks:
                del self.tasks[task_id]

class Worker(threading.Thread):
    def __init__(self, task_queue: TaskQueue, task_status_tracker: 'TaskStatusTracker'):
        super().__init__()
        self.task_queue = task_queue
        self.task_status_tracker = task_status_tracker

    def run(self):
        while True:
            task = self.task_queue.get_task()
            if task is None:
                break
            try:
                # Execute task
                logging.info(f"Executing task {task.id} with payload {task.payload}")
                time.sleep(1)  # Simulate task execution
                self.task_queue.update_task_status(task.id, "completed")
                self.task_status_tracker.update_task_status(task.id, "completed")
            except Exception as e:
                logging.error(f"Error executing task {task.id}: {e}")
                self.task_queue.update_task_status(task.id, "failed")
                self.task_status_tracker.update_task_status(task.id, "failed")
                if task.retries < task.max_retries:
                    task.retries += 1
                    self.task_queue.add_task(task)
                else:
                    logging.error(f"Task {task.id} failed after {task.max_retries} retries")
                    self.task_queue.remove_task(task.id)

    def __init__(self):
        self.tasks: Dict[int, str] = {}
        self.lock = threading.Lock()

    def update_task_status(self, task_id: int, status: str):
        with self.lock:
            self.tasks[task_id] = status

            return True
        return False

    def __init__(self, task_queue: TaskQueue, workers: list):
        self.task_queue = task_queue
        self.workers = workers

    def shutdown(self):
        for worker in self.workers:
            worker.join()
        self.task_queue.queue.join()

    task_queue = TaskQueue()
    task_status_tracker = TaskStatusTracker()
    retry_mechanism = RetryMechanism(3)
    shutdown_manager = ShutdownManager(task_queue, [])

    # Create tasks
    task1 = Task(1, 1, "Task 1")
    task2 = Task(2, 2, "Task 2")

    task3 = Task(3, 3, "Task 3")

    # Add tasks to task queue
    task_queue.add_task(task1)
    task_queue.add_task(task2)
    task_queue.add_task(task3)

    # Create workers
    worker1 = Worker(task_queue, task_status_tracker)
    worker2 = Worker(task_queue, task_status_tracker)
    worker3 = Worker(task_queue, task_status_tracker)

    # Start workers
    worker1.start()
    worker2.start()
    worker3.start()

    # Add workers to shutdown manager
    shutdown_manager.workers = [worker1, worker2, worker3]

    # Shutdown
    time.sleep(5)  # Let workers run for a bit
    shutdown_manager.shutdown()

if __name__ == "__main__":
    main()
