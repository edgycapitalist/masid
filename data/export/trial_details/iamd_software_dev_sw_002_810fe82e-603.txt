import threading
import time
from functools import wraps

class RateLimiter:
    def __init__(self, rate, burst_size):
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()
        self.lock = threading.Lock()

    def allow_request(self):
        with self.lock:
            now = time.time()
            elapsed = now - self.last_update
            self.last_update = now
            self.tokens = min(self.burst_size, self.tokens + elapsed * self.rate)
            if self.tokens < 1:
                return False
            self.tokens -= 1
            return True

    def reset(self):
        with self.lock:
            self.tokens = self.burst_size
            self.last_update = time.time()

import threading
import time
from functools import wraps

class RateLimiter:
    """
    A rate limiter class that implements the token bucket algorithm.

    Args:
        rate (float): The rate at which tokens are added to the bucket (requests per second).
        burst_size (int): The maximum number of tokens in the bucket.

    Attributes:
        rate (float): The rate at which tokens are added to the bucket.
        burst_size (int): The maximum number of tokens in the bucket.
        tokens (int): The current number of tokens in the bucket.
        last_update (float): The last time the bucket was updated.
        lock (threading.Lock): A lock to ensure thread-safe operation.
    """

    def __init__(self, rate, burst_size):
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()
        self.lock = threading.Lock()

    def allow_request(self):
        """
        Check if a request is allowed based on the token bucket algorithm.

        Returns:
            bool: True if the request is allowed, False otherwise.
        """
        with self.lock:
            now = time.time()
            elapsed = now - self.last_update
            self.last_update = now
            self.tokens = min(self.burst_size, self.tokens + elapsed * self.rate)
            if self.tokens < 1:
                return False
            self.tokens -= 1
            return True

    def reset(self):
        """
        Reset the rate limiter.

        This method resets the token bucket to its initial state.
        """
        with self.lock:
            self.tokens = self.burst_size
            self.last_update = time.time()

class NamedRateLimiter:
    """
    A named rate limiter class that extends the RateLimiter class.

    Args:
        name (str): The name of the rate limiter.
        rate (float): The rate at which tokens are added to the bucket (requests per second).
        burst_size (int): The maximum number of tokens in the bucket.

    Attributes:
        name (str): The name of the rate limiter.
        rate_limiter (RateLimiter): The underlying rate limiter.
    """

    def __init__(self, name, rate, burst_size):
        self.name = name
        self.rate_limiter = RateLimiter(rate, burst_size)

    def allow_request(self):
        """
        Check if a request is allowed based on the token bucket algorithm.

        Returns:
            bool: True if the request is allowed, False otherwise.
        """
        return self.rate_limiter.allow_request()

    def reset(self):
        """
        Reset the named rate limiter.

        This method resets the underlying rate limiter.
        """
        self.rate_limiter.reset()

    A decorator that applies rate limiting to a function.

    Args:

        self.tokens = burst_size

        self.lock = threading.Lock()

        Returns:

            elapsed = now - self.last_update
            self.last_update = now

            if self.tokens < 1:
                return False
            self.tokens -= 1
            return True
