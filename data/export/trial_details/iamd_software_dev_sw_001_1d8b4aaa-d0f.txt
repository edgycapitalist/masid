*   **Collision Detector**: This component will detect and handle collisions that occur when two different URLs are shortened to the same hash value.
*   **Input Validator**: This component will validate the input URL to ensure it is in a valid format.
*   **Thread Safety Manager**: This component will ensure that the library is thread-safe for concurrent usage.

```python
import hashlib
import threading
from urllib.parse import urlparse

class URLShortener:
    def __init__(self):
        self.url_mapper = {}
        self.lock = threading.Lock()

    def shorten_url(self, long_url: str) -> str:
        """
        Shorten a long URL using a hash-based approach.
        
        Args:
        long_url (str): The long URL to be shortened.
        
        Returns:
        str: The shortened URL.
        """
        if not self._validate_url(long_url):
            raise ValueError("Invalid URL format")

        with self.lock:
            short_url = self._generate_short_url(long_url)
            while self._detect_collision(short_url):
                short_url = self._resolve_collision(short_url)

            self.url_mapper[short_url] = long_url
            return short_url

    def get_long_url(self, short_url: str) -> str:
        """
        Retrieve the original URL for a given shortened URL.
        
        Args:
        short_url (str): The shortened URL.
        
        Returns:
        str: The original URL.
        """
        with self.lock:
            return self.url_mapper.get(short_url)

    def _validate_url(self, url: str) -> bool:
        """
        Validate if the input URL is in a valid format.
        
        Args:
        url (str): The URL to be validated.
        
        Returns:
        bool: True if the URL is valid, False otherwise.
        """
        try:
            result = urlparse(url)
            return all([result.scheme, result.netloc])
        except ValueError:
            return False

    def _generate_short_url(self, long_url: str) -> str:
        """
        Generate a shortened URL using a hash-based approach.
        
        Args:
        long_url (str): The long URL to be shortened.
        
        Returns:
        str: The shortened URL.
        """
        return hashlib.sha256(long_url.encode()).hexdigest()[:6]

    def _detect_collision(self, short_url: str) -> bool:
        """
        Detect if a collision has occurred for a given shortened URL.
        
        Args:
        short_url (str): The shortened URL.
        
        Returns:
        bool: True if a collision has occurred, False otherwise.
        """
        return short_url in self.url_mapper

    def _resolve_collision(self, short_url: str) -> str:
        """
        Resolve a collision by generating a new shortened URL.
        
        Args:
        short_url (str): The shortened URL that caused the collision.
        
        Returns:
        str: The new shortened URL.
        """
        return hashlib.sha256((short_url + "1").encode()).hexdigest()[:6]
```

```python
import hashlib
import threading
from urllib.parse import urlparse

class URLShortener:
    def __init__(self):
        self.url_mapper = {}
        self.lock = threading.Lock()
        self.counter = 0
        self.base_url = "http://short.url/"

    def shorten_url(self, long_url: str) -> str:
        """
        Shorten a long URL using a hash-based approach.
        
        Args:
        long_url (str): The long URL to be shortened.
        
        Returns:
        str: The shortened URL.
        """
        if not long_url:
            raise ValueError("URL cannot be empty")
        if not self._validate_url(long_url):
            raise ValueError("Invalid URL format")

        with self.lock:
            short_url = self._generate_short_url(long_url)
            while self._detect_collision(short_url):
                short_url = self._resolve_collision(short_url)

            self.url_mapper[short_url] = long_url
            return self.base_url + short_url

    def get_long_url(self, short_url: str) -> str:
        """
        Retrieve the original URL for a given shortened URL.
        
        Args:
        short_url (str): The shortened URL.
        
        Returns:
        str: The original URL.
        """
        with self.lock:
            short_url = short_url.replace(self.base_url, "")
            return self.url_mapper.get(short_url)

    def _validate_url(self, url: str) -> bool:
        """
        Validate if the input URL is in a valid format.
        
        Args:
        url (str): The URL to be validated.
        
        Returns:
        bool: True if the URL is valid, False otherwise.
        """
        try:
            result = urlparse(url)
            return all([result.scheme, result.netloc])
        except ValueError:
            return False

    def _generate_short_url(self, long_url: str) -> str:
        """
        Generate a shortened URL using a hash-based approach.
        
        Args:
        long_url (str): The long URL to be shortened.
        
        Returns:
        str: The shortened URL.
        """
        return hashlib.sha256(long_url.encode()).hexdigest()[:6]

    def _detect_collision(self, short_url: str) -> bool:
        """
        Detect if a collision has occurred for a given shortened URL.
        
        Args:
        short_url (str): The shortened URL.
        
        Returns:
        bool: True if a collision has occurred, False otherwise.
        """
        return short_url in self.url_mapper

    def _resolve_collision(self, short_url: str) -> str:
        """
        Resolve a collision by generating a new shortened URL.
        
        Args:
        short_url (str): The shortened URL that caused the collision.
        
        Returns:
        str: The new shortened URL.
        """
        self.counter += 1
        return f"{short_url}{self.counter}"
```
