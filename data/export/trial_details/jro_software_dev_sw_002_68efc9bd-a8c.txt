import threading
import time
from functools import wraps

class RateLimiter:
    def __init__(self, rate, burst_size):
        """
        Initializes the rate limiter with the specified rate and burst size.

        Args:
            rate (int): The rate at which tokens are added to the bucket per second.
            burst_size (int): The maximum number of tokens in the bucket.
        """
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()
        self.lock = threading.Lock()

    def allow(self):
        """
        Checks if a request is allowed based on the current token count.

        Returns:
            bool: True if the request is allowed, False otherwise.
        """
        with self.lock:
            current_time = time.time()
            elapsed_time = current_time - self.last_update
            self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)
            self.last_update = current_time
            return self.tokens > 0

    def consume(self):
        """
        Consumes a token from the bucket.
        """
        with self.lock:
            if self.allow():
                self.tokens -= 1

class NamedRateLimiter:
    def __init__(self):
        """
        Initializes the named rate limiter.
        """
        self.rate_limiters = {}
        self.lock = threading.Lock()

    def get_rate_limiter(self, name):
        """
        Returns a rate limiter with the specified name.

        Args:
            name (str): The name of the rate limiter.

        Returns:
            RateLimiter: The rate limiter with the specified name.
        """
        with self.lock:
            return self.rate_limiters.get(name)

    def create_rate_limiter(self, name, rate, burst_size):
        """
        Creates a new rate limiter with the specified name, rate, and burst size.

        Args:
            name (str): The name of the rate limiter.
            rate (int): The rate at which tokens are added to the bucket per second.
            burst_size (int): The maximum number of tokens in the bucket.
        """
        with self.lock:
            if name not in self.rate_limiters:
                self.rate_limiters[name] = RateLimiter(rate, burst_size)

def rate_limited(rate_limiter_name, rate, burst_size):
    """
    Applies rate limiting to the specified function.

    Args:
        rate_limiter_name (str): The name of the rate limiter.
        rate (int): The rate at which tokens are added to the bucket per second.
        burst_size (int): The maximum number of tokens in the bucket.

    Returns:
        function: The decorated function.
    """
    named_rate_limiter = NamedRateLimiter()
    rate_limiter = named_rate_limiter.get_rate_limiter(rate_limiter_name)
    if rate_limiter is None:
        named_rate_limiter.create_rate_limiter(rate_limiter_name, rate, burst_size)
        rate_limiter = named_rate_limiter.get_rate_limiter(rate_limiter_name)

    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            if rate_limiter.allow():
                rate_limiter.consume()
                return func(*args, **kwargs)
            else:
                raise Exception("Rate limit exceeded")
        return wrapper
    return decorator
