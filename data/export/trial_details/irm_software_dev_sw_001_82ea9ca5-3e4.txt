**Architecture Document: URL Shortener Library**

## Introduction
The URL shortener library is designed to provide a lightweight and efficient way to shorten URLs using a hash-based approach. The library will support reverse lookup, collision detection, input validation, and thread safety.

## System Components
The URL shortener library will consist of the following components:

*   **URLValidator**: Responsible for validating the input URL format.
*   **URLShortener**: Generates a shortened URL using a hash-based approach.
*   **URLMapper**: Maps shortened URLs to their original URLs and supports reverse lookup.
*   **CollisionDetector**: Detects and handles collisions when generating shortened URLs.

## System Design

### Class Diagram

*   **URLValidator**
    *   `validate_url(url: str) -> bool`: Validates the input URL format.
*   **URLShortener**
    *   `shorten_url(url: str) -> str`: Generates a shortened URL using a hash-based approach.
*   **URLMapper**
    *   `map_short_url(short_url: str, original_url: str) -> None`: Maps a shortened URL to its original URL.
    *   `get_original_url(short_url: str) -> str`: Retrieves the original URL for a given shortened URL.
*   **CollisionDetector**
    *   `detect_collision(short_url: str) -> bool`: Checks if a shortened URL already exists.

### Sequence Diagram

1.  The user provides a long URL to the **URLValidator** for validation.
2.  If the URL is valid, it is passed to the **URLShortener** to generate a shortened URL.
3.  The **URLShortener** generates a shortened URL using a hash-based approach and checks for collisions using the **CollisionDetector**.
4.  If a collision is detected, the **URLShortener** generates a new shortened URL.
5.  Once a unique shortened URL is generated, it is mapped to the original URL using the **URLMapper**.
6.  The user can then use the shortened URL to retrieve the original URL using the **URLMapper**.

*   The **URLShortener** generates a shortened URL and checks for collisions using the **CollisionDetector**.
*   If a collision is detected, the **URLShortener** generates a new shortened URL.

import hashlib
import threading
from urllib.parse import urlparse

class URLValidator:
    def validate_url(self, url: str) -> bool:
        try:
            result = urlparse(url)
            return all([result.scheme, result.netloc])
        except ValueError:
            return False

class URLShortener:
    def __init__(self):
        self.mapper = URLMapper()
        self.collision_detector = CollisionDetector()

    def shorten_url(self, url: str) -> str:
        if not URLValidator().validate_url(url):
            raise ValueError("Invalid URL")

        short_url = hashlib.sha256(url.encode()).hexdigest()[:6]
        while self.collision_detector.detect_collision(short_url):
            short_url = hashlib.sha256((short_url + str(hashlib.sha256(url.encode()).hexdigest())).encode()).hexdigest()[:6]

        self.mapper.map_short_url(short_url, url)
        return short_url

class URLMapper:
    def __init__(self):
        self.url_map = {}
        self.lock = threading.Lock()

    def map_short_url(self, short_url: str, original_url: str) -> None:
        with self.lock:
            self.url_map[short_url] = original_url

    def get_original_url(self, short_url: str) -> str:
        with self.lock:
            return self.url_map.get(short_url, None)

class CollisionDetector:
    def __init__(self):
        self.mapper = URLMapper()

    def detect_collision(self, short_url: str) -> bool:
        return self.mapper.get_original_url(short_url) is not None

# Example usage:
url_shortener = URLShortener()
short_url = url_shortener.shorten_url("https://www.example.com")
print(short_url)
print(URLMapper().get_original_url(short_url))

import hashlib
import threading
from urllib.parse import urlparse
import logging

# Configure logging
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class URLValidator:
    def validate_url(self, url: str) -> bool:
        try:
            result = urlparse(url)
            return all([result.scheme, result.netloc])
        except ValueError:
            return False

    def get_error_message(self, url: str) -> str:
        if not self.validate_url(url):
            return "Invalid URL"
        return ""

class URLShortener:
    def __init__(self):
        self.mapper = URLMapper()
        self.collision_detector = CollisionDetector()
        self.lock = threading.Lock()

    def shorten_url(self, url: str) -> str:
        if not URLValidator().validate_url(url):
            logger.error("Invalid URL: {}".format(url))
            raise ValueError("Invalid URL")

        with self.lock:
            short_url = hashlib.sha256(url.encode()).hexdigest()[:6]
            counter = 0
            while self.collision_detector.detect_collision(short_url):
                counter += 1
                short_url = hashlib.sha256((short_url + str(counter)).encode()).hexdigest()[:6]

            self.mapper.map_short_url(short_url, url)
            logger.info("Shortened URL: {}".format(short_url))
            return short_url

    def get_short_url(self, url: str) -> str:
        return self.shorten_url(url)

class URLMapper:
    def __init__(self):
        self.url_map = {}
        self.lock = threading.Lock()

    def map_short_url(self, short_url: str, original_url: str) -> None:
        with self.lock:
            self.url_map[short_url] = original_url

    def get_original_url(self, short_url: str) -> str:
        with self.lock:
            return self.url_map.get(short_url, None)

class CollisionDetector:
    def __init__(self):
        self.mapper = URLMapper()

    def detect_collision(self, short_url: str) -> bool:
        return self.mapper.get_original_url(short_url) is not None

import sqlite3

class URLShortener:
    def __init__(self, db_name: str):
        self.mapper = URLMapper(db_name)
        self.collision_detector = CollisionDetector(db_name)
        self.lock = threading.Lock()

    def shorten_url(self, url: str) -> str:

            raise ValueError("Invalid URL")

        with self.lock:

            while self.collision_detector.detect_collision(short_url):

        """)

import hashlib
import threading
from urllib.parse import urlparse
import sqlite3
import logging

# Configure logging
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class URLValidator:
    def validate_url(self, url: str) -> bool:
        try:
            result = urlparse(url)
            return all([result.scheme, result.netloc])
        except ValueError:
            return False

    def get_error_message(self, url: str) -> str:
        if not self.validate_url(url):
            return "Invalid URL"
        return ""

class URLShortener:
    def __init__(self, db_name: str):
        self.mapper = URLMapper(db_name)
        self.collision_detector = CollisionDetector(db_name)
        self.lock = threading.Lock()

    def shorten_url(self, url: str) -> str:
        if not URLValidator().validate_url(url):
            logger.error("Invalid URL: {}".format(url))
            raise ValueError("Invalid URL")

        with self.lock:
            short_url = hashlib.sha256(url.encode()).hexdigest()[:6]
            counter = 0
            while self.collision_detector.detect_collision(short_url):
                counter += 1
                short_url = hashlib.sha256((short_url + str(counter)).encode()).hexdigest()[:6]

            self.mapper.map_short_url(short_url, url)
            logger.info("Shortened URL: {}".format(short_url))
            return short_url

    def get_short_url(self, url: str) -> str:
        return self.shorten_url(url)

class URLMapper:
    def __init__(self, db_name: str):
        self.conn = sqlite3.connect(db_name)
        self.cursor = self.conn.cursor()
        self.cursor.execute("""
            CREATE TABLE IF NOT EXISTS url_map
            (short_url TEXT PRIMARY KEY, original_url TEXT)
        """)
        self.conn.commit()

    def map_short_url(self, short_url: str, original_url: str) -> None:
        self.cursor.execute("INSERT INTO url_map VALUES (?, ?)", (short_url, original_url))
        self.conn.commit()

    def get_original_url(self, short_url: str) -> str:
        self.cursor.execute("SELECT original_url FROM url_map WHERE short_url = ?", (short_url,))
        result = self.cursor.fetchone()
        if result:
            return result[0]
        return None

class CollisionDetector:
    def __init__(self, db_name: str):
        self.conn = sqlite3.connect(db_name)
        self.cursor = self.conn.cursor()

    def detect_collision(self, short_url: str) -> bool:
        self.cursor.execute("SELECT * FROM url_map WHERE short_url = ?", (short_url,))
        result = self.cursor.fetchone()
        if result:
            return True
        return False
