## Overview

4. **Task Manager**: Responsible for managing the task queue, worker threads, and task status tracking.

2. **Task Consumption**:

```python
import threading
import queue
import enum
import time

class TaskStatus(enum.Enum):
    PENDING = 1
    RUNNING = 2
    COMPLETED = 3
    FAILED = 4

class Task:
    def __init__(self, task_id, priority, callback, max_retries=3):
        self.task_id = task_id
        self.priority = priority
        self.callback = callback
        self.max_retries = max_retries
        self.retries = 0
        self.status = TaskStatus.PENDING

class TaskQueue:
    def __init__(self):
        self.queue = queue.PriorityQueue()

    def add_task(self, task):
        self.queue.put((task.priority, task))

    def get_task(self):
        if not self.queue.empty():
            return self.queue.get()[1]
        return None

    def update_task_status(self, task_id, status):
        for task in self.queue.queue:
            if task[1].task_id == task_id:
                task[1].status = status
                break

class WorkerThread(threading.Thread):
    def __init__(self, task_queue):
        super().__init__()
        self.task_queue = task_queue
        self.running = False

    def start(self):
        self.running = True
        super().start()

    def run(self):
        while self.running:
            task = self.task_queue.get_task()
            if task:
                try:
                    task.callback()
                    task.status = TaskStatus.COMPLETED
                except Exception as e:
                    task.retries += 1
                    if task.retries < task.max_retries:
                        task.status = TaskStatus.PENDING
                    else:
                        task.status = TaskStatus.FAILED
                finally:
                    self.task_queue.update_task_status(task.task_id, task.status)

    def stop(self):
        self.running = False

class TaskManager:
    def __init__(self, num_workers):
        self.task_queue = TaskQueue()
        self.worker_threads = [WorkerThread(self.task_queue) for _ in range(num_workers)]

    def add_task(self, task):
        self.task_queue.add_task(task)

    def get_task_status(self, task_id):
        for task in self.task_queue.queue.queue:
            if task[1].task_id == task_id:
                return task[1].status
        return None

    def shutdown(self):
        for worker in self.worker_threads:
            worker.stop()
        for worker in self.worker_threads:
            worker.join()

# Example usage
def task_callback():
    print("Task executed")

task_manager = TaskManager(2)
task = Task(1, 1, task_callback)
task_manager.add_task(task)
task_manager.worker_threads[0].start()
task_manager.worker_threads[1].start()
time.sleep(1)

    def __init__(self, max_size):
        self.queue = []
        self.max_size = max_size
        self.lock = threading.Lock()

    def add_task(self, task):
        with self.lock:
            if len(self.queue) < self.max_size:
                heapq.heappush(self.queue, (task.priority, task))
            else:
                print("Task queue is full. Cannot add task.")

    def get_task(self):
        with self.lock:
            if self.queue:
                return heapq.heappop(self.queue)[1]
            return None

    def update_task_status(self, task_id, status):
        with self.lock:
            for i, (priority, task) in enumerate(self.queue):
                if task.task_id == task_id:
                    task.status = status
                    break

## Overview
The in-memory task queue is designed to manage tasks with priorities, track their status, and provide a mechanism for worker threads to consume tasks. The system also supports task retry with a configurable maximum number of retries and includes a graceful shutdown mechanism.

## Components

1. **Task**: Represents a unit of work with a priority and a callback function to be executed.
2. **Task Queue**: A data structure that stores tasks with their respective priorities and statuses.
3. **Worker Thread**: A thread that consumes tasks from the task queue, executes them, and updates their status.
4. **Task Manager**: Responsible for managing the task queue, worker threads, and task status tracking.

## Interfaces

1. **Task Queue Interface**:
	* `add_task(task: Task)`: Adds a task to the queue with its priority.
	* `get_task()`: Retrieves a task from the queue based on its priority.
	* `update_task_status(task_id: int, status: str)`: Updates the status of a task.
2. **Worker Thread Interface**:
	* `start()`: Starts the worker thread.
	* `stop()`: Stops the worker thread.
3. **Task Manager Interface**:
	* `add_task(task: Task)`: Adds a task to the task queue.
	* `get_task_status(task_id: int)`: Retrieves the status of a task.
	* `shutdown()`: Initiates a graceful shutdown of the system.

## Data Flow

1. **Task Addition**:
	* A task is created with a priority and a callback function.
	* The task is added to the task queue using the `add_task` method.
2. **Task Consumption**:
	* A worker thread retrieves a task from the task queue using the `get_task` method.
	* The worker thread executes the task's callback function.
	* The worker thread updates the task's status using the `update_task_status` method.
3. **Task Status Tracking**:
	* The task manager tracks the status of each task.
	* The task manager updates the task's status based on the worker thread's updates.

import threading
import queue
import enum
import time
import logging
import heapq

# Set up logging
logging.basicConfig(level=logging.INFO)

class TaskStatus(enum.Enum):
    PENDING = 1
    RUNNING = 2
    COMPLETED = 3
    FAILED = 4

class Task:
    def __init__(self, task_id, priority, callback, max_retries=3):
        """
        Initialize a Task object.

        Args:
            task_id (int): Unique identifier for the task.
            priority (int): Priority of the task.
            callback (function): Function to be executed for the task.
            max_retries (int, optional): Maximum number of retries. Defaults to 3.
        """
        self.task_id = task_id
        self.priority = priority
        self.callback = callback
        self.max_retries = max_retries
        self.retries = 0
        self.status = TaskStatus.PENDING

class TaskQueue:
    def __init__(self, max_size):
        """
        Initialize a TaskQueue object.

        Args:
            max_size (int): Maximum size of the task queue.
        """
        self.queue = []
        self.max_size = max_size
        self.lock = threading.Lock()

    def add_task(self, task):
        """
        Add a task to the queue.

        Args:
            task (Task): Task to be added to the queue.
        """
        with self.lock:
            if len(self.queue) < self.max_size:
                heapq.heappush(self.queue, (task.priority, task))
            else:
                logging.warning("Task queue is full. Cannot add task.")

    def get_task(self):
        """
        Get a task from the queue.

        Returns:
            Task: Task retrieved from the queue, or None if the queue is empty.
        """
        with self.lock:
            if self.queue:
                return heapq.heappop(self.queue)[1]
            return None

    def update_task_status(self, task_id, status):
        """
        Update the status of a task in the queue.

        Args:
            task_id (int): ID of the task to be updated.
            status (TaskStatus): New status of the task.
        """
        with self.lock:
            for i, (priority, task) in enumerate(self.queue):
                if task.task_id == task_id:
                    task.status = status
                    break

class WorkerThread(threading.Thread):
    def __init__(self, task_queue):
        """
        Initialize a WorkerThread object.

        Args:
            task_queue (TaskQueue): Task queue to be used by the worker thread.
        """
        super().__init__()
        self.task_queue = task_queue
        self.running = False

    def start(self):
        """
        Start the worker thread.
        """
        self.running = True
        super().start()

    def run(self):
        """
        Run the worker thread.
        """
        while self.running:
            task = self.task_queue.get_task()
            if task:
                try:
                    task.callback()
                    task.status = TaskStatus.COMPLETED
                except Exception as e:
                    task.retries += 1
                    if task.retries < task.max_retries:
                        task.status = TaskStatus.PENDING
                        self.task_queue.add_task(task)
                    else:
                        task.status = TaskStatus.FAILED
                    logging.error(f"Task {task.task_id} failed with error: {str(e)}")
                finally:
                    self.task_queue.update_task_status(task.task_id, task.status)

    def stop(self):
        """
        Stop the worker thread.
        """
        self.running = False

class TaskManager:
    def __init__(self, num_workers, max_queue_size):
        """
        Initialize a TaskManager object.

        Args:
            num_workers (int): Number of worker threads to be used.
            max_queue_size (int): Maximum size of the task queue.
        """
        self.task_queue = TaskQueue(max_queue_size)
        self.worker_threads = [WorkerThread(self.task_queue) for _ in range(num_workers)]

    def add_task(self, task):
        """
        Add a task to the task queue.

        Args:
            task (Task): Task to be added to the queue.
        """
        self.task_queue.add_task(task)

    def get_task_status(self, task_id):
        """
        Get the status of a task.

        Args:
            task_id (int): ID of the task to retrieve the status for.

        Returns:
            TaskStatus: Status of the task, or None if the task is not found.
        """
        for task in self.task_queue.queue:
            if task[1].task_id == task_id:
                return task[1].status
        return None

    def shutdown(self):
        """
        Shut down the task manager.
        """
        for worker in self.worker_threads:
            worker.stop()
        for worker in self.worker_threads:
            worker.join()

    def start_workers(self):
        """
        Start all worker threads.
        """
        for worker in self.worker_threads:
            worker.start()

# Example usage
def task_callback():
    """
    Example task callback function.
    """
    print("Task executed")

task_manager = TaskManager(2, 10)
task = Task(1, 1, task_callback)
task_manager.add_task(task)
task_manager.start_workers()
time.sleep(1)
task_manager.shutdown()
