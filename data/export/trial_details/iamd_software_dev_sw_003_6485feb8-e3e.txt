import pandas as pd
import json
import logging

# Define the schema definition
schema_definition = {
    'name': str,
    'age': int
}

# Define the plugin interface
class Plugin:
    def __init__(self, name):
        self.name = name

    def transform(self, data):
        raise NotImplementedError

# Define a sample plugin
class FilterPlugin(Plugin):
    def __init__(self, name, filter_func):
        super().__init__(name)
        self.filter_func = filter_func

    def transform(self, data):
        return data[self.filter_func(data)]

# Define the data pipeline
class DataPipeline:
    def __init__(self, input_file, output_file, schema_definition, plugins=None):
        self.input_file = input_file
        self.output_file = output_file
        self.schema_definition = schema_definition
        self.plugins = plugins if plugins else []

    def read_data(self):
        try:
            data = pd.read_csv(self.input_file)
            # Perform schema validation
            for column, dtype in self.schema_definition.items():
                if column not in data.columns:
                    logging.error(f"Column '{column}' not found in data")
                    raise ValueError(f"Column '{column}' not found in data")
                if data[column].dtype != dtype:
                    logging.error(f"Column '{column}' has incorrect data type")
                    raise ValueError(f"Column '{column}' has incorrect data type")
            return data
        except Exception as e:
            logging.error(f"Error reading data: {str(e)}")
            raise

    def apply_transformations(self, data):
        for plugin in self.plugins:
            data = plugin.transform(data)
        return data

    def write_data(self, data):
        try:
            if self.output_file.endswith('.csv'):
                data.to_csv(self.output_file, index=False)
            elif self.output_file.endswith('.json'):
                data.to_json(self.output_file, orient='records')
        except Exception as e:
            logging.error(f"Error writing data: {str(e)}")
            raise

    def run(self):
        data = self.read_data()
        data = self.apply_transformations(data)
        self.write_data(data)

# Example usage
if __name__ == '__main__':
    input_file = 'input.csv'
    output_file = 'output.csv'
    schema_definition = {
        'name': str,
        'age': int
    }
    plugins = [FilterPlugin('filter_plugin', lambda x: x['age'] > 18)]

    pipeline = DataPipeline(input_file, output_file, schema_definition, plugins)
    pipeline.run()

        self.schema_definition = schema_definition

        try:
            if not os.path.exists(self.input_file):
                logging.error(f"Input file '{self.input_file}' not found")
                raise FileNotFoundError(f"Input file '{self.input_file}' not found")
            data = pd.read_csv(self.input_file)
            # Perform schema validation
            for column, dtype in self.schema_definition.items():
                if column not in data.columns:
                    logging.error(f"Column '{column}' not found in data")
                    raise ValueError(f"Column '{column}' not found in data")
                if data[column].dtype != dtype:

        self.plugins = plugins

        return data

class DataWriter:
    def __init__(self, output_file):
        self.output_file = output_file

    def write_data(self, data):
        try:

                data.to_csv(self.output_file, index=False)
            elif self.output_file.endswith('.json'):
                data.to_json(self.output_file, orient='records')
        except Exception as e:

## Data Writer
