import pandas as pd
import logging
from abc import ABC, abstractmethod
from typing import Dict, List

# Logging configuration
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

# Plugin interface
class Plugin(ABC):
    """
    Abstract base class for plugins.
    
    Plugins should implement the `transform` method to apply a transformation to a DataFrame.
    """
    @abstractmethod
    def transform(self, df: pd.DataFrame) -> pd.DataFrame:
        pass

# Plugin registration
plugins: Dict[str, Plugin] = {}

def register_plugin(plugin: Plugin) -> None:
    """
    Register a plugin with the pipeline.
    
    Args:
    plugin: The plugin to register.
    """
    plugins[plugin.__class__.__name__] = plugin

# Data ingestion
def ingest_data(file_path: str, schema: Dict[str, type]) -> pd.DataFrame:
    """
    Ingest data from a CSV file.
    
    Args:
    file_path: The path to the CSV file.
    schema: The expected schema of the data.
    
    Returns:
    A pandas DataFrame containing the ingested data.
    
    Raises:
    ValueError: If the schema validation fails.
    """
    try:
        df = pd.read_csv(file_path)
        # Validate schema
        if not validate_schema(df, schema):
            raise ValueError("Schema validation failed")
        return df
    except Exception as e:
        logger.error(f"Error ingesting data: {e}")
        raise

def validate_schema(df: pd.DataFrame, schema: Dict[str, type]) -> bool:
    """
    Validate the schema of a DataFrame.
    
    Args:
    df: The DataFrame to validate.
    schema: The expected schema of the data.
    
    Returns:
    True if the schema is valid, False otherwise.
    """
    for column, dtype in schema.items():
        if column not in df.columns:
            return False
        if not all(isinstance(x, dtype) for x in df[column]):
            return False
    return True

# Transformation chain
def apply_transformations(df: pd.DataFrame, transformations: List[Plugin]) -> pd.DataFrame:
    """
    Apply a chain of transformations to a DataFrame.
    
    Args:
    df: The DataFrame to transform.
    transformations: The list of transformations to apply.
    
    Returns:
    The transformed DataFrame.
    """
    for transformation in transformations:
        df = transformation.transform(df)
    return df

# Output
def output_data(df: pd.DataFrame, file_path: str, format: str) -> None:
    """
    Output data to a CSV or JSON file.
    
    Args:
    df: The DataFrame to output.
    file_path: The path to the output file.
    format: The format of the output file (CSV or JSON).
    
    Raises:
    ValueError: If the output format is invalid.
    """
    try:
        if format == "csv":
            df.to_csv(file_path, index=False)
        elif format == "json":
            df.to_json(file_path, orient="records")
        else:
            raise ValueError("Invalid output format")
    except Exception as e:
        logger.error(f"Error outputting data: {e}")
        raise

# Example plugin
class FilterPlugin(Plugin):
    """
    A plugin that filters a DataFrame based on a condition.
    
    Args:
    column: The column to filter on.
    value: The value to filter by.
    """
    def __init__(self, column: str, value: object):
        self.column = column
        self.value = value

    def transform(self, df: pd.DataFrame) -> pd.DataFrame:
        return df[df[self.column] == self.value]

# Register plugin
register_plugin(FilterPlugin)

# Example usage
if __name__ == "__main__":
    # Ingest data
    df = ingest_data("data.csv", {"name": str, "age": int})

    # Apply transformations
    transformations = [FilterPlugin("age", 25)]
    df = apply_transformations(df, transformations)

    # Output data
