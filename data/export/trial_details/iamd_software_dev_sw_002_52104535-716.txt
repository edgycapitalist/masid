import threading
import time
from functools import wraps

class TokenBucket:
    def __init__(self, rate, burst_size):
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_refill = time.time()

    def consume(self):
        if self.tokens > 0:
            self.tokens -= 1
            return True
        return False

    def refill(self):
        current_time = time.time()
        elapsed_time = current_time - self.last_refill
        self.last_refill = current_time
        self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)

class RateLimiter:
    def __init__(self, rate, burst_size):
        self.token_bucket = TokenBucket(rate, burst_size)
        self.lock = threading.Lock()

    def allow(self):
        with self.lock:
            self.token_bucket.refill()
            return self.token_bucket.consume()

    def get_token_bucket(self):
        return self.token_bucket

class RateLimiterDecorator:
    def __init__(self, rate_limiter):
        self.rate_limiter = rate_limiter

    def __call__(self, func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            if self.rate_limiter.allow():
                return func(*args, **kwargs)
            else:
                raise Exception("Rate limit exceeded")
        return wrapper

class RateLimiterFactory:
    def __init__(self):
        self.rate_limiters = {}

    def create_rate_limiter(self, name, rate, burst_size):
        self.rate_limiters[name] = RateLimiter(rate, burst_size)

    def get_rate_limiter(self, name):
        return self.rate_limiters.get(name)
