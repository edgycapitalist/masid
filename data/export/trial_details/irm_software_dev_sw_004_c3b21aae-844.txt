## Introduction
The in-memory task queue is designed to manage tasks with priorities, track their status, and provide a mechanism for worker threads to consume tasks. The system supports task retry with a configurable maximum number of retries and includes a graceful shutdown mechanism.

## Components
The system consists of the following components:

1. **Task**: Represents a unit of work to be executed by a worker thread.
2. **Task Queue**: A data structure that stores tasks with priorities.
3. **Worker Thread**: A thread that consumes tasks from the task queue.
4. **Task Status Tracker**: A component that tracks the status of tasks (pending, running, completed, failed).
5. **Retry Mechanism**: A component that handles task retries with a configurable maximum number of retries.
6. **Shutdown Manager**: A component that manages the shutdown process.

## Data Flow
The data flow between components is as follows:

1. **Task Addition**: A task is added to the task queue with a specified priority.
2. **Task Consumption**: A worker thread consumes a task from the task queue.
3. **Task Status Update**: The task status tracker updates the status of the task (running, completed, failed).
4. **Task Retry**: If a task fails, the retry mechanism attempts to retry the task up to a maximum number of retries.
5. **Shutdown**: The shutdown manager initiates the shutdown process, which stops worker threads and removes tasks from the task queue.

## Interfaces
The system provides the following interfaces:

1. **Task Queue Interface**: Provides methods for adding tasks, consuming tasks, and tracking task status.
2. **Worker Thread Interface**: Provides methods for worker threads to consume tasks and update task status.

## Technology Choices
The system is implemented in Python, using the following technologies:

1. **Queue**: The `queue` module is used to implement the task queue.
2. **Threading**: The `threading` module is used to implement worker threads.
3. **Locks**: The `threading` module is used to implement locks for synchronizing access to shared resources.

import queue
import threading
import time

class Task:
    def __init__(self, name, priority, max_retries):
        self.name = name
        self.priority = priority
        self.max_retries = max_retries
        self.retries = 0
        self.status = "pending"

    def run(self):
        # Simulate task execution
        time.sleep(1)
        self.status = "completed"

    def retry(self):
        if self.retries < self.max_retries:
            self.retries += 1
            self.status = "pending"
            return True
        return False

class TaskQueue:
    def __init__(self):
        self.queue = queue.PriorityQueue()

    def add_task(self, task):
        self.queue.put((task.priority, task))

    def consume_task(self):
        if not self.queue.empty():
            _, task = self.queue.get()
            return task
        return None

class WorkerThread(threading.Thread):
    def __init__(self, task_queue):
        threading.Thread.__init__(self)
        self.task_queue = task_queue

    def run(self):
        while True:
            task = self.task_queue.consume_task()
            if task is None:
                break
            task.status = "running"
            try:
                task.run()
            except Exception as e:
                task.status = "failed"
                if not task.retry():
                    print(f"Task {task.name} failed after {task.max_retries} retries")
            time.sleep(0.1)

class ShutdownManager:
    def __init__(self, worker_threads):
        self.worker_threads = worker_threads

    def shutdown(self):
        for thread in self.worker_threads:
            thread.join()

def main():
    task_queue = TaskQueue()
    worker_threads = []

    # Add tasks to the task queue
    task_queue.add_task(Task("Task 1", 1, 3))
    task_queue.add_task(Task("Task 2", 2, 2))
    task_queue.add_task(Task("Task 3", 3, 1))

    # Create and start worker threads
    for _ in range(3):
        thread = WorkerThread(task_queue)
        thread.start()
        worker_threads.append(thread)

    # Shutdown the system
    shutdown_manager = ShutdownManager(worker_threads)
    shutdown_manager.shutdown()

if __name__ == "__main__":
    main()

import queue
import threading
import time
import logging

# Configure logging
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class Task:
    def __init__(self, name, priority, max_retries):
        """
        Initialize a Task object.

        Args:
            name (str): The name of the task.
            priority (int): The priority of the task.
            max_retries (int): The maximum number of retries for the task.
        """
        self.name = name
        self.priority = priority
        self.max_retries = max_retries
        self.retries = 0
        self.status = "pending"

    def run(self):
        """
        Simulate task execution.

        Returns:
            None
        """
        try:
            # Simulate task execution
            time.sleep(1)
            self.status = "completed"
        except Exception as e:
            self.status = "failed"
            logger.error(f"Task {self.name} failed: {e}")

    def retry(self):
        """
        Retry the task if it has not exceeded the maximum number of retries.

        Returns:
            bool: True if the task should be retried, False otherwise.
        """
        if self.retries < self.max_retries:
            self.retries += 1
            self.status = "pending"
            return True
        return False

class TaskQueue:
    def __init__(self):
        """
        Initialize a TaskQueue object.
        """
        self.queue = queue.PriorityQueue()
        self.lock = threading.Lock()

    def add_task(self, task):
        """
        Add a task to the task queue.

        Args:
            task (Task): The task to add.
        """
        with self.lock:
            self.queue.put((task.priority, task))

    def consume_task(self):
        """
        Consume a task from the task queue.

        Returns:
            Task: The consumed task, or None if the queue is empty.
        """
        with self.lock:
            if not self.queue.empty():
                _, task = self.queue.get()
                return task
            return None

class WorkerThread(threading.Thread):
    def __init__(self, task_queue):
        """
        Initialize a WorkerThread object.

        Args:
            task_queue (TaskQueue): The task queue to consume tasks from.
        """
        threading.Thread.__init__(self)
        self.task_queue = task_queue
        self.stop_event = threading.Event()

    def run(self):
        """
        Run the worker thread.

        Returns:
            None
        """
        while not self.stop_event.is_set():
            task = self.task_queue.consume_task()
            if task is None:
                time.sleep(0.1)
                continue
            task.status = "running"
            try:
                task.run()
            except Exception as e:
                task.status = "failed"
                logger.error(f"Task {task.name} failed: {e}")
                if not task.retry():
                    logger.error(f"Task {task.name} failed after {task.max_retries} retries")
            time.sleep(0.1)

    def stop(self):
        """
        Stop the worker thread.

        Returns:
            None
        """
        self.stop_event.set()

class ShutdownManager:
    def __init__(self, worker_threads):
        """
        Initialize a ShutdownManager object.

        Args:
            worker_threads (list): The list of worker threads to manage.

        self.worker_threads = worker_threads

    def shutdown(self):
        """
        Shutdown the worker threads.

        Returns:
            None
        """
        for thread in self.worker_threads:
            thread.stop()
        for thread in self.worker_threads:
            thread.join()

**Architecture Document: In-Memory Task Queue**

## Introduction
The in-memory task queue is designed to manage tasks with priorities, track their status, and provide a mechanism for worker threads to consume tasks. The system supports task retry with a configurable maximum number of retries and includes a graceful shutdown mechanism.

## Components
The system consists of the following components:

1. **Task**: Represents a unit of work to be executed by a worker thread.
2. **Task Queue**: A data structure that stores tasks with priorities.
3. **Worker Thread**: A thread that consumes tasks from the task queue.
4. **Task Status Tracker**: A component that tracks the status of tasks (pending, running, completed, failed).
5. **Retry Mechanism**: A component that handles task retries with a configurable maximum number of retries.
6. **Shutdown Manager**: A component that manages the shutdown process.

## Data Flow
The data flow between components is as follows:

1. **Task Addition**: A task is added to the task queue with a specified priority.
2. **Task Consumption**: A worker thread consumes a task from the task queue.
3. **Task Status Update**: The task status tracker updates the status of the task (running, completed, failed).
4. **Task Retry**: If a task fails, the retry mechanism attempts to retry the task up to a maximum number of retries.
5. **Shutdown**: The shutdown manager initiates the shutdown process, which stops worker threads and removes tasks from the task queue.

## Interfaces
The system provides the following interfaces:

1. **Task Queue Interface**: Provides methods for adding tasks, consuming tasks, and tracking task status.
2. **Worker Thread Interface**: Provides methods for worker threads to consume tasks and update task status.

## Technology Choices
The system is implemented in Python, using the following technologies:

1. **Queue**: The `queue` module is used to implement the task queue.
2. **Threading**: The `threading` module is used to implement worker threads.
3. **Locks**: The `threading` module is used to implement locks for synchronizing access to shared resources.
4. **Logging**: The `logging` module is used to handle logging and error reporting.

import queue
import threading
import time
import logging

# Set up logging
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class Task:
    def __init__(self, name, priority, max_retries):
        self.name = name
        self.priority = priority
        self.max_retries = max_retries
        self.retries = 0
        self.status = "pending"

    def run(self):
        try:
            # Simulate task execution
            time.sleep(1)
            self.status = "completed"
        except Exception as e:
            logger.error(f"Task {self.name} failed: {e}")
            self.status = "failed"

    def retry(self):
        if self.retries < self.max_retries:
            self.retries += 1
            self.status = "pending"
            return True
        return False

class TaskQueue:
    def __init__(self):
        self.queue = queue.PriorityQueue()
        self.lock = threading.Lock()

    def add_task(self, task):
        with self.lock:
            self.queue.put((task.priority, task))

    def consume_task(self):
        with self.lock:
            if not self.queue.empty():
                _, task = self.queue.get()
                return task
            return None

class WorkerThread(threading.Thread):
    def __init__(self, task_queue):
        threading.Thread.__init__(self)
        self.task_queue = task_queue
        self.shutdown_event = threading.Event()

    def run(self):
        while not self.shutdown_event.is_set():
            task = self.task_queue.consume_task()
            if task is None:
                time.sleep(0.1)
                continue
            task.status = "running"
            try:
                task.run()
            except Exception as e:
                logger.error(f"Task {task.name} failed: {e}")
                if not task.retry():
                    logger.error(f"Task {task.name} failed after {task.max_retries} retries")
            time.sleep(0.1)

    def shutdown(self):
        self.shutdown_event.set()

class ShutdownManager:
    def __init__(self, worker_threads):
        self.worker_threads = worker_threads

    def shutdown(self):
        for thread in self.worker_threads:
            thread.shutdown()
        for thread in self.worker_threads:
            thread.join()

def main():
    task_queue = TaskQueue()
    worker_threads = []

    # Add tasks to the task queue
    task_queue.add_task(Task("Task 1", 1, 3))
    task_queue.add_task(Task("Task 2", 2, 2))
    task_queue.add_task(Task("Task 3", 3, 1))

    # Create and start worker threads
    for _ in range(3):
        thread = WorkerThread(task_queue)
        thread.start()
        worker_threads.append(thread)

    # Run for 10 seconds
    time.sleep(10)

    # Shutdown the system
    shutdown_manager = ShutdownManager(worker_threads)
    shutdown_manager.shutdown()
