import pandas as pd
import jsonschema
from abc import ABC, abstractmethod

# Data Ingestion
class CSVReader:
    def __init__(self, file_path, schema):
        self.file_path = file_path
        self.schema = schema

    def read(self):
        try:
            data = pd.read_csv(self.file_path)
            jsonschema.validate(instance=data.to_dict(orient='records'), schema=self.schema)
            return data
        except jsonschema.exceptions.ValidationError as e:
            raise ValueError(f"Invalid schema: {e}")

# Transformation Chain
class Transformation(ABC):
    @abstractmethod
    def apply(self, data):
        pass

class FilteringTransformation(Transformation):
    def __init__(self, condition):
        self.condition = condition

    def apply(self, data):
        return data[self.condition]

class MappingTransformation(Transformation):
    def __init__(self, mapping):
        self.mapping = mapping

    def apply(self, data):
        return data.apply(self.mapping)

class AggregationTransformation(Transformation):
    def __init__(self, function):
        self.function = function

    def apply(self, data):
        return data.agg(self.function)

# Plugin System
class Plugin(ABC):
    @abstractmethod
    def apply(self, data):
        pass

class PluginRegistry:
    def __init__(self):
        self.plugins = []

    def register(self, plugin):
        self.plugins.append(plugin)

    def apply(self, data):
        for plugin in self.plugins:
            data = plugin.apply(data)
        return data

# Data Output
class CSVWriter:
    def __init__(self, file_path):
        self.file_path = file_path

    def write(self, data):
        data.to_csv(self.file_path, index=False)

class JSONWriter:
    def __init__(self, file_path):
        self.file_path = file_path

    def write(self, data):
        data.to_json(self.file_path, orient='records')

# Error Reporting
class ErrorHandler:
    def __init__(self):
        self.errors = []

    def handle(self, error):
        self.errors.append(error)

    def report(self):
        return self.errors

# Pipeline
class Pipeline:
    def __init__(self, csv_reader, transformations, plugin_registry, writer, error_handler):
        self.csv_reader = csv_reader
        self.transformations = transformations
        self.plugin_registry = plugin_registry
        self.writer = writer
        self.error_handler = error_handler

    def run(self):
        try:
            data = self.csv_reader.read()
            for transformation in self.transformations:
                data = transformation.apply(data)
            data = self.plugin_registry.apply(data)
            self.writer.write(data)
        except Exception as e:
            self.error_handler.handle(e)

# Example usage
if __name__ == "__main__":
    csv_reader = CSVReader('input.csv', {'type': 'object', 'properties': {'name': {'type': 'string'}, 'age': {'type': 'integer'}}})
    filtering_transformation = FilteringTransformation(lambda x: x['age'] > 18)
    mapping_transformation = MappingTransformation(lambda x: x['name'].upper())
    aggregation_transformation = AggregationTransformation('mean')
    plugin_registry = PluginRegistry()
    csv_writer = CSVWriter('output.csv')
    error_handler = ErrorHandler()
    pipeline = Pipeline(csv_reader, [filtering_transformation, mapping_transformation, aggregation_transformation], plugin_registry, csv_writer, error_handler)
    pipeline.run()
    print(error_handler.report())

class Transformation(ABC):
    @abstractmethod
    def apply(self, data: pd.DataFrame) -> pd.DataFrame:
        pass

        self.condition = condition

    def apply(self, data: pd.DataFrame) -> pd.DataFrame:
        return data[self.condition]

class MappingTransformation(Transformation):
    def __init__(self, mapping: callable):
        self.mapping = mapping

    def apply(self, data: pd.DataFrame) -> pd.DataFrame:
        return data.apply(self.mapping)

        self.function = function

    def apply(self, data: pd.DataFrame) -> pd.DataFrame:

    @abstractmethod
    def apply(self, data: pd.DataFrame) -> pd.DataFrame:

        self.plugins.append(plugin)

    def apply(self, data: pd.DataFrame) -> pd.DataFrame:
        for plugin in self.plugins:
            data = plugin.apply(data)
        return data

        self.file_path = file_path

    def __init__(self):
        self.errors: List[Exception] = []

    def handle(self, error: Exception):
        self.errors.append(error)

    def report(self) -> List[Exception]:
        return self.errors

import pandas as pd
import jsonschema
from abc import ABC, abstractmethod
from typing import List

# Data Ingestion
class CSVReader:
    def __init__(self, file_path: str, schema: dict):
        self.file_path = file_path
        self.schema = schema

    def read(self) -> pd.DataFrame:
        try:
            data = pd.read_csv(self.file_path)
            jsonschema.validate(instance=data.to_dict(orient='records'), schema=self.schema)
            return data
        except jsonschema.exceptions.ValidationError as e:
            raise ValueError(f"Invalid schema: {e}")

# Transformation Chain
class Transformation(ABC):
    @abstractmethod
    def apply(self, data: pd.DataFrame) -> pd.DataFrame:
        pass

class FilteringTransformation(Transformation):
    def __init__(self, condition: callable):
        self.condition = condition

    def apply(self, data: pd.DataFrame) -> pd.DataFrame:
        return data[self.condition]

class MappingTransformation(Transformation):
    def __init__(self, mapping: callable):
        self.mapping = mapping

    def apply(self, data: pd.DataFrame) -> pd.DataFrame:
        return data.apply(self.mapping)

class AggregationTransformation(Transformation):
    def __init__(self, function: callable):
        self.function = function

    def apply(self, data: pd.DataFrame) -> pd.DataFrame:
        return data.agg(self.function)

# Plugin System
class Plugin(ABC):
    @abstractmethod
    def apply(self, data: pd.DataFrame) -> pd.DataFrame:
        pass

class PluginRegistry:
    def __init__(self):
        self.plugins: List[Plugin] = []

    def register(self, plugin: Plugin):
        self.plugins.append(plugin)

    def apply(self, data: pd.DataFrame) -> pd.DataFrame:
        for plugin in self.plugins:
            data = plugin.apply(data)
        return data

# Data Output
class Writer(ABC):
    @abstractmethod
    def write(self, data: pd.DataFrame):
        pass

class CSVWriter(Writer):
    def __init__(self, file_path: str):
        self.file_path = file_path

    def write(self, data: pd.DataFrame):
        data.to_csv(self.file_path, index=False)

class JSONWriter(Writer):
    def __init__(self, file_path: str):
        self.file_path = file_path

    def write(self, data: pd.DataFrame):
        data.to_json(self.file_path, orient='records')

# Error Reporting
class ErrorHandler:
    def __init__(self):
        self.errors: List[Exception] = []

    def handle(self, error: Exception):
        self.errors.append(error)

    def report(self) -> List[Exception]:
        return self.errors

# Pipeline
class Pipeline:
    def __init__(self, csv_reader: CSVReader, transformations: List[Transformation], plugin_registry: PluginRegistry, writer: Writer, error_handler: ErrorHandler):
        self.csv_reader = csv_reader
        self.transformations = transformations
        self.plugin_registry = plugin_registry
        self.writer = writer
        self.error_handler = error_handler

    def run(self):
        try:
            data = self.csv_reader.read()
            for transformation in self.transformations:
                data = transformation.apply(data)
            data = self.plugin_registry.apply(data)
            self.writer.write(data)
        except Exception as e:
            self.error_handler.handle(e)

    @classmethod
    def builder(cls, csv_reader: CSVReader):
        return PipelineBuilder(csv_reader)

class PipelineBuilder:
    def __init__(self, csv_reader: CSVReader):
        self.csv_reader = csv_reader
        self.transformations: List[Transformation] = []
        self.plugin_registry = PluginRegistry()
        self.writer = CSVWriter('output.csv')
        self.error_handler = ErrorHandler()

    def add_transformation(self, transformation: Transformation):
        self.transformations.append(transformation)
        return self

    def add_plugin(self, plugin: Plugin):
        self.plugin_registry.register(plugin)
        return self

    def set_writer(self, writer: Writer):
        self.writer = writer
        return self

    def set_error_handler(self, error_handler: ErrorHandler):
        self.error_handler = error_handler
        return self

    def build(self) -> Pipeline:
        return Pipeline(self.csv_reader, self.transformations, self.plugin_registry, self.writer, self.error_handler)
