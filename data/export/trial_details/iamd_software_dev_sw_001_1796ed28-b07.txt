import hashlib
import urllib.parse
import threading

class URLShortener:
    def __init__(self):
        self.url_map = {}
        self.lock = threading.Lock()

    def shorten_url(self, long_url):
        """
        Returns a shortened version of the input URL.

        Args:
            long_url (str): The input URL to be shortened.

        Returns:
            str: The shortened URL.
        """
        if not self.is_valid_url(long_url):
            raise ValueError("Invalid URL")

        with self.lock:
            hash_object = hashlib.sha256(long_url.encode())
            short_url = hash_object.hexdigest()[:6]

            if short_url in self.url_map:
                # Collision detected, generate a new shortened URL
                short_url = self.generate_new_short_url(long_url)

            self.url_map[short_url] = long_url

        return short_url

    def get_original_url(self, short_url):
        """
        Returns the original URL corresponding to the input short URL.

        Args:
            short_url (str): The input short URL.

        Returns:
            str: The original URL.
        """
        with self.lock:
            return self.url_map.get(short_url)

    def is_valid_url(self, url):
        """
        Returns a boolean indicating whether the input URL is valid.

        Args:
            url (str): The input URL.

        Returns:
            bool: True if the URL is valid, False otherwise.
        """
        try:
            result = urllib.parse.urlparse(url)
            return all([result.scheme, result.netloc])
        except ValueError:
            return False

    def generate_new_short_url(self, long_url):
        """
        Generates a new shortened URL in case of a collision.

        Args:
            long_url (str): The input URL.

        Returns:
            str: The new shortened URL.
        """
        hash_object = hashlib.sha256((long_url + str(threading.current_thread().ident)).encode())
        return hash_object.hexdigest()[:6]

# Example usage

long_url = "https://www.example.com"
short_url = url_shortener.shorten_url(long_url)
print(f"Short URL: {short_url}")
print(f"Original URL: {url_shortener.get_original_url(short_url)}")
