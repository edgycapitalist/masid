### Overview

The in-memory task queue is designed to manage tasks with priorities, track their status, and provide a mechanism for worker threads to consume tasks. The system will also support task retry with configurable max retries and include a graceful shutdown mechanism.

### Components

*   **Task**: Represents a unit of work to be executed. It has a unique ID, priority, and a callback function to be executed.
*   **Task Queue**: Manages the tasks and provides methods for adding tasks, consuming tasks, and tracking task status.
*   **Worker Thread**: Consumes tasks from the task queue and executes them.
*   **Task Status Tracker**: Tracks the status of tasks (pending, running, completed, failed).

### System Design

The system will use the following design:

*   The task queue will be implemented as a priority queue using a binary heap data structure.
*   The task status tracker will be implemented as a dictionary mapping task IDs to their respective status.
*   The worker threads will be implemented using Python's threading library.
*   The task retry mechanism will be implemented using a decorator that wraps the task callback function.

### Interfaces

*   `TaskQueue` class:
    *   `add_task(task)`: Adds a task to the task queue.
    *   `consume_task()`: Consumes a task from the task queue and returns it.
    *   `track_status(task_id, status)`: Updates the status of a task.
*   `WorkerThread` class:
    *   `run()`: Consumes tasks from the task queue and executes them.
*   `Task` class:
    *   `__init__(id, priority, callback)`: Initializes a task with an ID, priority, and callback function.
    *   `execute()`: Executes the task callback function.

### Data Flow

1.  A task is added to the task queue using the `add_task` method.
2.  A worker thread consumes a task from the task queue using the `consume_task` method.
3.  The worker thread executes the task using the `execute` method.
4.  The task status is updated using the `track_status` method.
5.  If a task fails, it is retried using the task retry mechanism.

import threading
import heapq
import logging
from functools import wraps

# Set up logging
logging.basicConfig(level=logging.INFO)

class Task:
    def __init__(self, id, priority, callback):
        self.id = id
        self.priority = priority
        self.callback = callback
        self.status = "pending"
        self.retries = 0

    def execute(self):
        try:
            self.callback()
            self.status = "completed"
        except Exception as e:
            self.status = "failed"
            logging.error(f"Task {self.id} failed: {e}")

class TaskQueue:
    def __init__(self):
        self.queue = []
        self.status_tracker = {}
        self.lock = threading.Lock()

    def add_task(self, task):
        with self.lock:
            heapq.heappush(self.queue, (task.priority, task))
            self.status_tracker[task.id] = task.status

    def consume_task(self):
        with self.lock:
            if not self.queue:
                return None
            _, task = heapq.heappop(self.queue)
            return task

    def track_status(self, task_id, status):
        with self.lock:
            self.status_tracker[task_id] = status

def retry(max_retries):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            task = args[0]
            for _ in range(max_retries):
                try:
                    func(*args, **kwargs)
                    break
                except Exception as e:
                    logging.error(f"Task {task.id} failed: {e}")
                    task.retries += 1
                    if task.retries >= max_retries:
                        task.status = "failed"
                        raise
        return wrapper
    return decorator

class WorkerThread(threading.Thread):
    def __init__(self, task_queue):
        super().__init__()
        self.task_queue = task_queue

    def run(self):
        while True:
            task = self.task_queue.consume_task()
            if task is None:
                break
            task.status = "running"
            self.task_queue.track_status(task.id, task.status)
            @retry(3)
            def execute_task(task):
                task.execute()
            execute_task(task)
            task.status = "completed"
            self.task_queue.track_status(task.id, task.status)

def main():
    task_queue = TaskQueue()

    # Add tasks to the task queue
    task1 = Task(1, 1, lambda: print("Task 1 executed"))
    task2 = Task(2, 2, lambda: print("Task 2 executed"))
    task_queue.add_task(task1)
    task_queue.add_task(task2)

    # Create and start worker threads
    worker_thread1 = WorkerThread(task_queue)
    worker_thread2 = WorkerThread(task_queue)
    worker_thread1.start()
    worker_thread2.start()

    # Wait for worker threads to finish
    worker_thread1.join()
    worker_thread2.join()

if __name__ == "__main__":
    main()

import logging
from functools import wraps

# Set up logging
logging.basicConfig(level=logging.INFO)

class Task:
    def __init__(self, id, priority, callback):
        """
        Initializes a task with an ID, priority, and callback function.

        Args:
            id (int): Unique task ID.
            priority (int): Task priority.
            callback (function): Task callback function.
        """
        self.id = id
        self.priority = priority
        self.callback = callback
        self.status = "pending"
        self.retries = 0

    def execute(self):
        """
        Executes the task callback function.
        """
        try:
            self.callback()
            self.status = "completed"
        except Exception as e:
            self.status = "failed"
            logging.error(f"Task {self.id} failed: {e}")

    def __repr__(self):
        """
        Returns a string representation of the task.
        """
        return f"Task {self.id} (Priority: {self.priority}, Status: {self.status})"

class TaskQueue:
    def __init__(self):
        """
        Initializes a task queue.
        """
        self.queue = []
        self.status_tracker = {}
        self.lock = threading.Lock()

    def add_task(self, task):
        """
        Adds a task to the task queue.

        Args:
            task (Task): Task to be added.
        """
        with self.lock:
            if task.id in self.status_tracker:
                logging.warning(f"Task {task.id} already exists in the queue.")
                return
            heapq.heappush(self.queue, (task.priority, task))
            self.status_tracker[task.id] = task.status

    def consume_task(self):
        """
        Consumes a task from the task queue.

        Returns:
            Task: Consumed task or None if the queue is empty.
        """
        with self.lock:
            if not self.queue:
                return None
            _, task = heapq.heappop(self.queue)
            return task

    def track_status(self, task_id, status):
        """
        Updates the status of a task.

        Args:
            task_id (int): Task ID.
            status (str): New task status.
        """
        with self.lock:
            if task_id not in self.status_tracker:
                logging.warning(f"Task {task_id} does not exist in the queue.")
                return
            self.status_tracker[task_id] = status

    def __len__(self):
        """
        Returns the number of tasks in the queue.
        """
        with self.lock:
            return len(self.queue)

def retry(max_retries):
    """
    Decorator to retry a function with a maximum number of retries.

    Args:
        max_retries (int): Maximum number of retries.

    Returns:
        function: Decorated function.
    """
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            task = args[0]
            for _ in range(max_retries):
                try:
                    func(*args, **kwargs)
                    break
                except Exception as e:
                    logging.error(f"Task {task.id} failed: {e}")
                    task.retries += 1
                    if task.retries >= max_retries:
                        task.status = "failed"
                        raise
        return wrapper
    return decorator

    def __init__(self, task_queue):
        """
        Initializes a worker thread.

        Args:
            task_queue (TaskQueue): Task queue.
        """
        super().__init__()
        self.task_queue = task_queue

    def run(self):
        """
        Runs the worker thread.

            task = self.task_queue.consume_task()
            if task is None:
