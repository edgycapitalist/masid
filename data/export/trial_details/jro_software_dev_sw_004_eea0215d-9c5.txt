**Architecture Document: In-Memory Task Queue**

## Introduction

This document outlines the design and implementation of an in-memory task queue in Python. The task queue supports adding tasks with priorities, allows worker threads to consume tasks, provides task status tracking, supports task retry with configurable max retries, and includes graceful shutdown.

## System Components

The in-memory task queue consists of the following components:

1. **Task**: Represents a single task to be executed. Each task has a unique ID, priority, and status.
2. **Task Queue**: A data structure that stores tasks in memory. The task queue is responsible for managing task priorities and ordering.
3. **Worker Thread**: A thread that consumes tasks from the task queue and executes them.
4. **Task Status Tracker**: A component that tracks the status of each task (pending, running, completed, failed).
5. **Retry Mechanism**: A component that handles task retries with configurable max retries.

The following diagram illustrates the system flow:

The system flow is as follows:

1. Tasks are submitted to the task queue with a priority.
2. The task queue stores the tasks in memory and orders them based on priority.
3. Worker threads consume tasks from the task queue and execute them.
4. The task status tracker updates the status of each task (pending, running, completed, failed).
5. If a task fails, the retry mechanism attempts to retry the task with a configurable max retries.

```python
import threading
import queue
import enum
import logging

# Task status tracking
class TaskStatus(enum.Enum):
    PENDING = 1
    RUNNING = 2
    COMPLETED = 3
    FAILED = 4

# Task class
class Task:
    def __init__(self, task_id, priority, func):
        self.task_id = task_id
        self.priority = priority
        self.func = func
        self.status = TaskStatus.PENDING

# Task queue class
class TaskQueue:
    def __init__(self):
        self.queue = queue.PriorityQueue()

    def add_task(self, task):
        self.queue.put((task.priority, task))

    def get_task(self):
        return self.queue.get()

# Worker thread class
class WorkerThread(threading.Thread):
    def __init__(self, task_queue):
        super().__init__()
        self.task_queue = task_queue

    def run(self):
        while True:
            task = self.task_queue.get_task()
            task.status = TaskStatus.RUNNING
            try:
                task.func()
                task.status = TaskStatus.COMPLETED
            except Exception as e:
                task.status = TaskStatus.FAILED
                logging.error(f"Task {task.task_id} failed: {e}")
                # Retry mechanism
                if task.status == TaskStatus.FAILED:
                    # Attempt to retry the task
                    self.task_queue.add_task(task)

# Task status tracker class
class TaskStatusTracker:
    def __init__(self):
        self.task_status = {}

    def update_task_status(self, task):
        self.task_status[task.task_id] = task.status

# Retry mechanism class
class RetryMechanism:
    def __init__(self, max_retries):
        self.max_retries = max_retries

    def attempt_retry(self, task):
        if task.status == TaskStatus.FAILED and task.retries < self.max_retries:
            task.retries += 1
            return True
        return False

# Main function
def main():
    task_queue = TaskQueue()
    worker_thread = WorkerThread(task_queue)
    task_status_tracker = TaskStatusTracker()
    retry_mechanism = RetryMechanism(3)

    # Add tasks to the task queue
    task1 = Task(1, 1, lambda: print("Task 1 executed"))
    task2 = Task(2, 2, lambda: print("Task 2 executed"))
    task_queue.add_task(task1)
    task_queue.add_task(task2)

    # Start the worker thread
    worker_thread.start()

    # Wait for the worker thread to finish
    worker_thread.join()

if __name__ == "__main__":
    main()
```

```python
import threading
import queue
import enum
import logging
import time

# Task status tracking
class TaskStatus(enum.Enum):
    PENDING = 1
    RUNNING = 2
    COMPLETED = 3
    FAILED = 4

# Task class
class Task:
    def __init__(self, task_id, priority, func):
        self.task_id = task_id
        self.priority = priority
        self.func = func
        self.status = TaskStatus.PENDING
        self.retries = 0

# Task queue class
class TaskQueue:
    def __init__(self):
        self.queue = queue.PriorityQueue()
        self.lock = threading.Lock()

    def add_task(self, task):
        with self.lock:
            self.queue.put((task.priority, task))

    def get_task(self):
        with self.lock:
            return self.queue.get()

    def clear(self):
        with self.lock:
            while not self.queue.empty():
                self.queue.get()

    def is_empty(self):
        with self.lock:
            return self.queue.empty()

# Worker thread class
class WorkerThread(threading.Thread):
    def __init__(self, task_queue, retry_mechanism):
        super().__init__()
        self.task_queue = task_queue
        self.retry_mechanism = retry_mechanism
        self.running = True

    def run(self):
        while self.running:
            try:
                task = self.task_queue.get_task()
            except queue.Empty:
                time.sleep(0.1)
                continue
            task.status = TaskStatus.RUNNING
            try:
                task.func()
                task.status = TaskStatus.COMPLETED
            except Exception as e:
                task.status = TaskStatus.FAILED
                logging.error(f"Task {task.task_id} failed: {e}")
                if self.retry_mechanism.attempt_retry(task):
                    self.task_queue.add_task(task)

        self.running = False
```

```python
        self.lock = threading.Lock()

    def get_task(self):

    def clear(self):
        with self.lock:
            while not self.queue.empty():

        self.retry_mechanism = retry_mechanism
        self.running = True

                    self.task_queue.add_task(task)

    def stop(self):
        self.running = False

        return f"WorkerThread(task_queue={self.task_queue}, retry_mechanism={self.retry_mechanism})"
```

```python
                task.status = TaskStatus.FAILED
```
```python
                logging.error(f"Task {task.task_id} failed: {e}")
                if self.retry_mechanism.attempt_retry(task):
                    self.task_queue.add_task(task)

    def stop(self):
        self.running = False

    def __repr__(self):
        return f"WorkerThread(task_queue={self.task_queue}, retry_mechanism={self.retry_mechanism})"

# Task status tracker class
class TaskStatusTracker:
    def __init__(self):
        self.task_status = {}

    def update_task_status(self, task):
        self.task_status[task.task_id] = task.status

    def __repr__(self):
        return f"TaskStatusTracker(task_status={self.task_status})"

# Retry mechanism class
class RetryMechanism:
    def __init__(self, max_retries):
        self.max_retries = max_retries

    def attempt_retry(self, task):
        if task.status == TaskStatus.FAILED and task.retries < self.max_retries:
            task.retries += 1
            return True
        return False

    def reset(self, task):
        task.retries = 0

    def __repr__(self):
        return f"RetryMechanism(max_retries={self.max_retries})"

# Main function
def main():
    task_queue = TaskQueue()
    retry_mechanism = RetryMechanism(3)
    worker_thread = WorkerThread(task_queue, retry_mechanism)
    task_status_tracker = TaskStatusTracker()

    # Add tasks to the task queue
    task1 = Task(1, 1, lambda: print("Task 1 executed"))
    task2 = Task(2, 2, lambda: print("Task 2 executed"))
    task_queue.add_task(task1)
    task_queue.add_task(task2)

    # Start the worker thread
    worker_thread.start()

    # Wait for 5 seconds
    time.sleep(5)

    # Stop the worker thread
    worker_thread.stop()
    worker_thread.join()

    # Print task status
    print(task_status_tracker.task_status)

if __name__ == "__main__":
    main()
```
