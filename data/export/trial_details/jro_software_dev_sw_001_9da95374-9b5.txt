import hashlib
import threading
from urllib.parse import urlparse

class UrlValidator:
    def validate_url(self, long_url):
        try:
            result = urlparse(long_url)
            return all([result.scheme, result.netloc])
        except ValueError:
            return False

class HashGenerator:
    def generate_hash(self, long_url):
        return hashlib.sha256(long_url.encode()).hexdigest()[:6]

class CollisionDetector:
    def __init__(self):
        self.lock = threading.Lock()
        self.short_url_map = {}

    def handle_collision(self, short_url):
        with self.lock:
            if short_url in self.short_url_map:
                new_short_url = self.generate_new_short_url()
                return new_short_url
            else:
                return short_url

    def generate_new_short_url(self):
        # Generate a new shortened URL that does not collide with existing shortened URLs
        new_short_url = hashlib.sha256(str(threading.current_thread().ident).encode()).hexdigest()[:6]
        return new_short_url

class UrlShortener:
    def __init__(self):
        self.url_validator = UrlValidator()
        self.hash_generator = HashGenerator()
        self.collision_detector = CollisionDetector()
        self.short_url_map = {}
        self.lock = threading.Lock()

    def shorten_url(self, long_url):
        if not self.url_validator.validate_url(long_url):
            raise ValueError("Invalid URL")

        with self.lock:
            short_url = self.hash_generator.generate_hash(long_url)
            if short_url in self.short_url_map:
                short_url = self.collision_detector.handle_collision(short_url)
            self.short_url_map[short_url] = long_url
            return short_url

    def reverse_lookup(self, short_url):
        with self.lock:
            return self.short_url_map.get(short_url)

# Example usage:
url_shortener = UrlShortener()
long_url = "https://www.example.com"
short_url = url_shortener.shorten_url(long_url)
print(f"Shortened URL: {short_url}")
print(f"Original URL: {url_shortener.reverse_lookup(short_url)}")
