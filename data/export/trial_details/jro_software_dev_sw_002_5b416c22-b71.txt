import threading
import time
from functools import wraps

class RateLimiter:
    def __init__(self, rate, burst_size):
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()

    def consume_tokens(self, tokens):
        if self.check_tokens(tokens):
            self.tokens -= tokens
            return True
        return False

    def check_tokens(self, tokens):
        current_time = time.time()
        elapsed_time = current_time - self.last_update
        self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)
        self.last_update = current_time
        return self.tokens >= tokens

class NamedRateLimiter:
    def __init__(self):
        self.rate_limiters = {}
        self.lock = threading.Lock()

    def get_rate_limiter(self, name):
        with self.lock:
            if name not in self.rate_limiters:
                raise ValueError(f"Rate limiter '{name}' not found")
            return self.rate_limiters[name]

    def add_rate_limiter(self, name, rate, burst_size):
        with self.lock:
            if name in self.rate_limiters:
                raise ValueError(f"Rate limiter '{name}' already exists")
            self.rate_limiters[name] = RateLimiter(rate, burst_size)

def rate_limited(name):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            rate_limiter = NamedRateLimiter().get_rate_limiter(name)
            if not rate_limiter.consume_tokens(1):
                raise Exception(f"Rate limit exceeded for '{name}'")
            return func(*args, **kwargs)
        return wrapper
    return decorator

# Example usage
named_rate_limiter = NamedRateLimiter()
named_rate_limiter.add_rate_limiter("example", 5, 10)

@rate_limited("example")
def example_function():
    print("Hello, world!")

class RateLimiter:
    def __init__(self, rate, burst_size):
        """

        Args:

import threading
import time
from functools import wraps

class RateLimiter:
    def __init__(self, rate, burst_size):
        """
        Initialize the rate limiter with the given rate and burst size.

        Args:
            rate (float): The rate at which tokens are added to the bucket.
            burst_size (int): The maximum number of tokens in the bucket.
        """
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()
        self.lock = threading.Lock()

    def consume_tokens(self, tokens):
        """
        Consume the given number of tokens from the rate limiter.

        Args:
            tokens (int): The number of tokens to consume.

        Returns:
            bool: True if the tokens were consumed, False otherwise.
        """
        with self.lock:
            if self.check_tokens(tokens):
                self.tokens -= tokens
                return True
            return False

    def check_tokens(self, tokens):
        """
        Check if the given number of tokens is available in the rate limiter.

        Args:
            tokens (int): The number of tokens to check.

        Returns:
            bool: True if the tokens are available, False otherwise.
        """
        with self.lock:
            current_time = time.time()
            elapsed_time = current_time - self.last_update
            self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)
            self.last_update = current_time
            return self.tokens >= tokens

class NamedRateLimiter:
    def __init__(self):
        """
        Initialize the named rate limiter.
        """
        self.rate_limiters = {}
        self.lock = threading.Lock()

    def get_rate_limiter(self, name):
        """
        Get the rate limiter instance for the given name.

        Args:
            name (str): The name of the rate limiter.

        Returns:
            RateLimiter: The rate limiter instance.
        """
        with self.lock:
            if name not in self.rate_limiters:
                raise ValueError(f"Rate limiter '{name}' not found")
            return self.rate_limiters[name]

    def add_rate_limiter(self, name, rate, burst_size):
        """
        Add a new rate limiter with the given name, rate, and burst size.

        Args:
            name (str): The name of the rate limiter.
            rate (float): The rate at which tokens are added to the bucket.
            burst_size (int): The maximum number of tokens in the bucket.
        """
        with self.lock:
            if name in self.rate_limiters:
                raise ValueError(f"Rate limiter '{name}' already exists")
            self.rate_limiters[name] = RateLimiter(rate, burst_size)

def rate_limited(name):
    """
    Decorator to apply rate limiting to a function.

    Args:
        name (str): The name of the rate limiter.

    Returns:
        callable: The decorated function.
    """
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            named_rate_limiter = NamedRateLimiter()
            try:

            if not rate_limiter.consume_tokens(1):
                raise Exception(f"Rate limit exceeded for '{name}'")
            return func(*args, **kwargs)
        return wrapper
    return decorator
