import hashlib
import threading
from urllib.parse import urlparse

class URLValidator:
    def __init__(self):
        pass

    def validate(self, url):
        try:
            result = urlparse(url)
            return all([result.scheme, result.netloc])
        except ValueError:
            return False

class HashGenerator:
    def __init__(self):
        pass

    def generate(self, url):
        return hashlib.sha256(url.encode()).hexdigest()[:6]

class CollisionDetector:
    def __init__(self):
        self.lock = threading.Lock()
        self.hash_map = {}

    def detect(self, hash):
        with self.lock:
            return hash in self.hash_map

    def add(self, hash, url):
        with self.lock:
            self.hash_map[hash] = url

class URLShortener:
    def __init__(self):
        self.validator = URLValidator()
        self.generator = HashGenerator()
        self.detector = CollisionDetector()

    def shorten(self, url):
        if not self.validator.validate(url):
            raise ValueError("Invalid URL")

        hash = self.generator.generate(url)
        while self.detector.detect(hash):
            hash = self.generator.generate(url + str(hash))

        self.detector.add(hash, url)
        return f"http://short.url/{hash}"

    def lookup(self, short_url):
        hash = short_url.split("/")[-1]
        return self.detector.hash_map.get(hash)

# Example usage
shortener = URLShortener()
long_url = "https://www.example.com"
short_url = shortener.shorten(long_url)
print(short_url)
print(shortener.lookup(short_url))

import hashlib
import threading
from urllib.parse import urlparse
import sqlite3
import logging

class URLValidator:
    def __init__(self):
        pass

    def validate(self, url):
        try:
            result = urlparse(url)
            return all([result.scheme, result.netloc])
        except ValueError:
            return False

class HashGenerator:
    def __init__(self):
        pass

    def generate(self, url):
        return hashlib.sha256(url.encode()).hexdigest()[:6]

class CollisionDetector:
    def __init__(self, storage):
        self.storage = storage

    def detect(self, hash):
        return self.storage.exists(hash)

class Storage:
    def __init__(self, db_file):
        self.conn = sqlite3.connect(db_file)
        self.cursor = self.conn.cursor()
        self.cursor.execute("""CREATE TABLE IF NOT EXISTS urls
                                (hash TEXT PRIMARY KEY, url TEXT)""")
        self.conn.commit()

    def exists(self, hash):
        self.cursor.execute("SELECT * FROM urls WHERE hash=?", (hash,))
        return self.cursor.fetchone() is not None

    def add(self, hash, url):
        self.cursor.execute("INSERT INTO urls VALUES (?, ?)", (hash, url))
        self.conn.commit()

    def get(self, hash):
        self.cursor.execute("SELECT url FROM urls WHERE hash=?", (hash,))
        return self.cursor.fetchone()[0]

class ErrorHandler:
    def __init__(self):
        self.logger = logging.getLogger(__name__)

    def handle(self, exception):
        self.logger.error(exception)

class URLShortener:
    def __init__(self, storage, error_handler):
        self.validator = URLValidator()
        self.generator = HashGenerator()
        self.detector = CollisionDetector(storage)
        self.storage = storage
        self.error_handler = error_handler

    def shorten(self, url):
        try:
            if not self.validator.validate(url):
                raise ValueError("Invalid URL")

            hash = self.generator.generate(url)
            while self.detector.detect(hash):
                hash = self.generator.generate(url + str(hash))

            self.storage.add(hash, url)
            return f"http://short.url/{hash}"
        except Exception as e:
            self.error_handler.handle(e)
            raise

    def lookup(self, short_url):
        try:
            hash = short_url.split("/")[-1]
            return self.storage.get(hash)
        except Exception as e:
            self.error_handler.handle(e)
            raise

class URLValidator:
    def __init__(self):

    def validate(self, url):
        try:
            result = urlparse(url)
            if not all([result.scheme, result.netloc, result.path]):
                return False
            query = parse_qs(result.query)
