**Architecture Document: URL Shortening Library**

**Overview**
-----------

The URL shortening library is designed to provide a thread-safe, hash-based URL shortening functionality. The library will accept a long URL, generate a shortened version, and support reverse lookup from short URL to original URL. It will also handle collision detection and include input validation for valid URL formats.

**Module Decomposition**
----------------------

The library will consist of the following modules:

*   `url_shortener`: The main module responsible for URL shortening and reverse lookup.
*   `hash_generator`: A utility module that generates a hash-based shortened URL.
*   `collision_detector`: A module that detects and handles collisions.
*   `url_validator`: A module that validates the input URL format.

**Interfaces**
------------

The library will provide the following interfaces:

*   `shorten_url(long_url: str) -> str`: Generates a shortened URL from a long URL.
*   `lookup_short_url(short_url: str) -> str`: Returns the original URL from a shortened URL.
*   `validate_url(url: str) -> bool`: Validates the input URL format.

**Data Flow**
------------

The data flow of the library will be as follows:

1.  The user inputs a long URL and calls the `shorten_url` function.
2.  The `shorten_url` function calls the `url_validator` module to validate the input URL format.
3.  If the URL is valid, the `shorten_url` function calls the `hash_generator` module to generate a shortened URL.
4.  The `shorten_url` function then checks for collisions using the `collision_detector` module.
5.  If a collision is detected, the `shorten_url` function generates a new shortened URL and repeats the collision check.
6.  Once a unique shortened URL is generated, it is stored in a database along with the original URL.
7.  The `lookup_short_url` function can be used to retrieve the original URL from a shortened URL.

**Technology Choices**
--------------------

The library will be implemented in Python 3.8 or higher, using the following technologies:

*   `hashlib` library for generating hashes.
*   `sqlite3` library for storing shortened URLs and original URLs.
*   `threading` library for ensuring thread safety.

```python
import hashlib
import sqlite3
import threading
from urllib.parse import urlparse

# Create a lock for thread safety
lock = threading.Lock()

class URLShortener:
    def __init__(self, db_name):
        self.db_name = db_name
        self.conn = sqlite3.connect(db_name)
        self.cursor = self.conn.cursor()
        self.cursor.execute(
            """CREATE TABLE IF NOT EXISTS urls
            (short_url TEXT PRIMARY KEY, long_url TEXT)"""
        )
        self.conn.commit()

    def shorten_url(self, long_url):
        # Validate the input URL format
        if not self.validate_url(long_url):
            raise ValueError("Invalid URL format")

        # Generate a shortened URL
        short_url = self.generate_short_url(long_url)

        # Check for collisions
        while self.check_collision(short_url):
            short_url = self.generate_short_url(long_url)

        # Store the shortened URL and original URL in the database
        with lock:
            self.cursor.execute(
                """INSERT INTO urls (short_url, long_url) VALUES (?, ?)""",
                (short_url, long_url),
            )
            self.conn.commit()

        return short_url

    def lookup_short_url(self, short_url):
        with lock:
            self.cursor.execute(
                """SELECT long_url FROM urls WHERE short_url = ?""", (short_url,)
            )
            result = self.cursor.fetchone()
            if result:
                return result[0]
            else:
                return None

    def validate_url(self, url):
        try:
            result = urlparse(url)
            return all([result.scheme, result.netloc])
        except ValueError:
            return False

    def generate_short_url(self, long_url):
        # Generate a hash-based shortened URL
        hash_object = hashlib.sha256(long_url.encode())
        return hash_object.hexdigest()[:6]

    def check_collision(self, short_url):
        with lock:
            self.cursor.execute(
                """SELECT * FROM urls WHERE short_url = ?""", (short_url,)
            )
            result = self.cursor.fetchone()
            return result is not None

# Example usage
url_shortener = URLShortener("urls.db")
long_url = "https://www.example.com/very/long/url"
short_url = url_shortener.shorten_url(long_url)
print(f"Short URL: {short_url}")
print(f"Original URL: {url_shortener.lookup_short_url(short_url)}")
```

**Implementation Code**

```python
import hashlib
import sqlite3
import threading
from urllib.parse import urlparse
import logging

# Create a logger for logging
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

# Create a lock for thread safety
lock = threading.Lock()

class URLShortener:
    def __init__(self, db_name):
        self.db_name = db_name
        self.conn = sqlite3.connect(db_name)
        self.cursor = self.conn.cursor()
        self.cursor.execute(
            """CREATE TABLE IF NOT EXISTS urls
            (short_url TEXT PRIMARY KEY, long_url TEXT)"""
        )
        self.conn.commit()
        self.cache = {}  # Cache for lookup

    def shorten_url(self, long_url):
        # Validate the input URL format
        if not self.validate_url(long_url):
            raise ValueError("Invalid URL format")

        # Generate a shortened URL
        short_url = self.generate_short_url(long_url)

        # Check for collisions
        while self.check_collision(short_url):
            short_url = self.generate_short_url(long_url)

        # Store the shortened URL and original URL in the database
        with lock:
            self.cursor.execute(
                """INSERT INTO urls (short_url, long_url) VALUES (?, ?)""",
                (short_url, long_url),
            )
            self.conn.commit()
            self.cache[short_url] = long_url  # Update cache

        return short_url

    def lookup_short_url(self, short_url):
        # Check cache first
        if short_url in self.cache:
            return self.cache[short_url]

        with lock:
            self.cursor.execute(
                """SELECT long_url FROM urls WHERE short_url = ?""", (short_url,)
            )
            result = self.cursor.fetchone()
            if result:
                self.cache[short_url] = result[0]  # Update cache
                return result[0]
            else:
                return None

    def validate_url(self, url):
        try:
            result = urlparse(url)
            return all([result.scheme, result.netloc])
        except ValueError:
            return False

    def generate_short_url(self, long_url):
        # Generate a hash-based shortened URL using SHA-512
        hash_object = hashlib.sha512(long_url.encode())
        return hash_object.hexdigest()[:6]

    def check_collision(self, short_url):
        with lock:
            self.cursor.execute(
                """SELECT * FROM urls WHERE short_url = ?""", (short_url,)
            )
            result = self.cursor.fetchone()
            return result is not None

    def close_connection(self):
        self.conn.close()
```

```python
import hashlib
import sqlite3
import threading
from urllib.parse import urlparse
import logging

# Create a logger for logging
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

# Create a lock for thread safety
lock = threading.Lock()

class URLShortener:
    def __init__(self, db_name):
        self.db_name = db_name
        self.conn = sqlite3.connect(db_name)
        self.cursor = self.conn.cursor()
        self.cursor.execute(
            """CREATE TABLE IF NOT EXISTS urls
            (short_url TEXT PRIMARY KEY, long_url TEXT)"""
        )
        self.conn.commit()
        self.cache = {}  # Cache for lookup

    def shorten_url(self, long_url):
        # Validate the input URL format
        if not self.validate_url(long_url):
            raise ValueError("Invalid URL format")

        # Generate a shortened URL
        short_url = self.generate_short_url(long_url)

        # Check for collisions
        while self.check_collision(short_url):
            short_url = self.generate_short_url(long_url)

        # Store the shortened URL and original URL in the database
        with lock:
            self.cursor.execute(
                """INSERT INTO urls (short_url, long_url) VALUES (?, ?)""",
                (short_url, long_url),
            )
            self.conn.commit()
            self.cache[short_url] = long_url  # Update cache

        return short_url

    def lookup_short_url(self, short_url):
        # Check cache first
        if short_url in self.cache:
            return self.cache[short_url]

        with lock:
            self.cursor.execute(
                """SELECT long_url FROM urls WHERE short_url = ?""", (short_url,)
            )
            result = self.cursor.fetchone()
            if result:
                self.cache[short_url] = result[0]  # Update cache
                return result[0]
            else:
                return None

    def validate_url(self, url):
        try:
            result = urlparse(url)
            return all([result.scheme, result.netloc])
        except ValueError:
            return False

    def generate_short_url(self, long_url):
        # Generate a hash-based shortened URL using SHA-512
        hash_object = hashlib.sha512(long_url.encode())
        return hash_object.hexdigest()[:6]

    def check_collision(self, short_url):
        with lock:
            self.cursor.execute(
                """SELECT * FROM urls WHERE short_url = ?""", (short_url,)
            )
            result = self.cursor.fetchone()
            return result is not None

    def close_connection(self):
        self.conn.close()
```
