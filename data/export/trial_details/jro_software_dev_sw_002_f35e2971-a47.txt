import threading
import time
from functools import wraps

class RateLimiter:
    def __init__(self, rate, burst_size):
        """
        Initialize the rate limiter.

        :param rate: The rate at which tokens are added to the bucket (requests per second).
        :param burst_size: The maximum number of tokens that can be in the bucket at any given time.
        """
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()
        self.lock = threading.Lock()

    def acquire_token(self):
        """
        Attempt to acquire a token from the bucket.

        :return: True if the token was acquired, False otherwise.
        """
        with self.lock:
            current_time = time.time()
            elapsed_time = current_time - self.last_update
            self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)
            self.last_update = current_time

            if self.tokens < 1:
                return False
            else:
                self.tokens -= 1
                return True

    def release_token(self):
        """
        Release a token back to the bucket.
        """
        with self.lock:
            self.tokens = min(self.burst_size, self.tokens + 1)

def rate_limiter_decorator(rate_limiter):
    """
    A decorator that applies rate limiting to a function.

    :param rate_limiter: The rate limiter to use.
    :return: A decorator that applies rate limiting to a function.
    """
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            if not rate_limiter.acquire_token():
                raise Exception("Rate limit exceeded")
            try:
                return func(*args, **kwargs)
            finally:
                rate_limiter.release_token()
        return wrapper
    return decorator

class RateLimiterFactory:
    def __init__(self):
        """
        Initialize the rate limiter factory.
        """
        self.rate_limiters = {}
        self.lock = threading.Lock()

    def create_rate_limiter(self, name, rate, burst_size):
        """
        Create a new rate limiter.

        :param name: The name of the rate limiter.
        :param rate: The rate at which tokens are added to the bucket (requests per second).
        :param burst_size: The maximum number of tokens that can be in the bucket at any given time.
        :return: The created rate limiter.
        """
        with self.lock:
            if name in self.rate_limiters:
                raise Exception("Rate limiter already exists")
            rate_limiter = RateLimiter(rate, burst_size)
            self.rate_limiters[name] = rate_limiter
            return rate_limiter

    def get_rate_limiter(self, name):
        """
        Get an existing rate limiter.

        :param name: The name of the rate limiter.
        :return: The rate limiter, or None if it does not exist.
        """
        with self.lock:
            return self.rate_limiters.get(name)

    def delete_rate_limiter(self, name):
        """
        Delete a rate limiter.

        :param name: The name of the rate limiter.
        """
        with self.lock:
            if name in self.rate_limiters:
                del self.rate_limiters[name]

# Example usage:
