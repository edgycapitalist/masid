5. **Shortener**: The main entry point for the library, responsible for shortening URLs and performing reverse lookups.

* **HashGenerator**: Utilizes the `hashlib` library to generate a unique hash for the input URL.

* **URLMapper**: Stores the mapping between shortened URLs and original URLs in a thread-safe dictionary.
* **Shortener**: The main entry point for the library, responsible for shortening URLs and performing reverse lookups.

import hashlib
import threading
from urllib.parse import urlparse

class URLValidator:
    def __init__(self):
        pass

    def validate(self, url):
        try:
            result = urlparse(url)
            return all([result.scheme, result.netloc])
        except ValueError:
            return False

class HashGenerator:
    def __init__(self):
        self.salt = 0

    def generate(self, url):
        hash_object = hashlib.sha256()
        hash_object.update((url + str(self.salt)).encode('utf-8'))
        return hash_object.hexdigest()[:6]

class CollisionDetector:
    def __init__(self):
        self.count = 0

    def detect(self, hash_value):
        self.count += 1
        if self.count > 1000:
            raise Exception("Collision detection failed")

class URLMapper:
    def __init__(self):
        self.lock = threading.Lock()
        self.url_map = {}

    def store(self, short_url, original_url):
        with self.lock:
            self.url_map[short_url] = original_url

    def lookup(self, short_url):
        with self.lock:
            return self.url_map.get(short_url)

class Shortener:
    def __init__(self):
        self.url_validator = URLValidator()
        self.hash_generator = HashGenerator()
        self.collision_detector = CollisionDetector()
        self.url_mapper = URLMapper()

    def shorten(self, url):
        if not self.url_validator.validate(url):
            raise ValueError("Invalid URL")

        hash_value = self.hash_generator.generate(url)
        while self.collision_detector.detect(hash_value):
            self.hash_generator.salt += 1
            hash_value = self.hash_generator.generate(url)

        short_url = hash_value
        self.url_mapper.store(short_url, url)
        return short_url

    def reverse_lookup(self, short_url):
        return self.url_mapper.lookup(short_url)

import hashlib
import threading
from urllib.parse import urlparse
import logging
import argparse

# Configure logging
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class URLValidator:
    def __init__(self):
        pass

    def validate(self, url):
        try:
            result = urlparse(url)
            return all([result.scheme, result.netloc])
        except ValueError:
            return False

class HashGenerator:
    def __init__(self):
        self.salt = 0

    def generate(self, url):
        try:
            hash_object = hashlib.sha256()
            hash_object.update((url + str(self.salt)).encode('utf-8'))
            return hash_object.hexdigest()[:6]
        except Exception as e:
            logger.error(f"Error generating hash: {e}")
            raise

class CollisionDetector:
    def __init__(self):
        self.count = 0
        self.max_retries = 1000

    def detect(self, hash_value, url_mapper):
        self.count += 1
        if self.count > self.max_retries:
            raise Exception("Collision detection failed")
        return hash_value not in url_mapper.url_map

class URLMapper:
    def __init__(self):
        self.lock = threading.Lock()
        self.url_map = {}

    def store(self, short_url, original_url):
        with self.lock:
            try:
                self.url_map[short_url] = original_url
            except Exception as e:
                logger.error(f"Error storing URL mapping: {e}")
                raise

    def lookup(self, short_url):
        with self.lock:
            try:
                return self.url_map.get(short_url)
            except Exception as e:
                logger.error(f"Error looking up URL mapping: {e}")
                raise

class Shortener:
    def __init__(self):
        self.url_validator = URLValidator()
        self.hash_generator = HashGenerator()
        self.collision_detector = CollisionDetector()
        self.url_mapper = URLMapper()

    def shorten(self, url):
        if not self.url_validator.validate(url):
            raise ValueError("Invalid URL")

        hash_value = self.hash_generator.generate(url)
        while not self.collision_detector.detect(hash_value, self.url_mapper):
            self.hash_generator.salt += 1
            hash_value = self.hash_generator.generate(url)

        short_url = hash_value
        self.url_mapper.store(short_url, url)
        logger.info(f"Shortened URL: {url} -> {short_url}")
        return short_url

    def reverse_lookup(self, short_url):
        try:
            original_url = self.url_mapper.lookup(short_url)
            if original_url:
                logger.info(f"Original URL: {short_url} -> {original_url}")
                return original_url
            else:
                raise ValueError("Short URL not found")
        except Exception as e:
            logger.error(f"Error performing reverse lookup: {e}")
            raise

def main():
    parser = argparse.ArgumentParser(description="URL Shortener")
    parser.add_argument("--log-level", help="Set the logging level", choices=["DEBUG", "INFO", "WARNING", "ERROR", "CRITICAL"])
    args = parser.parse_args()

    if args.log_level:
        logging.basicConfig(level=getattr(logging, args.log_level.upper()))

    shortener = Shortener()
    long_url = "https://www.example.com"
    short_url = shortener.shorten(long_url)
    print(f"Short URL: {short_url}")

    original_url = shortener.reverse_lookup(short_url)
    print(f"Original URL: {original_url}")

if __name__ == "__main__":
    main()
