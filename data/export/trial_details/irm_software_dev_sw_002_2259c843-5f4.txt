### Overview

The rate limiter module is designed to provide a flexible and thread-safe way to limit the rate at which requests are made to a system. The module uses the token bucket algorithm to manage the rate limiting.

### Components

*   **Token Bucket Algorithm**: The token bucket algorithm is used to manage the rate limiting. The algorithm works by adding tokens to a bucket at a constant rate. When a request is made, a token is removed from the bucket. If the bucket is empty, the request is blocked until a token is available.
*   **Rate Limiter Class**: The `RateLimiter` class is the core component of the module. It is responsible for managing the token bucket and providing methods for checking the rate limit.
*   **Named Rate Limiters**: The module supports multiple named rate limiters. Each named rate limiter has its own token bucket and can be configured independently.
*   **Decorator**: The module provides a decorator that can be used to easily apply rate limiting to functions.

### Interfaces

*   **`RateLimiter` Class**: The `RateLimiter` class has the following methods:
    *   `__init__(rate, burst_size)`: Initializes the rate limiter with the specified rate and burst size.
    *   `allow()`: Checks if a request is allowed based on the current rate limit.

    *   `allow()`: Checks if a request is allowed based on the current rate limit.
    *   `deny()`: Blocks the current request until a token is available.

### Data Flow

The data flow of the module is as follows:

1.  A request is made to the system.
2.  The request is passed to the `RateLimiter` class to check if it is allowed based on the current rate limit.
3.  If the request is allowed, the `RateLimiter` class removes a token from the token bucket and allows the request to proceed.
4.  If the request is not allowed, the `RateLimiter` class blocks the request until a token is available.

import threading
import time

class RateLimiter:
    def __init__(self, rate, burst_size):
        """
        Initializes the rate limiter with the specified rate and burst size.

        Args:
            rate (float): The rate at which tokens are added to the bucket (tokens per second).
            burst_size (int): The maximum number of tokens in the bucket.
        """
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()
        self.lock = threading.Lock()

    def allow(self):
        """
        Checks if a request is allowed based on the current rate limit.

        Returns:
            bool: True if the request is allowed, False otherwise.
        """
        with self.lock:
            now = time.time()
            elapsed = now - self.last_update
            self.tokens = min(self.burst_size, self.tokens + elapsed * self.rate)
            self.last_update = now
            if self.tokens < 1:
                return False
            self.tokens -= 1
            return True

    def deny(self):
        """
        Blocks the current request until a token is available.
        """
        with self.lock:
            while self.tokens < 1:
                time.sleep(0.01)  # sleep for 10ms
                now = time.time()
                elapsed = now - self.last_update
                self.tokens = min(self.burst_size, self.tokens + elapsed * self.rate)
                self.last_update = now

class NamedRateLimiter:
    def __init__(self, name, rate, burst_size):
        """
        Initializes the named rate limiter with the specified name, rate, and burst size.

        Args:
            name (str): The name of the rate limiter.
            rate (float): The rate at which tokens are added to the bucket (tokens per second).
            burst_size (int): The maximum number of tokens in the bucket.
        """
        self.name = name
        self.rate_limiter = RateLimiter(rate, burst_size)

    def allow(self):
        """
        Checks if a request is allowed based on the current rate limit.

        Returns:
            bool: True if the request is allowed, False otherwise.
        """
        return self.rate_limiter.allow()

    def deny(self):
        """
        Blocks the current request until a token is available.
        """
        self.rate_limiter.deny()

def rate_limited(rate, burst_size):
    """
    Decorator to apply rate limiting to a function.

    Args:
        rate (float): The rate at which tokens are added to the bucket (tokens per second).
        burst_size (int): The maximum number of tokens in the bucket.

    Returns:
        function: The decorated function.
    """
    rate_limiter = RateLimiter(rate, burst_size)

    def decorator(func):
        def wrapper(*args, **kwargs):
            if not rate_limiter.allow():
                rate_limiter.deny()
            return func(*args, **kwargs)
        return wrapper
    return decorator

import threading
import time
import logging

logging.basicConfig(level=logging.INFO)

class RateLimiter:
    def __init__(self, rate, burst_size):
        """
        Initializes the rate limiter with the specified rate and burst size.

        Args:
            rate (float): The rate at which tokens are added to the bucket (tokens per second).
            burst_size (int): The maximum number of tokens in the bucket.
        """
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()
        self.lock = threading.Lock()

    def allow(self):
        """
        Checks if a request is allowed based on the current rate limit.

        Returns:
            bool: True if the request is allowed, False otherwise.
        """
        with self.lock:
            now = time.time()
            elapsed = now - self.last_update
            self.tokens = min(self.burst_size, self.tokens + elapsed * self.rate)
            self.last_update = now
            if self.tokens < 1:
                return False
            self.tokens -= 1
            return True

    def deny(self):
        """
        Blocks the current request until a token is available.
        """
        with self.lock:
            while self.tokens < 1:
                time.sleep(0.01)  # sleep for 10ms
                now = time.time()
                elapsed = now - self.last_update
                self.tokens = min(self.burst_size, self.tokens + elapsed * self.rate)
                self.last_update = now

    def reset(self):
        """
        Resets the token bucket.
        """
        with self.lock:
            self.tokens = self.burst_size
            self.last_update = time.time()

        Args:
            name (str): The name of the rate limiter.

import threading
import time
import logging
from functools import wraps

logging.basicConfig(level=logging.INFO)

class RateLimiter:
    def __init__(self, rate, burst_size):
        """
        Initializes the rate limiter with the specified rate and burst size.

        Args:
            rate (float): The rate at which tokens are added to the bucket (tokens per second).
            burst_size (int): The maximum number of tokens in the bucket.
        """
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()
        self.lock = threading.Lock()

    def allow(self):
        """
        Checks if a request is allowed based on the current rate limit.

        Returns:
            bool: True if the request is allowed, False otherwise.
        """
        with self.lock:
            now = time.time()
            elapsed = now - self.last_update
            self.tokens = min(self.burst_size, self.tokens + elapsed * self.rate)
            self.last_update = now
            if self.tokens < 1:
                return False
            self.tokens -= 1
            return True

    def deny(self):
        """
        Blocks the current request until a token is available.
        """
        with self.lock:
            while self.tokens < 1:
                time.sleep(0.01)  # sleep for 10ms
                now = time.time()
                elapsed = now - self.last_update
                self.tokens = min(self.burst_size, self.tokens + elapsed * self.rate)
                self.last_update = now

    def reset(self):
        """
        Resets the token bucket.
        """
        with self.lock:
            self.tokens = self.burst_size
            self.last_update = time.time()
