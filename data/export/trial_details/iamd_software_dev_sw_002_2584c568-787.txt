import threading
import time
from functools import wraps

class TokenBucket:
    def __init__(self, rate, burst_size):
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_refill = time.time()

    def consume(self, amount=1):
        if self.tokens >= amount:
            self.tokens -= amount
            return True
        return False

    def refill(self):
        now = time.time()
        elapsed = now - self.last_refill
        self.last_refill = now
        self.tokens = min(self.burst_size, self.tokens + elapsed * self.rate)

class RateLimiter:
    def __init__(self, rate, burst_size):
        self.token_bucket = TokenBucket(rate, burst_size)
        self.lock = threading.Lock()

    def is_allowed(self):
        with self.lock:
            if self.token_bucket.consume():
                return True
            return False

    def get_rate(self):
        return self.token_bucket.rate

    def get_burst_size(self):
        return self.token_bucket.burst_size

class NamedRateLimiters:
    def __init__(self):
        self.rate_limiters = {}
        self.lock = threading.Lock()

    def get_rate_limiter(self, name):
        with self.lock:
            if name not in self.rate_limiters:
                raise ValueError("Rate limiter not found")
            return self.rate_limiters[name]

    def set_rate_limiter(self, name, rate_limiter):
        with self.lock:
            self.rate_limiters[name] = rate_limiter

            if rate_limiter.is_allowed():

        return wrapper

class TokenBucket:
    """
    Token bucket implementation for rate limiting.

    Attributes:
    rate (float): The rate at which tokens are refilled per second.
    burst_size (int): The maximum number of tokens in the bucket.
    tokens (int): The current number of tokens in the bucket.
    last_refill (float): The time of the last refill in seconds since the epoch.
    """

    def __init__(self, rate, burst_size):
        """
        Initialize the token bucket with the given rate and burst size.

        Args:
        rate (float): The rate at which tokens are refilled per second.
        burst_size (int): The maximum number of tokens in the bucket.
        """
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_refill = time.time()

    def consume(self, amount=1):
        """
        Consume the given number of tokens from the bucket.

        Args:
        amount (int, optional): The number of tokens to consume. Defaults to 1.

        Returns:
        bool: True if the tokens were consumed, False otherwise.
        """
        if self.tokens >= amount:
            self.tokens -= amount
            return True
        return False

    def refill(self):
        """
        Refill the token bucket based on the elapsed time since the last refill.
        """
        now = time.time()
        elapsed = now - self.last_refill
        self.last_refill = now

import pytest
from solution import RateLimiter, NamedRateLimiters, rate_limited
import time
import threading

def test_rate_limiter():
    rate_limiter = RateLimiter(5, 10)
    assert rate_limiter.is_allowed()
    assert rate_limiter.is_allowed()
    assert rate_limiter.is_allowed()
    assert rate_limiter.is_allowed()
    assert rate_limiter.is_allowed()
    assert not rate_limiter.is_allowed()
    time.sleep(1)
    assert rate_limiter.is_allowed()

def test_named_rate_limiters():
    named_rate_limiters = NamedRateLimiters()
    rate_limiter = RateLimiter(5, 10)
    named_rate_limiters.set_rate_limiter("example", rate_limiter)
    assert named_rate_limiters.get_rate_limiter("example") == rate_limiter

def test_rate_limited_decorator():
    rate_limiter = RateLimiter(1, 1)
    @rate_limited(rate_limiter)
    def example_function():
        pass
    example_function()
    with pytest.raises(Exception):
        example_function()

def test_rate_limiter_concurrent_access():
    rate_limiter = RateLimiter(5, 10)
    def test_function():
        for _ in range(10):
            rate_limiter.is_allowed()
    threads = []
    for _ in range(10):
        thread = threading.Thread(target=test_function)
        thread.start()
        threads.append(thread)
    for thread in threads:
        thread.join()
    assert rate_limiter.token_bucket.tokens <= rate_limiter.token_bucket.burst_size

def test_rate_limiter_reset():
    rate_limiter = RateLimiter(5, 10)
    assert rate_limiter.is_allowed()
    assert not rate_limiter.is_allowed()
    rate_limiter.reset()
    assert rate_limiter.is_allowed()

def test_rate_limiter_edge_cases():
    rate_limiter = RateLimiter(0, 10)
    assert not rate_limiter.is_allowed()

    rate_limiter = RateLimiter(5, 0)
    assert not rate_limiter.is_allowed()

    rate_limiter = RateLimiter(5, 10)
    rate_limiter.reset()
    assert rate_limiter.is_allowed()
