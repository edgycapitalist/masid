### Overview

The data pipeline is designed to read CSV files, apply a chain of transformations, and output the results to CSV or JSON. The pipeline includes configurable schema validation, a plugin system for custom transformations, and clear error reporting for malformed data.

### Components

1. **CSV Reader**: Responsible for reading CSV files and performing schema validation.
2. **Transformation Chain**: Applies a series of transformations to the data, including filtering, mapping, and aggregation.
3. **Plugin System**: Allows for custom transformations to be added to the pipeline.
4. **Output Writer**: Writes the transformed data to CSV or JSON.

### Data Flow

1. The CSV Reader reads a CSV file and performs schema validation.
2. The validated data is passed to the Transformation Chain.
3. The Transformation Chain applies a series of transformations to the data.
4. The transformed data is passed to the Output Writer.

    def __init__(self, filter_func):
        self.filter_func = filter_func

    def apply_transformation(self, data):
        return data[self.filter_func(data)]

class MapTransformation(Transformation):
    def __init__(self, map_func):
        self.map_func = map_func

    def apply_transformation(self, data):

    def __init__(self, aggregation_func):

class PluginSystem:
    def __init__(self):
        self.plugins = []

    def register_plugin(self, plugin):
        self.plugins.append(plugin)

    def apply_plugins(self, data):
        for plugin in self.plugins:
            data = plugin.apply_transformation(data)
        return data

    def write_data(self, data, output_format):
        if output_format == "csv":
            data.to_csv("output.csv", index=False)
        elif output_format == "json":
            data.to_json("output.json", orient="records")
        else:
            raise ValueError("Invalid output format")

class DataPipeline:

import pandas as pd
import json
from abc import ABC, abstractmethod

class CSVReader:
    def __init__(self, schema):
        self.schema = schema

    def read_csv(self, file_path):
        try:
            data = pd.read_csv(file_path)
            self.validate_schema(data)
            return data
        except Exception as e:
            raise ValueError(f"Error reading CSV file: {e}")

    def validate_schema(self, data):
        # Perform schema validation
        for column, dtype in self.schema.items():
            if column not in data.columns:
                raise ValueError(f"Column '{column}' not found in CSV file")
            if not all(isinstance(x, dtype) for x in data[column]):
                raise ValueError(f"Column '{column}' has incorrect data type")

class Transformation(ABC):
    @abstractmethod
    def apply_transformation(self, data):
        pass

class FilterTransformation(Transformation):
    def __init__(self, filter_func):
        self.filter_func = filter_func

    def apply_transformation(self, data):
        return data[self.filter_func(data)]

class MapTransformation(Transformation):
    def __init__(self, map_func):
        self.map_func = map_func

    def apply_transformation(self, data):
        return data.apply(self.map_func)

class AggregationTransformation(Transformation):
    def __init__(self, aggregation_func):
        self.aggregation_func = aggregation_func

    def apply_transformation(self, data):
        return data.agg(self.aggregation_func)

class PluginSystem:
    def __init__(self):
        self.plugins = []

    def register_plugin(self, plugin):
        if plugin not in self.plugins:
            self.plugins.append(plugin)

    def apply_plugins(self, data):
        for plugin in self.plugins:
            data = plugin.apply_transformation(data)
        return data

class OutputWriter:
    def write_data(self, data, output_format):
        if output_format == "csv":
            if not data.empty:
                data.to_csv("output.csv", index=False)
            else:
                raise ValueError("No data to write to CSV file")
        elif output_format == "json":
            if not data.empty:
                data.to_json("output.json", orient="records")
            else:
                raise ValueError("No data to write to JSON file")
        else:
            raise ValueError("Invalid output format")

class DataPipeline:
    def __init__(self, csv_reader, transformation_chain, plugin_system, output_writer):

### Overview

The data pipeline is designed to read CSV files, apply a chain of transformations, and output the results to CSV or JSON. The pipeline includes configurable schema validation, a plugin system for custom transformations, and clear error reporting for malformed data.

import pandas as pd
import json
from abc import ABC, abstractmethod
import logging

# Configure logging
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class CSVReader:
    def __init__(self, schema):
        self.schema = schema

    def read_csv(self, file_path):
        try:
            data = pd.read_csv(file_path)
            self.validate_schema(data)
            return data
        except Exception as e:
            logger.error(f"Error reading CSV file: {e}")
            raise ValueError(f"Error reading CSV file: {e}")

    def validate_schema(self, data):
        # Perform schema validation
        for column, dtype in self.schema.items():
            if column not in data.columns:
                logger.error(f"Column '{column}' not found in CSV file")
                raise ValueError(f"Column '{column}' not found in CSV file")
            if not all(isinstance(x, dtype) for x in data[column]):
                logger.error(f"Column '{column}' has incorrect data type")
                raise ValueError(f"Column '{column}' has incorrect data type")

class Transformation(ABC):
    @abstractmethod
    def apply_transformation(self, data):
        pass

class FilterTransformation(Transformation):
    def __init__(self, filter_func):
        self.filter_func = filter_func

    def apply_transformation(self, data):
        return data[self.filter_func(data)]

class MapTransformation(Transformation):
    def __init__(self, map_func):
        self.map_func = map_func

    def apply_transformation(self, data):
        return data.apply(self.map_func)

class AggregationTransformation(Transformation):
    def __init__(self, aggregation_func):
        self.aggregation_func = aggregation_func

    def apply_transformation(self, data):
        return data.agg(self.aggregation_func)

class PluginSystem:
    def __init__(self):
        self.plugins = []

    def register_plugin(self, plugin):
        if plugin not in self.plugins:
            self.plugins.append(plugin)

    def apply_plugins(self, data):
        for plugin in self.plugins:
            data = plugin.apply_transformation(data)
        return data

class OutputWriter:
    def write_data(self, data, output_format):
        if output_format == "csv":
            if not data.empty:
                data.to_csv("output.csv", index=False)
            else:
                logger.error("No data to write to CSV file")
                raise ValueError("No data to write to CSV file")
        elif output_format == "json":
            if not data.empty:
                data.to_json("output.json", orient="records")
            else:
                logger.error("No data to write to JSON file")
                raise ValueError("No data to write to JSON file")
        else:
            logger.error("Invalid output format")
            raise ValueError("Invalid output format")

class DataPipeline:
    def __init__(self, csv_reader, transformation_chain, plugin_system, output_writer):
        self.csv_reader = csv_reader
        self.transformation_chain = transformation_chain
        self.plugin_system = plugin_system
        self.output_writer = output_writer

    def run(self, file_path, output_format):
        try:
            data = self.csv_reader.read_csv(file_path)
            data = self.transformation_chain.apply_transformation(data)
            data = self.plugin_system.apply_plugins(data)
            self.output_writer.write_data(data, output_format)
        except Exception as e:
            logger.error(f"Error running data pipeline: {e}")
            raise ValueError(f"Error running data pipeline: {e}")

```python
import pandas as pd
import json
from abc import ABC, abstractmethod
import logging

# Configure logging
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class CSVReader:
    def __init__(self, schema):
        self.schema = schema

    def read_csv(self, file_path):
        try:
            data = pd.read_csv(file_path)
            self.validate_schema(data)
            return data
        except Exception as e:
            logger.error(f"Error reading CSV file: {e}")
            raise ValueError(f"Error reading CSV file: {e}")

    def validate_schema(self, data):
        # Perform schema validation
        for column, dtype in self.schema.items():
            if column not in data.columns:
                logger.error(f"Column '{column}' not found in CSV file")
                raise ValueError(f"Column '{column}' not found in CSV file")
            if not all(isinstance(x, dtype) for x in data[column]):
                logger.error(f"Column '{column}' has incorrect data type")
                raise ValueError(f"Column '{column}' has incorrect data type")

class Transformation(ABC):
    @abstractmethod
    def apply_transformation(self, data):
        pass

class FilterTransformation(Transformation):
    def __init__(self, filter_func):
        self.filter_func = filter_func

    def apply_transformation(self, data):
        return data[self.filter_func(data)]

class MapTransformation(Transformation):
    def __init__(self, map_func):
        self.map_func = map_func

    def apply_transformation(self, data):
        return data.apply(self.map_func)

class AggregationTransformation(Transformation):
    def __init__(self, aggregation_func):
        self.aggregation_func = aggregation_func

    def apply_transformation(self, data):
        return data.agg(self.aggregation_func)

class PluginSystem:
    def __init__(self):
        self.plugins = []

    def register_plugin(self, plugin):
        if plugin not in self.plugins:
            self.plugins.append(plugin)
            logger.info(f"Plugin '{plugin.__class__.__name__}' registered")

    def apply_plugins(self, data):
        for plugin in self.plugins:
            data = plugin.apply_transformation(data)
            logger.info(f"Applied plugin '{plugin.__class__.__name__}'")
        return data

class OutputWriter:
    def write_data(self, data, output_format, output_file):
        if output_format == "csv":
            if not data.empty:
                data.to_csv(output_file, index=False)
                logger.info(f"Data written to CSV file '{output_file}'")
            else:
                logger.error("No data to write to CSV file")
                raise ValueError("No data to write to CSV file")
        elif output_format == "json":
            if not data.empty:
                data.to_json(output_file, orient="records")
                logger.info(f"Data written to JSON file '{output_file}'")
            else:
                logger.error("No data to write to JSON file")
                raise ValueError("No data to write to JSON file")
        else:
            logger.error("Invalid output format")
            raise ValueError("Invalid output format")

class DataPipeline:
    def __init__(self, csv_reader, transformation_chain, plugin_system, output_writer):
        self.csv_reader = csv_reader
        self.transformation_chain = transformation_chain
        self.plugin_system = plugin_system
        self.output_writer = output_writer

    def run(self, file_path, output_format, output_file):
        try:
            data = self.csv_reader.read_csv(file_path)
            data = self.transformation_chain.apply_transformation(data)
            data = self.plugin_system.apply_plugins(data)
            self.output_writer.write_data(data, output_format, output_file)
            logger.info("Data pipeline completed successfully")
        except Exception as e:
            logger.error(f"Error running data pipeline: {e}")
            raise ValueError(f"Error running data pipeline: {e}")

# Example usage
csv_reader = CSVReader({"name": str, "age": int})
transformation_chain = FilterTransformation(lambda x: x["age"] > 18)
plugin_system = PluginSystem()
output_writer = OutputWriter()

plugin_system.register_plugin(MapTransformation(lambda x: x["name"].upper()))
plugin_system.register_plugin(AggregationTransformation("mean"))

data_pipeline = DataPipeline(csv_reader, transformation_chain, plugin_system, output_writer)
data_pipeline.run("input.csv", "csv", "output.csv")
