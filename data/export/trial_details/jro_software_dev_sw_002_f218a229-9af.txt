import threading
import time
from functools import wraps

class TokenBucket:
    def __init__(self, rate, burst_size):
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()

    def refill(self):
        now = time.time()
        elapsed = now - self.last_update
        self.tokens = min(self.burst_size, self.tokens + elapsed * self.rate)
        self.last_update = now

    def consume(self, amount=1):
        self.refill()
        if self.tokens >= amount:
            self.tokens -= amount
            return True
        return False


class RateLimiter:
    def __init__(self, rate, burst_size):
        self.rate = rate
        self.burst_size = burst_size
        self.token_bucket = TokenBucket(rate, burst_size)
        self.lock = threading.Lock()

    def consume(self, amount=1):
        with self.lock:
            return self.token_bucket.consume(amount)


class RateLimiterDecorator:
    def __init__(self, rate_limiter):
        self.rate_limiter = rate_limiter

    def __call__(self, func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            if self.rate_limiter.consume():
                return func(*args, **kwargs)
            else:
                raise Exception("Rate limit exceeded")

        return wrapper


class RateLimiterManager:
    def __init__(self):
        self.rate_limiters = {}
        self.lock = threading.Lock()

    def get_rate_limiter(self, name):
        with self.lock:
            return self.rate_limiters.get(name)

    def add_rate_limiter(self, name, rate_limiter):
        with self.lock:
            self.rate_limiters[name] = rate_limiter


# Example usage
rate_limiter_manager = RateLimiterManager()
rate_limiter = RateLimiter(5, 10)  # 5 requests per second, burst size 10
rate_limiter_manager.add_rate_limiter("example", rate_limiter)

@RateLimiterDecorator(rate_limiter)

import threading
import time
from functools import wraps

class TokenBucket:
    def __init__(self, rate: float, burst_size: int):
        """
        Initialize the token bucket with the specified rate and burst size.

        Args:
        rate (float): The rate at which tokens are added to the bucket per second.
        burst_size (int): The maximum number of tokens the bucket can hold.
        """
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()

    def refill(self) -> None:
        """
        Refill the token bucket with the specified rate.
        """
        now = time.time()
        elapsed = now - self.last_update
        self.tokens = min(self.burst_size, self.tokens + elapsed * self.rate)
        self.last_update = now

    def consume(self, amount: int = 1) -> bool:
        """
        Consume a token from the bucket and return True if successful, False otherwise.

        Args:
        amount (int): The number of tokens to consume. Defaults to 1.
        """
        self.refill()
        if self.tokens >= amount:

        return False

import threading
import time
from functools import wraps

class TokenBucket:
    def __init__(self, rate: float, burst_size: int):
        """
        Initialize the token bucket with the specified rate and burst size.

        Args:
        rate (float): The rate at which tokens are added to the bucket per second.
        burst_size (int): The maximum number of tokens the bucket can hold.
        """
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()

    def refill(self) -> None:
        """
        Refill the token bucket with the specified rate.
        """
        now = time.time()
        elapsed = now - self.last_update
        self.tokens = min(self.burst_size, self.tokens + elapsed * self.rate)
        self.last_update = now

    def consume(self, amount: int = 1) -> bool:
        """
        Consume a token from the bucket and return True if successful, False otherwise.

        Args:
        amount (int): The number of tokens to consume. Defaults to 1.
        """
        self.refill()
        if self.tokens >= amount:
            self.tokens -= amount
            return True
        return False

    def leaky_bucket_algorithm(self) -> None:
        """
        Implement the leaky bucket algorithm to improve performance and accuracy.
        """
        self.tokens -= self.rate


class RateLimiter:
    def __init__(self, rate: float, burst_size: int):
        """
        Initialize the rate limiter with the specified rate and burst size.

        Args:
        rate (float): The rate at which tokens are added to the bucket per second.
        burst_size (int): The maximum number of tokens the bucket can hold.
        """
        self.rate = rate
        self.burst_size = burst_size
        self.token_bucket = TokenBucket(rate, burst_size)
        self.lock = threading.Lock()

    def consume(self, amount: int = 1) -> bool:
        """
        Consume a token from the bucket and return True if successful, False otherwise.

        Args:
        amount (int): The number of tokens to consume. Defaults to 1.
        """
        with self.lock:
            return self.token_bucket.consume(amount)

    def set_rate(self, rate: float) -> None:
        """
        Set the rate of the rate limiter.

        Args:
        rate (float): The new rate of the rate limiter.
        """
        self.rate = rate
        self.token_bucket.rate = rate

    def set_burst_size(self, burst_size: int) -> None:
        """
        Set the burst size of the rate limiter.

        Args:
        burst_size (int): The new burst size of the rate limiter.
        """
        self.burst_size = burst_size
        self.token_bucket.burst_size = burst_size


class RateLimiterDecorator:
    def __init__(self, rate_limiter: RateLimiter):
        """
        Initialize the decorator with the specified rate limiter.

        Args:
        rate_limiter (RateLimiter): The rate limiter to use for decorating functions.
        """
        self.rate_limiter = rate_limiter

    def __call__(self, func):
        """
        Apply the rate limiter to the specified function.

        Args:
        func (function): The function to decorate with rate limiting.

        Returns:
        function: The decorated function.
        """
        @wraps(func)
        def wrapper(*args, **kwargs):
            if self.rate_limiter.consume():
                return func(*args, **kwargs)
            else:
                raise Exception("Rate limit exceeded")

        return wrapper

        Returns:
        RateLimiter: The rate limiter with the specified name, or None if not found.
        """
        with self.lock:
            return self.rate_limiters.get(name)
