import threading
import time
import functools

class TokenBucket:
    def __init__(self, rate, burst_size):
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()
        self.lock = threading.Lock()

    def allow(self):
        with self.lock:
            self.update()
            if self.tokens > 0:
                self.tokens -= 1
                return True
            return False

    def update(self):
        current_time = time.time()
        elapsed_time = current_time - self.last_update
        self.last_update = current_time
        self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)

class RateLimiter:
    def __init__(self, rate, burst_size):
        self.token_bucket = TokenBucket(rate, burst_size)

    def allow(self):
        return self.token_bucket.allow()

class NamedRateLimiter:
    def __init__(self, name, rate, burst_size):
        self.name = name
        self.token_bucket = TokenBucket(rate, burst_size)

    def allow(self):
        return self.token_bucket.allow()

def rate_limited(rate, burst_size):
    def decorator(func):
        token_bucket = TokenBucket(rate, burst_size)
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            if token_bucket.allow():

            raise Exception("Rate limit exceeded")
        return wrapper
    return decorator

class TokenBucket:

        """
        Initializes the token bucket with the specified rate and burst size.

        Args:
            rate (float): The rate at which tokens are added to the bucket per second.
            burst_size (int): The maximum number of tokens in the bucket.

        Raises:
            ValueError: If the rate or burst size is invalid (negative or zero).

        self.tokens = burst_size

        Returns:
            bool: True if the request is allowed, False otherwise.

        with self.lock:
            current_time = time.time()
            elapsed_time = current_time - self.last_update
            self.last_update = current_time
            self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)
            if self.tokens > 0:
                self.tokens -= 1
                return True
            return False

        Returns:
            int: The current number of tokens in the bucket.
        """
        with self.lock:
            return self.tokens
