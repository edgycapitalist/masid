import pandas as pd

class DataReader:
    def __init__(self, schema):
        self.schema = schema

    def read_csv(self, file_path):
        try:
            data = pd.read_csv(file_path)
            self.validate_schema(data)
            return data
        except Exception as e:
            raise ValueError(f"Failed to read CSV file: {e}")

    def validate_schema(self, data):
        # Perform schema validation using the provided schema
        for column, dtype in self.schema.items():
            if column not in data.columns:
                raise ValueError(f"Column '{column}' not found in data")
            if data[column].dtype != dtype:
                raise ValueError(f"Column '{column}' has incorrect data type")
```

from abc import ABC, abstractmethod

class Transformation(ABC):
    @abstractmethod
    def apply(self, data):
        pass

class FilteringTransformation(Transformation):
    def __init__(self, filter_func):
        self.filter_func = filter_func

    def apply(self, data):
        return data[self.filter_func(data)]

class MappingTransformation(Transformation):
    def __init__(self, mapping_func):
        self.mapping_func = mapping_func

    def apply(self, data):
        return data.apply(self.mapping_func)

class AggregationTransformation(Transformation):
    def __init__(self, aggregation_func):
        self.aggregation_func = aggregation_func

    def apply(self, data):
        return data.agg(self.aggregation_func)

class TransformationChain:
    def __init__(self):
        self.transformations = []

    def add_transformation(self, transformation):
        self.transformations.append(transformation)

    def apply(self, data):
        for transformation in self.transformations:
            data = transformation.apply(data)
        return data
```
