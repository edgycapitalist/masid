import pandas as pd
import logging

# Data Ingestion
class DataIngestion:
    def read_csv(self, file_path: str, schema: dict) -> pd.DataFrame:
        try:
            data = pd.read_csv(file_path)
            self.validate_schema(data, schema)
            return data
        except pd.errors.EmptyDataError:
            logging.error("Empty data file")
            return None
        except pd.errors.ParserError:
            logging.error("Error parsing data file")
            return None

    def validate_schema(self, data: pd.DataFrame, schema: dict):
        """Validates the schema of the ingested data"""
        for column, dtype in schema.items():
            if column not in data.columns:
                logging.error(f"Column '{column}' is missing from the data")
                return None
            if dtype == int:
                if not pd.api.types.is_integer_dtype(data[column]):
                    logging.error(f"Column '{column}' is not of type int")
                    return None
            elif dtype == str:
                if not pd.api.types.is_string_dtype(data[column]):
                    logging.error(f"Column '{column}' is not of type str")
                    return None
            else:
                logging.error(f"Unsupported data type '{dtype}' for column '{column}'")
                return None
        return data

# Transformation Chain
class TransformationChain:
    def apply_transformations(self, data: pd.DataFrame, transformations: list) -> pd.DataFrame:
        for transformation in transformations:
            data = transformation(data)
        return data

# Plugin System
class PluginSystem:
    def __init__(self):
        self.transformations = {}

    def register_transformation(self, name: str, transformation: callable):
        self.transformations[name] = transformation

    def get_transformation(self, name: str):
        return self.transformations.get(name)

# Data Output
class DataOutput:
    def write_to_csv(self, data: pd.DataFrame, file_path: str):
        data.to_csv(file_path, index=False)

    def write_to_json(self, data: pd.DataFrame, file_path: str):
        data.to_json(file_path, orient='records')

# Custom Transformation Example
def filter_by_age(data: pd.DataFrame) -> pd.DataFrame:

        try:
            data = pd.read_csv(file_path)

        except pd.errors.EmptyDataError:

            return None

    @cached(cache=TTLCache(maxsize=100, ttl=300))
    def validate_schema(self, data: pd.DataFrame, schema: Dict[str, type]) -> pd.DataFrame:
        for column, dtype in schema.items():
            if column not in data.columns:
                self.logger.error(f"Column '{column}' is missing from the data")
                return None
            if dtype == int:
                if not pd.api.types.is_integer_dtype(data[column]):
                    self.logger.error(f"Column '{column}' is not of type int")
                    return None
            elif dtype == str:
                if not pd.api.types.is_string_dtype(data[column]):
                    self.logger.error(f"Column '{column}' is not of type str")
                    return None
            else:
                self.logger.error(f"Unsupported data type '{dtype}' for column '{column}'")
                return None
        return data

        profile = {}
        for column in data.columns:
            profile[column] = {
                "count": data[column].count(),
                "mean": data[column].mean(),
                "std": data[column].std(),
                "min": data[column].min(),
                "max": data[column].max()
            }
        return profile

    def validate_data(self, data: pd.DataFrame) -> bool:
        for column in data.columns:
            if data[column].isnull().any():
                self.logger.error(f"Column '{column}' contains null values")
                return False
        return True

    @cached(cache=TTLCache(maxsize=100, ttl=300))
    def validate_schema(self, data: pd.DataFrame, schema: Dict[str, type]) -> pd.DataFrame:
        """
        Validates the schema of the ingested data.

        Args:
        - data (pd.DataFrame): The ingested data.
        - schema (Dict[str, type]): The schema of the data.

        Returns:
        - pd.DataFrame: The validated data.
        """
        for column, dtype in schema.items():
            if column not in data.columns:
                self.logger.error(f"Column '{column}' is missing from the data")
                return None
            if dtype == int:
                if not pd.api.types.is_integer_dtype(data[column]):
                    self.logger.error(f"Column '{column}' is not of type int")
                    return None
            elif dtype == str:
                if not pd.api.types.is_string_dtype(data[column]):
                    self.logger.error(f"Column '{column}' is not of type str")
                    return None
            else:
                self.logger.error(f"Unsupported data type '{dtype}' for column '{column}'")
                return None
        return data

    def profile_data(self, data: pd.DataFrame) -> Dict:
        """
        Performs data profiling.

        Args:
        - data (pd.DataFrame): The data to profile.

        Returns:
        - Dict: The data profile.
        """
        profile = {}
        for column in data.columns:
            profile[column] = {
                "count": data[column].count(),
                "mean": data[column].mean(),
                "std": data[column].std(),
                "min": data[column].min(),
                "max": data[column].max()
            }
        return profile

    def validate_data(self, data: pd.DataFrame) -> bool:
        """
        Performs data validation.

        Args:
        - data (pd.DataFrame): The data to validate.

        Returns:
        - bool: True if the data is valid, False otherwise.
        """
        for column in data.columns:
            if data[column].isnull().any():
                self.logger.error(f"Column '{column}' contains null values")
                return False
        return True

# Transformation Chain
class TransformationChain:
    def __init__(self):
        self.logger = logging.getLogger(__name__)

    def apply_transformations(self, data: pd.DataFrame, transformations: List[Callable]) -> pd.DataFrame:
        """
        Applies a chain of transformations to the data.

        Args:
        - data (pd.DataFrame): The data to transform.
        - transformations (List[Callable]): The transformations to apply.

        Returns:
        - pd.DataFrame: The transformed data.
        """
        for transformation in transformations:
            data = transformation(data)
        return data

# Plugin System
class PluginSystem:
    def __init__(self):
        self.transformations = {}

    def register_transformation(self, name: str, transformation: Callable):
        """
        Registers a custom transformation.

        Args:
        - name (str): The name of the transformation.
        - transformation (Callable): The transformation function.
        """
        self.transformations[name] = transformation

    def get_transformation(self, name: str) -> Callable:
        """
        Gets a registered transformation.

        Args:
        - name (str): The name of the transformation.

        Returns:
        - Callable: The transformation function.
        """
        return self.transformations.get(name)
