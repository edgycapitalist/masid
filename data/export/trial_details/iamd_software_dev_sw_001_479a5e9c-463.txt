import hashlib
import threading
from .hash_generator import generate_hash
from .collision_detector import detect_collision
from .url_validator import validate_url

class URLShortener:
    def __init__(self):
        self.url_map = {}
        self.lock = threading.Lock()

    def shorten_url(self, long_url):
        """
        Shorten a URL and return the shortened version.

        Args:
            long_url (str): The long URL to shorten.

        Returns:
            str: The shortened URL.
        """
        if not validate_url(long_url):
            raise ValueError("Invalid URL format")

        with self.lock:
            hash_value = generate_hash(long_url)
            if detect_collision(self.url_map, hash_value):
                # Handle collision by appending a counter to the hash
                counter = 0
                while detect_collision(self.url_map, f"{hash_value}{counter}"):
                    counter += 1
                hash_value = f"{hash_value}{counter}"

            self.url_map[hash_value] = long_url
            return f"http://short.url/{hash_value}"

    def get_original_url(self, short_url):
        """
        Get the original URL from a shortened URL.

        Args:
            short_url (str): The shortened URL.

        Returns:
            str: The original URL.
        """
        hash_value = short_url.split("/")[-1]
        with self.lock:
            return self.url_map.get(hash_value)

# url_shortener.py
class URLShortener:
    def __init__(self):
        self.url_map = {}
        self.counter = 0

    def shorten_url(self, long_url):
        if not self.is_valid_url(long_url):
            raise ValueError("Invalid URL format")
        short_url = f"url{self.counter}"
        self.url_map[short_url] = long_url
        self.counter += 1
        return short_url

    def get_original_url(self, short_url):
        return self.url_map.get(short_url)

    def is_valid_url(self, url):
        return url.startswith("http://") or url.startswith("https://")

# Example usage:
shortener = URLShortener()
long_url = "https://www.example.com"
