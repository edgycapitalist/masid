**Architecture Document: In-Memory Task Queue**
=====================================================

**Overview**
------------

The in-memory task queue is a critical component of the overall project, responsible for managing tasks with varying priorities, tracking their status, and ensuring reliable execution. This document outlines the high-level design and implementation details of the task queue, ensuring seamless integration with other project components.

**Module Decomposition**
----------------------

The task queue is composed of the following modules:

1.  **Task**: Represents a single task with properties such as `id`, `priority`, `status`, and `max_retries`.
2.  **TaskQueue**: Manages the task queue, providing methods for adding tasks, consuming tasks, and tracking task status.
3.  **WorkerThread**: Responsible for consuming tasks from the task queue and executing them.
4.  **TaskStatusTracker**: Tracks the status of tasks, including pending, running, completed, and failed.

**Interfaces**
-------------

The following interfaces are defined to ensure loose coupling between modules:

1.  **TaskQueueInterface**: Defines methods for adding tasks, consuming tasks, and tracking task status.
2.  **WorkerThreadInterface**: Defines methods for executing tasks and reporting task status.

**Data Flow**
-------------

The data flow between modules is as follows:

1.  Tasks are added to the `TaskQueue` using the `add_task` method.
2.  `WorkerThread` instances consume tasks from the `TaskQueue` using the `consume_task` method.
3.  The `WorkerThread` executes the task and reports the task status to the `TaskStatusTracker`.
4.  The `TaskStatusTracker` updates the task status and notifies the `TaskQueue` of any changes.

**Technology Choices**
---------------------

The implementation will utilize the following technologies:

1.  **Python**: As the primary programming language.
2.  **threading**: For creating and managing worker threads.
3.  **queue**: For implementing the task queue.
4.  **logging**: For logging task status and errors.

```python
import threading
import queue
import logging
from enum import Enum

# Define task status enum
class TaskStatus(Enum):
    PENDING = 1
    RUNNING = 2
    COMPLETED = 3
    FAILED = 4

# Define task class
class Task:
    def __init__(self, id, priority, max_retries):
        self.id = id
        self.priority = priority
        self.max_retries = max_retries
        self.status = TaskStatus.PENDING

# Define task queue class
class TaskQueue:
    def __init__(self):
        self.queue = queue.PriorityQueue()
        self.task_status_tracker = TaskStatusTracker()

    def add_task(self, task):
        self.queue.put((task.priority, task))
        self.task_status_tracker.add_task(task)

    def consume_task(self):
        if self.queue.empty():
            return None
        _, task = self.queue.get()
        self.task_status_tracker.update_task_status(task, TaskStatus.RUNNING)
        return task

# Define worker thread class
class WorkerThread(threading.Thread):
    def __init__(self, task_queue):
        super().__init__()
        self.task_queue = task_queue

    def run(self):
        while True:
            task = self.task_queue.consume_task()
            if task is None:
                break
            try:
                # Execute task
                logging.info(f"Executing task {task.id}")
                # Simulate task execution
                import time
                time.sleep(1)
                self.task_queue.task_status_tracker.update_task_status(task, TaskStatus.COMPLETED)
            except Exception as e:
                self.task_queue.task_status_tracker.update_task_status(task, TaskStatus.FAILED)
                logging.error(f"Error executing task {task.id}: {str(e)}")

# Define task status tracker class
class TaskStatusTracker:
    def __init__(self):
        self.tasks = {}

    def add_task(self, task):
        self.tasks[task.id] = task

    def update_task_status(self, task, status):
        self.tasks[task.id].status = status

# Define main function
def main():
    task_queue = TaskQueue()
    worker_threads = []

    # Add tasks to queue
    task1 = Task(1, 1, 3)
    task2 = Task(2, 2, 3)
    task_queue.add_task(task1)
    task_queue.add_task(task2)

    # Create and start worker threads
    for _ in range(2):
        worker_thread = WorkerThread(task_queue)
        worker_thread.start()
        worker_threads.append(worker_thread)

    # Wait for worker threads to finish
    for worker_thread in worker_threads:
        worker_thread.join()

if __name__ == "__main__":
    logging.basicConfig(level=logging.INFO)
    main()
```

import threading
import queue
import logging
from enum import Enum

# Define task status enum
class TaskStatus(Enum):
    PENDING = 1
    RUNNING = 2
    COMPLETED = 3
    FAILED = 4

# Define task class
class Task:
    def __init__(self, id, priority, max_retries, func, args=None, kwargs=None):
        self.id = id
        self.priority = priority
        self.max_retries = max_retries
        self.status = TaskStatus.PENDING
        self.func = func
        self.args = args if args else []
        self.kwargs = kwargs if kwargs else {}
        self.retries = 0

    def execute(self):
        try:
            self.func(*self.args, **self.kwargs)
            self.status = TaskStatus.COMPLETED
        except Exception as e:
            self.status = TaskStatus.FAILED
            self.retries += 1
            logging.error(f"Error executing task {self.id}: {str(e)}")
            if self.retries < self.max_retries:
                logging.info(f"Retrying task {self.id}...")
                # Simulate task retry
                import time
                time.sleep(1)
                self.execute()

# Define task queue class
class TaskQueue:
    def __init__(self, max_size=100):
        self.queue = queue.PriorityQueue(max_size)
        self.task_status_tracker = TaskStatusTracker()
        self.shutdown = threading.Event()

    def add_task(self, task):
        if not self.queue.full():
            self.queue.put((task.priority, task))
            self.task_status_tracker.add_task(task)
        else:
            logging.warning("Task queue is full. Task not added.")

    def consume_task(self):
        if self.shutdown.is_set() or self.queue.empty():
            return None
        _, task = self.queue.get()
        self.task_status_tracker.update_task_status(task, TaskStatus.RUNNING)
        return task

    def shutdown_queue(self):
        self.shutdown.set()

# Define worker thread class
class WorkerThread(threading.Thread):
    def __init__(self, task_queue):
        super().__init__()
        self.task_queue = task_queue

    def run(self):
        while True:
            task = self.task_queue.consume_task()
            if task is None:
                break
            task.execute()
            if task.status == TaskStatus.COMPLETED:
                self.task_queue.task_status_tracker.update_task_status(task, TaskStatus.COMPLETED)
            elif task.status == TaskStatus.FAILED and task.retries >= task.max_retries:
                self.task_queue.task_status_tracker.update_task_status(task, TaskStatus.FAILED)
            elif task.status == TaskStatus.FAILED and task.retries < task.max_retries:
                self.task_queue.add_task(task)

class TaskStatusTracker:
    def __init__(self):
        self.tasks = {}

    def add_task(self, task):
        self.tasks[task.id] = task

    def update_task_status(self, task, status):
        self.tasks[task.id].status = status

    task_queue = TaskQueue(max_size=100)
    worker_threads = []

    # Add tasks to queue

        logging.info("Example task executed")

    task2 = Task(2, 2, 3, example_task)

2.  **TaskQueue**: Manages the task queue, providing methods for adding tasks, consuming tasks, and tracking task status.
3.  **WorkerThread**: Responsible for consuming tasks from the task queue and executing them.
4.  **TaskStatusTracker**: Tracks the status of tasks, including pending, running, completed, and failed.

**Interfaces**
-------------

The following interfaces are defined to ensure loose coupling between modules:

1.  **TaskQueueInterface**: Defines methods for adding tasks, consuming tasks, and tracking task status.
2.  **WorkerThreadInterface**: Defines methods for executing tasks and reporting task status.

**Data Flow**
-------------

The data flow between modules is as follows:

1.  Tasks are added to the `TaskQueue` using the `add_task` method.
2.  `WorkerThread` instances consume tasks from the `TaskQueue` using the `consume_task` method.
3.  The `WorkerThread` executes the task and reports the task status to the `TaskStatusTracker`.
4.  The `TaskStatusTracker` updates the task status and notifies the `TaskQueue` of any changes.

```python
import threading
import queue
import logging
from enum import Enum
import signal
import sys

# Define task status enum
class TaskStatus(Enum):
    PENDING = 1
    RUNNING = 2
    COMPLETED = 3
    FAILED = 4

# Define task class
class Task:
    def __init__(self, id, priority, max_retries, func, args=None, kwargs=None):
        self.id = id
        self.priority = priority
        self.max_retries = max_retries
        self.status = TaskStatus.PENDING
        self.func = func
        self.args = args if args else []
        self.kwargs = kwargs if kwargs else {}
        self.retries = 0

    def execute(self):
        try:
            self.func(*self.args, **self.kwargs)
            self.status = TaskStatus.COMPLETED
        except Exception as e:
            self.status = TaskStatus.FAILED
            self.retries += 1
            logging.error(f"Error executing task {self.id}: {str(e)}")
            if self.retries < self.max_retries:
                logging.info(f"Retrying task {self.id}...")
                # Simulate task retry
                import time
                time.sleep(1)
                self.execute()

# Define task queue class
class TaskQueue:
    def __init__(self, max_size=100, num_worker_threads=2):
        self.queue = queue.PriorityQueue(max_size)
        self.task_status_tracker = TaskStatusTracker()
        self.shutdown = threading.Event()
        self.num_worker_threads = num_worker_threads

    def add_task(self, task):
        if not self.queue.full():
            self.queue.put((task.priority, task))
            self.task_status_tracker.add_task(task)
        else:
            logging.warning("Task queue is full. Task not added.")

    def consume_task(self):
        if self.shutdown.is_set() or self.queue.empty():
            return None
        _, task = self.queue.get()
        self.task_status_tracker.update_task_status(task, TaskStatus.RUNNING)
        return task

    def shutdown_queue(self):
        self.shutdown.set()

    def start_worker_threads(self):
        worker_threads = []
        for _ in range(self.num_worker_threads):
            worker_thread = WorkerThread(self)
            worker_thread.start()
            worker_threads.append(worker_thread)
        return worker_threads

# Define worker thread class
class WorkerThread(threading.Thread):
    def __init__(self, task_queue):
        super().__init__()
        self.task_queue = task_queue

    def run(self):
        while True:
            task = self.task_queue.consume_task()
            if task is None:
                break
            try:
                task.execute()
                self.task_queue.task_status_tracker.update_task_status(task, TaskStatus.COMPLETED)
            except Exception as e:
                self.task_queue.task_status_tracker.update_task_status(task, TaskStatus.FAILED)
                logging.error(f"Error executing task {task.id}: {str(e)}")

# Define task status tracker class
class TaskStatusTracker:
    def __init__(self):
        self.tasks = {}

    def add_task(self, task):
        self.tasks[task.id] = task

    def update_task_status(self, task, status):
        self.tasks[task.id].status = status

# Define main function
def main():
    task_queue = TaskQueue(max_size=100, num_worker_threads=2)
    worker_threads = task_queue.start_worker_threads()

    # Add tasks to queue
    def example_task():
        logging.info("Example task executed")

    task1 = Task(1, 1, 3, example_task)
    task2 = Task(2, 2, 3, example_task)
    task_queue.add_task(task1)
    task_queue.add_task(task2)

    # Wait for worker threads to finish
    import time
    time.sleep(5)

    # Shutdown task queue
    def signal_handler(sig, frame):
        task_queue.shutdown_queue()
        for worker_thread in worker_threads:
            worker_thread.join()
        sys.exit(0)

    signal.signal(signal.SIGINT, signal_handler)
    signal.signal(signal.SIGTERM, signal_handler)

if __name__ == "__main__":
    logging.basicConfig(level=logging.INFO)
    main()
```
