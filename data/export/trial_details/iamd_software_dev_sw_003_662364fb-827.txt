# data_reader.py
import pandas as pd
from typing import Dict

class DataReader:
    def __init__(self, schema: Dict):
        self.schema = schema

    def read_csv(self, file_path: str):
        try:
            data = pd.read_csv(file_path)
            self.validate_schema(data)
            return data
        except Exception as e:
            raise ValueError(f"Error reading CSV file: {e}")

    def validate_schema(self, data: pd.DataFrame):
        for column, dtype in self.schema.items():
            if column not in data.columns:
                raise ValueError(f"Column {column} not found in data")
            if data[column].dtype != dtype:
                raise ValueError(f"Column {column} has incorrect data type")

# transformation_chain.py
from typing import List
from abc import ABC, abstractmethod

class Transformation(ABC):
    @abstractmethod
    def apply(self, data: pd.DataFrame):
        pass

class FilterTransformation(Transformation):
    def __init__(self, column: str, value: str):
        self.column = column
        self.value = value

    def apply(self, data: pd.DataFrame):
        return data[data[self.column] == self.value]

class MapTransformation(Transformation):
    def __init__(self, column: str, mapping: Dict):
        self.column = column
        self.mapping = mapping

    def apply(self, data: pd.DataFrame):
        data[self.column] = data[self.column].map(self.mapping)
        return data

class AggregationTransformation(Transformation):
    def __init__(self, column: str, aggregation: str):
        self.column = column
        self.aggregation = aggregation

    def apply(self, data: pd.DataFrame):
        if self.aggregation == "sum":
            return data.groupby(self.column).sum()
        elif self.aggregation == "mean":
            return data.groupby(self.column).mean()
        else:
            raise ValueError(f"Invalid aggregation: {self.aggregation}")

class TransformationChain:
    def __init__(self, transformations: List[Transformation]):
        self.transformations = transformations

    def apply(self, data: pd.DataFrame):
        for transformation in self.transformations:
            data = transformation.apply(data)
        return data

# plugin_system.py
from typing import Dict

class Plugin:
    def __init__(self, name: str):
        self.name = name

    def apply(self, data: pd.DataFrame):
        raise NotImplementedError

class PluginSystem:
    def __init__(self):
        self.plugins = {}

    def register_plugin(self, plugin: Plugin):
        self.plugins[plugin.name] = plugin

    def apply_plugin(self, plugin_name: str, data: pd.DataFrame):
        if plugin_name in self.plugins:
            return self.plugins[plugin_name].apply(data)
        else:
            raise ValueError(f"Plugin {plugin_name} not found")

import csv

class DataWriter:
    def __init__(self, output_format: str):
        self.output_format = output_format

    def write(self, data: pd.DataFrame, file_path: str):
        if self.output_format == "csv":
            data.to_csv(file_path, index=False)
        elif self.output_format == "json":

        else:

# main.py
from data_reader import DataReader
from transformation_chain import TransformationChain
from plugin_system import PluginSystem
from data_writer import DataWriter
from error_reporter import ErrorReporter

def main():
    schema = {"name": str, "age": int}
    data_reader = DataReader(schema)
    data = data_reader.read_csv("input.csv")

    transformations = [
        FilterTransformation("age", 30),
        MapTransformation("name", {"John": "Jane"}),
        AggregationTransformation("name", "sum")
    ]
    transformation_chain = TransformationChain(transformations)
    transformed_data = transformation_chain.apply(data)

    plugin_system = PluginSystem()
    plugin_system.register_plugin(Plugin("custom_plugin"))
    transformed_data = plugin_system.apply_plugin("custom_plugin", transformed_data)

    data_writer = DataWriter("csv")
    data_writer.write(transformed_data, "output.csv")

    error_reporter = ErrorReporter()
    error_reporter.report_error("Error reading CSV file")
    print(error_reporter.get_errors())

if __name__ == "__main__":
    main()

    @abstractmethod
    def apply(self, data):
        pass

        self.column = column

from abc import ABC, abstractmethod
import pandas as pd

class Transformation(ABC):
    @abstractmethod
    def apply(self, data: pd.DataFrame) -> pd.DataFrame:
        """
        Apply the transformation.

        Args:
        - data (pd.DataFrame): The data to transform.

        Returns:
        - pd.DataFrame: The transformed data.
        """
        pass

class FilterTransformation(Transformation):
    def __init__(self, column: str, value: str):
        """
        Initialize the filter transformation.

        Args:
        - column (str): The column to filter.
        - value (str): The value to filter.
        """
        self.column = column
        self.value = value

    def apply(self, data: pd.DataFrame) -> pd.DataFrame:
        return data[data[self.column] == self.value]

class MapTransformation(Transformation):
    def __init__(self, column: str, mapping: dict):
        """
        Initialize the map transformation.

        Args:
        - column (str): The column to map.
        - mapping (dict): The mapping to apply.
        """
        self.column = column
        self.mapping = mapping

    def apply(self, data: pd.DataFrame) -> pd.DataFrame:
        data[self.column] = data[self.column].map(self.mapping)
        return data

class TransformationChain:
    def __init__(self, transformations: list):
        """
        Initialize the transformation chain.

        Args:
        - transformations (list): The list of transformations to apply.
        """
        self.transformations = transformations

    def apply(self, data: pd.DataFrame) -> pd.DataFrame:
        """
        Apply the transformation chain.

        Args:
        - data (pd.DataFrame): The data to transform.

        Returns:
        - pd.DataFrame: The transformed data.
        """

3. **Plugin System**: Allows for custom transformations to be added to the pipeline.
4. **Data Writer**: Outputs the transformed data to CSV or JSON.
5. **Error Reporter**: Provides clear error reporting for malformed data.

import pandas as pd
import json

class DataReader:
    def __init__(self, schema):
        self.schema = schema

    def read_csv(self, file_path):
        try:
            data = pd.read_csv(file_path)
            self.validate_schema(data)
            return data
        except Exception as e:
            raise ValueError(f"Error reading CSV file: {e}")

    def validate_schema(self, data):
        for column, dtype in self.schema.items():
            if column not in data.columns:
                raise ValueError(f"Column {column} not found in data")
            if data[column].dtype != dtype:
                raise ValueError(f"Column {column} has incorrect data type")

class Transformation:
    def apply(self, data):
        raise NotImplementedError

class FilterTransformation(Transformation):
    def __init__(self, column, value):
        self.column = column
        self.value = value

    def apply(self, data):
        return data[data[self.column] == self.value]

class MapTransformation(Transformation):
    def __init__(self, column, mapping):
        self.column = column
        self.mapping = mapping

    def apply(self, data):
        data[self.column] = data[self.column].map(self.mapping)
        return data

class TransformationChain:
    def __init__(self, transformations):
        self.transformations = transformations

    def apply(self, data):
        for transformation in self.transformations:
            data = transformation.apply(data)
        return data

class Plugin:
    def apply(self, data):
        raise NotImplementedError

class DataWriter:
    def __init__(self, output_format):
        self.output_format = output_format

    def write(self, data, file_path):
        if self.output_format == "csv":
            data.to_csv(file_path, index=False)
        elif self.output_format == "json":
            data.to_json(file_path, orient="records")
        else:
            raise ValueError(f"Invalid output format: {self.output_format}")

class ErrorReporter:
    def __init__(self):
        self.errors = []

    def report_error(self, error):
        self.errors.append(error)

    def report_errors(self):
        for error in self.errors:
            print(error)

def main():
    schema = {"name": str, "age": int}
    transformations = [
        FilterTransformation("age", 30),
        MapTransformation("name", {"John": "Jane"})
    ]
    data_reader = DataReader(schema)
