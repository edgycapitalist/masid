## Introduction

6. Once a unique short URL is generated, the **URLShortener** class stores the mapping between the short and long URLs in the **URLStore**.
7. The **URLShortener** class returns the shortened URL to the user.
8. The user can call the **get_original_url** method with a short URL as input to retrieve the original URL.

import hashlib
import threading
import sqlite3
from urllib.parse import urlparse

class URLValidator:
    def is_valid_url(self, url):
        try:
            result = urlparse(url)
            return all([result.scheme, result.netloc])
        except ValueError:
            return False

class URLStore:
    def __init__(self, db_name):
        self.db_name = db_name
        self.lock = threading.Lock()
        self.conn = sqlite3.connect(db_name)
        self.cursor = self.conn.cursor()
        self.cursor.execute('''
            CREATE TABLE IF NOT EXISTS urls
            (short_url TEXT PRIMARY KEY, long_url TEXT)
        ''')
        self.conn.commit()

    def store_url(self, short_url, long_url):
        with self.lock:
            self.cursor.execute('INSERT INTO urls VALUES (?, ?)', (short_url, long_url))
            self.conn.commit()

    def get_original_url(self, short_url):
        with self.lock:
            self.cursor.execute('SELECT long_url FROM urls WHERE short_url = ?', (short_url,))
            result = self.cursor.fetchone()
            if result:
                return result[0]
            else:
                return None

class URLShortener:
    def __init__(self, db_name):
        self.url_validator = URLValidator()
        self.url_store = URLStore(db_name)

    def shorten_url(self, long_url):
        if not self.url_validator.is_valid_url(long_url):
            raise ValueError('Invalid URL format')
        short_url = hashlib.sha256(long_url.encode()).hexdigest()[:6]
        while self.url_store.get_original_url(short_url):
            short_url = hashlib.sha256((short_url + '1').encode()).hexdigest()[:6]
        self.url_store.store_url(short_url, long_url)
        return short_url

    def get_original_url(self, short_url):
        return self.url_store.get_original_url(short_url)

# Example usage
url_shortener = URLShortener('urls.db')
long_url = 'https://www.example.com/very/long/url'
short_url = url_shortener.shorten_url(long_url)
print(f'Short URL: {short_url}')
print(f'Original URL: {url_shortener.get_original_url(short_url)}')

import hashlib
import threading
import sqlite3
from urllib.parse import urlparse
from typing import Optional

class URLValidator:
    def is_valid_url(self, url: str) -> bool:
        try:
            result = urlparse(url)
            return all([result.scheme, result.netloc])
        except ValueError:
            return False

class URLStore:
    def __init__(self, db_name: str):
        self.db_name = db_name
        self.lock = threading.Lock()
        self.conn = sqlite3.connect(db_name)
        self.cursor = self.conn.cursor()
        self.cursor.execute('''
            CREATE TABLE IF NOT EXISTS urls
            (short_url TEXT PRIMARY KEY, long_url TEXT)
        ''')
        self.conn.commit()

    def store_url(self, short_url: str, long_url: str) -> None:
        with self.lock:
            self.cursor.execute('INSERT INTO urls VALUES (?, ?)', (short_url, long_url))
            self.conn.commit()

    def get_original_url(self, short_url: str) -> Optional[str]:
        with self.lock:
            self.cursor.execute('SELECT long_url FROM urls WHERE short_url = ?', (short_url,))
            result = self.cursor.fetchone()
            if result:
                return result[0]
            else:
                return None

    def close_connection(self) -> None:
        self.conn.close()

class URLShortener:
    def __init__(self, db_name: str):
        self.url_validator = URLValidator()
        self.url_store = URLStore(db_name)

    def shorten_url(self, long_url: str) -> str:
        if not self.url_validator.is_valid_url(long_url):
            raise ValueError('Invalid URL format')
        short_url = hashlib.sha256(long_url.encode()).hexdigest()[:6]
        while self.url_store.get_original_url(short_url):
            short_url = hashlib.sha256((short_url + '1').encode()).hexdigest()[:6]
        self.url_store.store_url(short_url, long_url)
        return short_url

    def get_original_url(self, short_url: str) -> Optional[str]:
        return self.url_store.get_original_url(short_url)

    def close(self) -> None:
        self.url_store.close_connection()

def test_url_shortener_thread_safety():
    """Test thread safety of URLShortener."""

    shortener = URLShortener('test.db')
    long_url = 'https://www.example.com'
    def shorten_url():
        shortener.shorten_url(long_url)
    threads = [threading.Thread(target=shorten_url) for _ in range(10)]
    for thread in threads:
        thread.start()
    for thread in threads:
        thread.join()
    shortener.close()
    os.remove('test.db')
