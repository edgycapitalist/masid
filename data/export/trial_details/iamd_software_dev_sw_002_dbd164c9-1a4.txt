import threading
import time

class TokenBucket:
    def __init__(self, rate, burst_size):
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()

    def allow_request(self):
        current_time = time.time()
        elapsed_time = current_time - self.last_update
        self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)
        self.last_update = current_time
        if self.tokens >= 1:
            self.tokens -= 1
            return True
        return False

class RateLimiter:
    def __init__(self, name, rate, burst_size):
        self.name = name
        self.token_bucket = TokenBucket(rate, burst_size)
        self.lock = threading.Lock()

    def allow_request(self):
        with self.lock:
            return self.token_bucket.allow_request()

    def consume_token(self):
        with self.lock:
            self.token_bucket.tokens -= 1

class RateLimiterManager:
    def __init__(self):
        self.rate_limiters = {}
        self.lock = threading.Lock()

    def create_rate_limiter(self, name, rate, burst_size):
        with self.lock:
            if name not in self.rate_limiters:
                self.rate_limiters[name] = RateLimiter(name, rate, burst_size)

    def get_rate_limiter(self, name):
        with self.lock:
            return self.rate_limiters.get(name)

    def delete_rate_limiter(self, name):
        with self.lock:
            if name in self.rate_limiters:
                del self.rate_limiters[name]

def rate_limited(rate_limiter_name):
    def decorator(func):
        def wrapper(*args, **kwargs):
            rate_limiter_manager = RateLimiterManager()
            rate_limiter = rate_limiter_manager.get_rate_limiter(rate_limiter_name)
            if rate_limiter and not rate_limiter.allow_request():
                time.sleep(1)
                return wrapper(*args, **kwargs)
            return func(*args, **kwargs)
        return wrapper
    return decorator

import time
import threading
from typing import Callable

class TokenBucket:
    def __init__(self, rate: float, burst_size: int):
        """
        Initialize a token bucket with the given rate and burst size.

        :param rate: The rate at which tokens are added to the bucket.
        :param burst_size: The maximum number of tokens in the bucket.
        """
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()
        self.lock = threading.Lock()

    def allow_request(self) -> bool:
        """
        Check if a request is allowed based on the current token count.

        :return: True if the request is allowed, False otherwise.
        """
        with self.lock:
            current_time = time.time()
            elapsed_time = current_time - self.last_update
            self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)
            self.last_update = current_time
            if self.tokens >= 1:
                self.tokens -= 1
                return True
            return False

class RateLimiter:
    def __init__(self, rate: float, burst_size: int):
        """
        Initialize a rate limiter with the given rate and burst size.

        :param rate: The rate at which tokens are added to the bucket.
        :param burst_size: The maximum number of tokens in the bucket.
        """
        self.token_bucket = TokenBucket(rate, burst_size)

    def allow_request(self) -> bool:
        """
        Check if a request is allowed based on the current token count.

        :return: True if the request is allowed, False otherwise.
        """
        return self.token_bucket.allow_request()

class RateLimiterManager:
    def __init__(self):
        """
        Initialize a rate limiter manager.
        """
        self.rate_limiters = {}
        self.lock = threading.Lock()

    def create_rate_limiter(self, name: str, rate: float, burst_size: int):
        """
        Create a new rate limiter with the given name, rate, and burst size.

        :param name: The name of the rate limiter.
        :param rate: The rate at which tokens are added to the bucket.
        :param burst_size: The maximum number of tokens in the bucket.
        """
        with self.lock:
            if name not in self.rate_limiters:
                self.rate_limiters[name] = RateLimiter(rate, burst_size)

    def get_rate_limiter(self, name: str) -> RateLimiter:
        """
        Get a rate limiter by name.

        :param name: The name of the rate limiter.
        :return: The rate limiter, or None if not found.
        """
        with self.lock:
            return self.rate_limiters.get(name)

def rate_limited(rate_limiter_name: str, rate_limiter_manager: RateLimiterManager):
    """
    Decorator to apply rate limiting to a function.

    :param rate_limiter_name: The name of the rate limiter to use.
    :param rate_limiter_manager: The rate limiter manager.
    :return: A decorator that applies rate limiting to the function.
    """
    def decorator(func: Callable):
        def wrapper(*args, **kwargs):
            try:
                rate_limiter = rate_limiter_manager.get_rate_limiter(rate_limiter_name)
                if rate_limiter and not rate_limiter.allow_request():
                    time.sleep(1)
                    return wrapper(*args, **kwargs)
                return func(*args, **kwargs)
            except Exception as e:
                print(f"Error: {e}")
                return None
        return wrapper
    return decorator
