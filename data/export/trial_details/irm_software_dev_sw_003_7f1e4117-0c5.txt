import pandas as pd
from typing import Dict

class DataIngestion:
    def __init__(self, schema: Dict):
        self.schema = schema

    def read_csv(self, file_path: str) -> pd.DataFrame:
        try:
            df = pd.read_csv(file_path)
            self.validate_schema(df)
            return df
        except Exception as e:
            raise ValueError(f"Error reading CSV file: {e}")

    def validate_schema(self, df: pd.DataFrame) -> None:
        for column, dtype in self.schema.items():
            if column not in df.columns:
                raise ValueError(f"Column {column} not found in CSV file")
            if df[column].dtype != dtype:
                raise ValueError(f"Column {column} has incorrect data type")
```

from typing import List
from abc import ABC, abstractmethod

class Transformation(ABC):
    @abstractmethod
    def apply(self, df: pd.DataFrame) -> pd.DataFrame:
        pass

class FilterTransformation(Transformation):
    def __init__(self, column: str, value: str):
        self.column = column
        self.value = value

    def apply(self, df: pd.DataFrame) -> pd.DataFrame:
        return df[df[self.column] == self.value]

class MapTransformation(Transformation):
    def __init__(self, column: str, func: callable):
        self.column = column
        self.func = func

    def apply(self, df: pd.DataFrame) -> pd.DataFrame:
        df[self.column] = df[self.column].apply(self.func)
        return df

class AggregationTransformation(Transformation):
    def __init__(self, column: str, agg_func: callable):
        self.column = column
        self.agg_func = agg_func

    def apply(self, df: pd.DataFrame) -> pd.DataFrame:
        return df.groupby(self.column).agg(self.agg_func)

class TransformationChain:
    def __init__(self, transformations: List[Transformation]):
        self.transformations = transformations

    def apply(self, df: pd.DataFrame) -> pd.DataFrame:
        for transformation in self.transformations:
            df = transformation.apply(df)
        return df

import pandas as pd
import json
import csv
from typing import Dict, List
from abc import ABC, abstractmethod

# Data Ingestion
class DataIngestion:
    def __init__(self, schema: Dict):
        self.schema = schema

    def read_csv(self, file_path: str) -> pd.DataFrame:
        try:
            df = pd.read_csv(file_path)
            self.validate_schema(df)
            return df
        except pd.errors.EmptyDataError as e:
            raise ValueError(f"CSV file is empty: {e}")
        except pd.errors.ParserError as e:
            raise ValueError(f"Error parsing CSV file: {e}")

    def validate_schema(self, df: pd.DataFrame) -> None:
        for column, dtype in self.schema.items():
            if column not in df.columns:
                raise ValueError(f"Column {column} not found in CSV file")
            if df[column].dtype != dtype:
                raise ValueError(f"Column {column} has incorrect data type")

# Transformation Chain
class Transformation(ABC):
    @abstractmethod
    def apply(self, df: pd.DataFrame) -> pd.DataFrame:
        pass

class FilterTransformation(Transformation):
    def __init__(self, column: str, value: str):
        self.column = column
        self.value = value

    def apply(self, df: pd.DataFrame) -> pd.DataFrame:
        return df[df[self.column] == self.value]

class MapTransformation(Transformation):
    def __init__(self, column: str, func: callable):
        self.column = column
        self.func = func

    def apply(self, df: pd.DataFrame) -> pd.DataFrame:
        df[self.column] = df[self.column].apply(self.func)
        return df

class AggregationTransformation(Transformation):
    def __init__(self, column: str, agg_func: callable):
        self.column = column
        self.agg_func = agg_func

    def apply(self, df: pd.DataFrame) -> pd.DataFrame:
        return df.groupby(self.column).agg(self.agg_func)

class TransformationChain:
    def __init__(self, transformations: List[Transformation]):
        self.transformations = transformations

    def apply(self, df: pd.DataFrame) -> pd.DataFrame:
        for transformation in self.transformations:
            df = transformation.apply(df)
        return df

# Plugin System
class Plugin(ABC):
    @abstractmethod
    def apply(self, df: pd.DataFrame) -> pd.DataFrame:
        pass

class CustomPlugin(Plugin):
    def __init__(self, func: callable):
        self.func = func

    def apply(self, df: pd.DataFrame) -> pd.DataFrame:
        return self.func(df)

class PluginSystem:
    def __init__(self, plugins: Dict[str, Plugin]):
        self.plugins = plugins

    def apply(self, df: pd.DataFrame, plugin_name: str) -> pd.DataFrame:
        if plugin_name in self.plugins:
            return self.plugins[plugin_name].apply(df)
        else:
            raise ValueError(f"Plugin {plugin_name} not found")

# Output
class Output:
    def __init__(self, output_format: str):
        self.output_format = output_format

    def write(self, df: pd.DataFrame, file_path: str) -> None:
        if self.output_format == "csv":
            df.to_csv(file_path, index=False)
        elif self.output_format == "json":
            df.to_json(file_path, orient="records")
        else:
            raise ValueError(f"Invalid output format: {self.output_format}")

# Error Handling
class ErrorHandler:
    def __init__(self):
        pass

    def handle_error(self, exception: Exception) -> None:
        print(f"Error: {exception}")
        if isinstance(exception, ValueError):
            print("ValueError occurred")
        elif isinstance(exception, TypeError):
            print("TypeError occurred")
        else:
            print("Unknown error occurred")

# Pipeline
class Pipeline:
    def __init__(self, data_ingestion: DataIngestion, transformation_chain: TransformationChain, plugin_system: PluginSystem, output: Output):
        self.data_ingestion = data_ingestion
        self.transformation_chain = transformation_chain
        self.plugin_system = plugin_system
        self.output = output

    def run(self, file_path: str, plugin_name: str = None) -> None:
        try:
            df = self.data_ingestion.read_csv(file_path)
            df = self.transformation_chain.apply(df)
            if plugin_name:
                df = self.plugin_system.apply(df, plugin_name)
            self.output.write(df, "output.csv")
        except Exception as e:
            self.output.write(pd.DataFrame(), "output.csv")
            self.error_handler.handle_error(e)
