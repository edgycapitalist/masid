## Introduction
The in-memory task queue is designed to support adding tasks with priorities, allowing worker threads to consume tasks, providing task status tracking, supporting task retry with configurable max retries, and including graceful shutdown. This document outlines the high-level structure, module decomposition, interfaces, data flow, and technology choices for the task queue.

The in-memory task queue consists of the following modules:

1. **Task**: Represents a single task with a unique ID, priority, and status.

3. **Worker**: Consumes tasks from the task queue and executes them.
4. **RetryPolicy**: Defines the retry policy for failed tasks, including the maximum number of retries and the retry delay.

## Data Flow
The data flow for the in-memory task queue is as follows:

1. A task is added to the task queue using the `add_task` method.
2. The task is assigned a unique ID and priority.
3. The task is stored in the in-memory task queue.
4. A worker thread consumes the task from the task queue using the `get_task` method.
5. The worker executes the task using the `execute_task` method.
6. The task status is updated to "running".
7. If the task fails, the retry policy is applied, and the task is retried if the maximum number of retries has not been reached.
8. If the task succeeds or the maximum number of retries is reached, the task status is updated to "completed" or "failed", respectively.

import threading
import queue
import time
from enum import Enum

class TaskStatus(Enum):
    PENDING = 1
    RUNNING = 2
    COMPLETED = 3
    FAILED = 4

class Task:
    def __init__(self, id, priority, func, args=None, kwargs=None):
        self.id = id
        self.priority = priority
        self.func = func
        self.args = args
        self.kwargs = kwargs
        self.status = TaskStatus.PENDING

    def execute(self):
        try:
            self.func(*self.args, **self.kwargs)
            self.status = TaskStatus.COMPLETED
        except Exception as e:
            self.status = TaskStatus.FAILED
            raise e

class TaskQueue:
    def __init__(self):
        self.queue = queue.PriorityQueue()
        self.tasks = {}

    def add_task(self, task):
        self.queue.put((task.priority, task.id))
        self.tasks[task.id] = task

    def remove_task(self, task_id):
        if task_id in self.tasks:
            del self.tasks[task_id]

    def get_task(self):
        if not self.queue.empty():
            _, task_id = self.queue.get()
            task = self.tasks[task_id]
            return task
        return None

class Worker(threading.Thread):
    def __init__(self, task_queue):
        super().__init__()
        self.task_queue = task_queue

    def run(self):
        while True:
            task = self.task_queue.get_task()
            if task is None:
                break
            task.status = TaskStatus.RUNNING
            try:
                task.execute()
            except Exception as e:
                print(f"Task {task.id} failed: {e}")
            finally:
                self.task_queue.remove_task(task.id)

class RetryPolicy:
    def __init__(self, max_retries, retry_delay):
        self.max_retries = max_retries
        self.retry_delay = retry_delay

    def retry(self, task):
        if task.status == TaskStatus.FAILED and task.retries < self.max_retries:
            time.sleep(self.retry_delay)
            task.retries += 1
            return True
        return False

def main():
    task_queue = TaskQueue()
    worker = Worker(task_queue)
    worker.start()

    def example_task():
        print("Example task executed")

    task = Task(1, 1, example_task)
    task_queue.add_task(task)

    worker.join()

if __name__ == "__main__":
    main()

            self.status = TaskStatus.COMPLETED
        except Exception as e:

    def __init__(self):
        self.queue = queue.PriorityQueue()
        self.tasks = {}
        self.lock = threading.Lock()

    def add_task(self, task):
        with self.lock:
            self.queue.put((task.priority, task.id))
            self.tasks[task.id] = task

    def remove_task(self, task_id):
        with self.lock:
            if task_id in self.tasks:
                del self.tasks[task_id]

    def get_task(self):
        with self.lock:
            if not self.queue.empty():
                _, task_id = self.queue.get()
                task = self.tasks[task_id]
                return task
            return None

import threading
import queue
import time
from enum import Enum

class TaskStatus(Enum):
    PENDING = 1
    RUNNING = 2
    COMPLETED = 3
    FAILED = 4

class Task:
    def __init__(self, id, priority, func, args=None, kwargs=None):
        self.id = id
        self.priority = priority
        self.func = func
        self.args = args
        self.kwargs = kwargs
        self.status = TaskStatus.PENDING
        self.retries = 0

    def execute(self):
        try:
            self.func(*self.args, **self.kwargs)
            self.status = TaskStatus.COMPLETED
        except Exception as e:
            self.status = TaskStatus.FAILED
            raise e

class TaskQueue:
    def __init__(self):
        self.queue = queue.PriorityQueue()
        self.tasks = {}
        self.lock = threading.Lock()

    def add_task(self, task):
        with self.lock:
            self.queue.put((task.priority, task.id))
            self.tasks[task.id] = task

    def remove_task(self, task_id):
        with self.lock:
            if task_id in self.tasks:
                del self.tasks[task_id]

    def get_task(self):
        with self.lock:
            if not self.queue.empty():
                _, task_id = self.queue.get()
                task = self.tasks[task_id]
                return task
            return None

class Worker(threading.Thread):
    def __init__(self, task_queue, retry_policy):
        super().__init__()
        self.task_queue = task_queue
        self.retry_policy = retry_policy

    def run(self):
        while True:
            task = self.task_queue.get_task()
            if task is None:
                break
            task.status = TaskStatus.RUNNING
            try:
                task.execute()
            except Exception as e:
                print(f"Task {task.id} failed: {e}")
                if self.retry_policy.retry(task):
                    self.task_queue.add_task(task)
            finally:
                self.task_queue.remove_task(task.id)

class RetryPolicy:
    def __init__(self, max_retries, retry_delay):
        self.max_retries = max_retries
        self.retry_delay = retry_delay

    def retry(self, task):
        if task.status == TaskStatus.FAILED and task.retries < self.max_retries:
            time.sleep(self.retry_delay)
            task.retries += 1
            return True
        return False

    task_queue = TaskQueue()

    worker = Worker(task_queue, retry_policy)
    worker.start()

    def example_task():
