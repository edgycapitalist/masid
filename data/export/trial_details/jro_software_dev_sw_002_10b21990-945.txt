import threading
import time
from functools import wraps

class RateLimiter:
    def __init__(self, rate, burst_size):
        """
        Initializes the rate limiter with a rate and burst size.

        Args:
            rate (int): The rate at which tokens are added to the bucket per second.
            burst_size (int): The maximum number of tokens in the bucket.
        """
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()
        self.lock = threading.Lock()

    def allow(self):
        """
        Checks if a request is allowed based on the current token count.

        Returns:
            bool: True if the request is allowed, False otherwise.
        """
        with self.lock:
            current_time = time.time()
            elapsed_time = current_time - self.last_update
            self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)
            self.last_update = current_time
            return self.tokens > 0

    def consume(self):
        """
        Consumes a token if available.
        """
        with self.lock:
            if self.allow():
                self.tokens -= 1

class RateLimiterDecorator:
    def __init__(self, rate_limiter):
        """
        Initializes the decorator with a rate limiter.

        Args:
            rate_limiter (RateLimiter): The rate limiter to apply to the function.
        """
        self.rate_limiter = rate_limiter

    def __call__(self, func):
        """
        Applies the rate limiter to a function.

        Args:
            func (function): The function to decorate.

        Returns:
            function: The decorated function.
        """
        @wraps(func)
        def wrapper(*args, **kwargs):
            self.rate_limiter.consume()
            return func(*args, **kwargs)
        return wrapper

        Initializes the rate limiter manager.

import logging

# Set up logging

class RateLimiter:
    """
    A rate limiter class that implements the token bucket algorithm.
    """
    def __init__(self, rate, burst_size):
        """
        Initializes the rate limiter with a rate and burst size.

        Args:
            rate (int): The rate at which tokens are added to the bucket per second.
            burst_size (int): The maximum number of tokens in the bucket.
        """
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()
        self.lock = threading.Lock()

    def allow(self):
        """
        Checks if a request is allowed based on the current token count.

        Returns:
            bool: True if the request is allowed, False otherwise.
        """
        with self.lock:
            current_time = time.time()
            elapsed_time = current_time - self.last_update
            self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)
            self.last_update = current_time
            return self.tokens > 0

    def consume(self):
        """
        Consumes a token if available.
        """
        with self.lock:
            if self.allow():
                self.tokens -= 1
                logger.info(f"Consumed token. Remaining tokens: {self.tokens}")
            else:
                logger.warning("Rate limit exceeded")

        Args:

        Returns:
            bool: True if the request is allowed, False otherwise.
