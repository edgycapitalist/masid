### Overview

The in-memory task queue is designed to manage tasks with priorities, allow worker threads to consume tasks, track task status, and support task retry with configurable max retries. The system will provide a robust and scalable solution for task management.

### Components

1. **Task Queue**: The central component responsible for storing and managing tasks.
2. **Task**: Represents a single task with a unique ID, priority, and status.
3. **Worker Thread**: Consumes tasks from the task queue and executes them.
4. **Task Status Tracker**: Maintains the status of each task (pending, running, completed, failed).
5. **Retry Mechanism**: Handles task retries with configurable max retries.

### System Design

#### Task Queue

* Implemented as a priority queue using Python's `heapq` module.
* Tasks are stored as tuples containing the task ID, priority, and task data.

#### Task

* Represents a single task with attributes:
	+ `id`: Unique task ID.
	+ `priority`: Task priority (higher values indicate higher priority).
	+ `status`: Task status (pending, running, completed, failed).
	+ `data`: Task data (e.g., function to execute, arguments).
	+ `max_retries`: Maximum number of retries allowed.
	+ `retries`: Current number of retries.

#### Worker Thread

* Consumes tasks from the task queue and executes them.
* Updates task status accordingly (running, completed, failed).
* Handles task retries using the retry mechanism.

#### Task Status Tracker

* Maintains a dictionary mapping task IDs to their current status.
* Updates task status when a worker thread completes or fails a task.

#### Retry Mechanism

* Checks if a task has exceeded its maximum retries.
* If not, schedules the task for retry with a configurable delay.

### System Flow

1. **Task Submission**:
	* Client submits a task to the task queue with a priority and task data.
	* Task queue adds the task to the priority queue.
2. **Worker Thread Execution**:
	* Worker thread consumes the highest-priority task from the task queue.
	* Worker thread executes the task and updates its status.
	* If the task fails, the worker thread schedules a retry if allowed.
3. **Task Status Tracking**:
	* Task status tracker updates the task status in the dictionary.
4. **Retry Mechanism**:
	* If a task has exceeded its maximum retries, it is marked as failed.
	* Otherwise, the task is scheduled for retry with a configurable delay.

### Implementation Code

```python
import heapq
import threading
import time
from typing import Dict, Tuple

class Task:
    def __init__(self, id: int, priority: int, data: str, max_retries: int = 3):
        self.id = id
        self.priority = priority
        self.data = data
        self.max_retries = max_retries
        self.retries = 0
        self.status = "pending"

class TaskQueue:
    def __init__(self):
        self.queue = []
        self.status_tracker: Dict[int, str] = {}

    def add_task(self, task: Task):
        heapq.heappush(self.queue, (task.priority, task))
        self.status_tracker[task.id] = task.status

    def get_task(self) -> Task:
        if not self.queue:
            return None
        _, task = heapq.heappop(self.queue)
        return task

class WorkerThread(threading.Thread):
    def __init__(self, task_queue: TaskQueue):
        super().__init__()
        self.task_queue = task_queue

    def run(self):
        while True:
            task = self.task_queue.get_task()
            if task is None:
                break
            task.status = "running"
            self.task_queue.status_tracker[task.id] = task.status
            try:
                # Execute task
                print(f"Executing task {task.id} with data {task.data}")
                time.sleep(1)  # Simulate task execution
                task.status = "completed"
            except Exception as e:
                task.status = "failed"
                task.retries += 1
                if task.retries < task.max_retries:
                    # Schedule retry
                    self.task_queue.add_task(task)
            finally:
                self.task_queue.status_tracker[task.id] = task.status

def main():
    task_queue = TaskQueue()
    worker_thread = WorkerThread(task_queue)

    # Submit tasks
    task1 = Task(1, 1, "Task 1 data")
    task2 = Task(2, 2, "Task 2 data")
    task_queue.add_task(task1)
    task_queue.add_task(task2)

    # Start worker thread
    worker_thread.start()

    # Wait for worker thread to finish
    worker_thread.join()

if __name__ == "__main__":
    main()
```

import heapq
import threading
import time
from typing import Dict, Tuple

class Task:
    def __init__(self, id: int, priority: int, data: str, max_retries: int = 3):
        self.id = id
        self.priority = priority
        self.data = data
        self.max_retries = max_retries
        self.retries = 0
        self.status = "pending"

class TaskQueue:
    def __init__(self, max_tasks: int = 100):
        self.queue = []
        self.status_tracker: Dict[int, str] = {}
        self.max_tasks = max_tasks
        self.lock = threading.Lock()

    def add_task(self, task: Task):
        with self.lock:
            if len(self.queue) < self.max_tasks:
                heapq.heappush(self.queue, (task.priority, task))
                self.status_tracker[task.id] = task.status
                return True
            else:
                return False

    def get_task(self) -> Task:
        with self.lock:
            if not self.queue:
                return None
            _, task = heapq.heappop(self.queue)
            return task

class WorkerThread(threading.Thread):
    def __init__(self, task_queue: TaskQueue):
        super().__init__()
        self.task_queue = task_queue
        self.stop_event = threading.Event()

    def run(self):
        while not self.stop_event.is_set():
            task = self.task_queue.get_task()
            if task is None:
                time.sleep(0.1)  # Wait for new tasks
                continue
            task.status = "running"
            self.task_queue.status_tracker[task.id] = task.status
            try:
                # Execute task
                print(f"Executing task {task.id} with data {task.data}")
                time.sleep(1)  # Simulate task execution
                task.status = "completed"
            except Exception as e:
                task.status = "failed"
                task.retries += 1
                if task.retries < task.max_retries:
                    # Schedule retry with exponential backoff
                    delay = 2 ** task.retries
                    print(f"Task {task.id} failed, retrying in {delay} seconds")
                    time.sleep(delay)
                    self.task_queue.add_task(task)
            finally:
                self.task_queue.status_tracker[task.id] = task.status

    def stop(self):
        self.stop_event.set()

def main():
    task_queue = TaskQueue(max_tasks=10)
    worker_thread = WorkerThread(task_queue)

    # Submit tasks
    task1 = Task(1, 1, "Task 1 data")
    task2 = Task(2, 2, "Task 2 data")
    task_queue.add_task(task1)
    task_queue.add_task(task2)

    # Start worker thread
    worker_thread.start()

    # Wait for 5 seconds
    time.sleep(5)

    # Stop worker thread
    worker_thread.stop()
    worker_thread.join()

if __name__ == "__main__":
    main()

import heapq
import threading
import time
from typing import Dict, Tuple

class Task:
    def __init__(self, id: int, priority: int, data: str, max_retries: int = 3):
        self.id = id
        self.priority = priority
        self.data = data
        self.max_retries = max_retries
        self.retries = 0
        self.status = "pending"

class TaskQueue:
    def __init__(self, max_tasks: int = 100):
        self.queue = []
        self.status_tracker: Dict[int, str] = {}
        self.max_tasks = max_tasks
        self.lock = threading.Lock()

    def add_task(self, task: Task):
        with self.lock:
            if len(self.queue) < self.max_tasks:
                heapq.heappush(self.queue, (task.priority, task))
                self.status_tracker[task.id] = task.status
                return True
            else:
                return False

    def get_task(self) -> Task:
        with self.lock:
            if not self.queue:
                return None
            _, task = heapq.heappop(self.queue)
            return task

class WorkerThread(threading.Thread):
    def __init__(self, task_queue: TaskQueue):
        super().__init__()
        self.task_queue = task_queue
        self.stop_event = threading.Event()

    def run(self):
        while not self.stop_event.is_set():
            task = self.task_queue.get_task()
            if task is None:
                time.sleep(0.1)  # Wait for new tasks
                continue
            task.status = "running"
            self.task_queue.status_tracker[task.id] = task.status
            try:
                # Execute task
                print(f"Executing task {task.id} with data {task.data}")
                time.sleep(1)  # Simulate task execution
                task.status = "completed"
            except Exception as e:
                task.status = "failed"
                task.retries += 1
                if task.retries < task.max_retries:
                    # Schedule retry with exponential backoff
                    delay = 2 ** task.retries
                    print(f"Task {task.id} failed, retrying in {delay} seconds")
                    time.sleep(delay)
                    self.task_queue.add_task(task)
            finally:
                self.task_queue.status_tracker[task.id] = task.status

    def stop(self):
        self.stop_event.set()

def main():
    task_queue = TaskQueue(max_tasks=10)
    worker_thread = WorkerThread(task_queue)

    # Submit tasks
    task1 = Task(1, 1, "Task 1 data")
    task2 = Task(2, 2, "Task 2 data")
    task_queue.add_task(task1)
    task_queue.add_task(task2)

    # Start worker thread
    worker_thread.start()

    # Wait for 5 seconds
    time.sleep(5)

    # Stop worker thread
    worker_thread.stop()
    worker_thread.join()

	* If the task fails, the worker thread schedules a retry if allowed.

	* Otherwise, the task is scheduled for retry with a configurable delay.

### Implementation Code

```python
import heapq
import threading
import time
from typing import Dict, Tuple

class Task:
    def __init__(self, id: int, priority: int, data: str, max_retries: int = 3):
        self.id = id
        self.priority = priority
        self.data = data
        self.max_retries = max_retries
        self.retries = 0
        self.status = "pending"

class TaskQueue:
    def __init__(self, max_tasks: int = 100):
        self.queue = []
        self.status_tracker: Dict[int, str] = {}
        self.max_tasks = max_tasks
        self.lock = threading.Lock()

    def add_task(self, task: Task):
        with self.lock:
            if len(self.queue) < self.max_tasks:
                heapq.heappush(self.queue, (task.priority, task))
                self.status_tracker[task.id] = task.status
                return True
            else:
                return False

    def get_task(self) -> Task:
        with self.lock:
            if not self.queue:
                return None
            _, task = heapq.heappop(self.queue)
            return task

class WorkerThread(threading.Thread):
    def __init__(self, task_queue: TaskQueue):
        super().__init__()
        self.task_queue = task_queue
        self.stop_event = threading.Event()

    def run(self):
        while not self.stop_event.is_set():
            task = self.task_queue.get_task()
            if task is None:
                time.sleep(0.1)  # Wait for new tasks
                continue
            task.status = "running"
            self.task_queue.status_tracker[task.id] = task.status
            try:
                # Execute task
                print(f"Executing task {task.id} with data {task.data}")
                time.sleep(1)  # Simulate task execution
                task.status = "completed"
            except Exception as e:
                task.status = "failed"
                task.retries += 1
                if task.retries < task.max_retries:
                    # Schedule retry with exponential backoff
                    delay = 2 ** task.retries
                    print(f"Task {task.id} failed, retrying in {delay} seconds")
                    time.sleep(delay)
                    self.task_queue.add_task(task)
            finally:
                self.task_queue.status_tracker[task.id] = task.status

    def stop(self):
        self.stop_event.set()

def main():
    task_queue = TaskQueue(max_tasks=10)
    worker_thread = WorkerThread(task_queue)

    # Submit tasks
    task1 = Task(1, 1, "Task 1 data")
    task2 = Task(2, 2, "Task 2 data")
    task_queue.add_task(task1)
    task_queue.add_task(task2)

    # Start worker thread
    worker_thread.start()

    # Wait for 5 seconds
    time.sleep(5)

    # Stop worker thread
    worker_thread.stop()
    worker_thread.join()

if __name__ == "__main__":
    main()
```
