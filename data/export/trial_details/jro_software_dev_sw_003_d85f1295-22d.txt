import pandas as pd
import logging
from abc import ABC, abstractmethod
from typing import Callable, Dict

# Define the Transformation abstract base class
class Transformation(ABC):
    @abstractmethod
    def filter(self, data: pd.DataFrame) -> pd.DataFrame:
        pass

    @abstractmethod
    def map(self, data: pd.DataFrame) -> pd.DataFrame:
        pass

    @abstractmethod
    def aggregate(self, data: pd.DataFrame) -> pd.DataFrame:
        pass

# Define the Plugin Registry
class PluginRegistry:
    def __init__(self):
        self.plugins: Dict[str, Transformation] = {}

    def register_plugin(self, name: str, plugin: Transformation):
        self.plugins[name] = plugin

    def get_plugin(self, name: str) -> Transformation:
        return self.plugins.get(name)

# Define the Data Pipeline class
class DataPipeline:
    def __init__(self, plugin_registry: PluginRegistry):
        self.plugin_registry = plugin_registry

    def read_csv(self, file_path: str, schema: Dict[str, str]) -> pd.DataFrame:
        try:
            data = pd.read_csv(file_path)
            data = data.astype(schema)
            return data
        except Exception as e:
            logging.error(f"Error reading CSV file: {e}")
            raise

    def apply_transformations(self, data: pd.DataFrame, transformation_names: List[str]) -> pd.DataFrame:
        for transformation_name in transformation_names:
            plugin = self.plugin_registry.get_plugin(transformation_name)
            if plugin:
                data = plugin.filter(data)
                data = plugin.map(data)
                data = plugin.aggregate(data)
            else:
                logging.error(f"Transformation plugin {transformation_name} not found")
                raise ValueError(f"Transformation plugin {transformation_name} not found")
        return data

    def output_to_csv(self, data: pd.DataFrame, file_path: str):
        try:
            data.to_csv(file_path, index=False)
        except Exception as e:
            logging.error(f"Error outputting to CSV: {e}")
            raise

    def output_to_json(self, data: pd.DataFrame, file_path: str):
        try:
            data.to_json(file_path, orient="records")
        except Exception as e:
            logging.error(f"Error outputting to JSON: {e}")
            raise

# Define a custom transformation plugin
class CustomTransformation(Transformation):
    def filter(self, data: pd.DataFrame) -> pd.DataFrame:
        return data[data["age"] > 18]

    def map(self, data: pd.DataFrame) -> pd.DataFrame:
        data["name"] = data["name"].str.upper()
        return data

    def aggregate(self, data: pd.DataFrame) -> pd.DataFrame:
        return data.groupby("city").count()

# Usage example
plugin_registry = PluginRegistry()
plugin_registry.register_plugin("custom", CustomTransformation())

pipeline = DataPipeline(plugin_registry)
data = pipeline.read_csv("input.csv", {"name": str, "age": int, "city": str})
data = pipeline.apply_transformations(data, ["custom"])
pipeline.output_to_csv(data, "output.csv")

import pandas as pd
import logging
from abc import ABC, abstractmethod
from typing import Callable, Dict, List
import os
import unittest
from threading import Lock

# Define the Transformation abstract base class
class Transformation(ABC):
    @abstractmethod
    def filter(self, data: pd.DataFrame) -> pd.DataFrame:
        pass

    @abstractmethod
    def map(self, data: pd.DataFrame) -> pd.DataFrame:
        pass

    @abstractmethod
    def aggregate(self, data: pd.DataFrame) -> pd.DataFrame:
        pass

# Define the Plugin Registry
class PluginRegistry:
    def __init__(self):
        self.plugins: Dict[str, Transformation] = {}
        self.lock = Lock()

    def register_plugin(self, name: str, plugin: Transformation):
        with self.lock:
            self.plugins[name] = plugin

    def get_plugin(self, name: str) -> Transformation:
        with self.lock:
            return self.plugins.get(name)

# Define the Data Pipeline class
class DataPipeline:
    def __init__(self, plugin_registry: PluginRegistry):
        self.plugin_registry = plugin_registry

    def read_csv(self, file_path: str, schema: Dict[str, str]) -> pd.DataFrame:
        try:
            data = pd.read_csv(file_path)
            data = data.astype(schema)
            return data
        except Exception as e:
            logging.error(f"Error reading CSV file: {e}")
            raise

    def apply_transformations(self, data: pd.DataFrame, transformation_names: List[str]) -> pd.DataFrame:
        for transformation_name in transformation_names:
            plugin = self.plugin_registry.get_plugin(transformation_name)
            if plugin:
                data = plugin.filter(data)
                data = plugin.map(data)
                data = plugin.aggregate(data)
            else:
                logging.error(f"Transformation plugin {transformation_name} not found")
                raise ValueError(f"Transformation plugin {transformation_name} not found")
        return data

    def output_to_csv(self, data: pd.DataFrame, file_path: str):
        try:
            data.to_csv(file_path, index=False)
        except Exception as e:
            logging.error(f"Error outputting to CSV: {e}")
            raise

    def output_to_json(self, data: pd.DataFrame, file_path: str):
        try:
            data.to_json(file_path, orient="records")
        except Exception as e:
            logging.error(f"Error outputting to JSON: {e}")
            raise

# Define a custom transformation plugin
class CustomTransformation(Transformation):
    def filter(self, data: pd.DataFrame) -> pd.DataFrame:
        return data[data["age"] > 18]

    def map(self, data: pd.DataFrame) -> pd.DataFrame:
        data["name"] = data["name"].str.upper()
        return data

    def aggregate(self, data: pd.DataFrame) -> pd.DataFrame:
        return data.groupby("city").count()

# Define another custom transformation plugin
class AnotherCustomTransformation(Transformation):
    def filter(self, data: pd.DataFrame) -> pd.DataFrame:
        return data[data["age"] < 30]

    def map(self, data: pd.DataFrame) -> pd.DataFrame:
        data["name"] = data["name"].str.lower()
        return data

    def aggregate(self, data: pd.DataFrame) -> pd.DataFrame:
        return data.groupby("city").sum()

# Usage example
plugin_registry = PluginRegistry()
plugin_registry.register_plugin("custom", CustomTransformation())
plugin_registry.register_plugin("another_custom", AnotherCustomTransformation())

pipeline = DataPipeline(plugin_registry)
data = pipeline.read_csv("input.csv", {"name": str, "age": int, "city": str})
data = pipeline.apply_transformations(data, ["custom", "another_custom"])
pipeline.output_to_csv(data, "output.csv")
