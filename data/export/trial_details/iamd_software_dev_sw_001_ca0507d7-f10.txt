import hashlib
import threading
from urllib.parse import urlparse

class URLShortener:
    def __init__(self):
        self.url_dict = {}
        self.lock = threading.Lock()

    def shorten_url(self, long_url):
        """
        Shorten a long URL and return the shortened version.

        Args:
            long_url (str): The long URL to shorten.

        Returns:
            str: The shortened URL.
        """
        if not self.validate_url(long_url):
            raise ValueError("Invalid URL")

        with self.lock:
            hash_value = self.generate_hash(long_url)
            short_url = f"http://short.url/{hash_value}"

            if short_url in self.url_dict:
                # Collision detected, resolve it
                short_url = self.resolve_collision(short_url)

            self.url_dict[short_url] = long_url

        return short_url

    def reverse_lookup(self, short_url):
        """
        Perform a reverse lookup from a shortened URL to the original long URL.

        Args:
            short_url (str): The shortened URL to lookup.

        Returns:
            str: The original long URL.
        """
        with self.lock:
            return self.url_dict.get(short_url)

    def validate_url(self, long_url):
        """
        Validate the input URL to ensure it is in a valid format.

        Args:
            long_url (str): The input URL to validate.

        Returns:
            bool: True if the URL is valid, False otherwise.
        """
        try:
            result = urlparse(long_url)
            return all([result.scheme, result.netloc])
        except ValueError:
            return False

    def generate_hash(self, long_url):
        """
        Generate a unique hash for the shortened URL.

        Args:
            long_url (str): The long URL to generate a hash for.

        Returns:
            str: The generated hash.
        """
        return hashlib.sha256(long_url.encode()).hexdigest()[:6]

    def resolve_collision(self, short_url):
        """
        Resolve a collision by generating a new hash and checking for collisions again.

        Args:
            short_url (str): The shortened URL that collided.

        Returns:
            str: The new shortened URL.
        """
        new_hash = self.generate_hash(short_url)
        new_short_url = f"http://short.url/{new_hash}"

        if new_short_url in self.url_dict:
            # Recursively resolve the collision
            return self.resolve_collision(new_short_url)

        return new_short_url

short_url = url_shortener.shorten_url(long_url)
print(f"Shortened URL: {short_url}")

original_url = url_shortener.reverse_lookup(short_url)
print(f"Original URL: {original_url}")

import hashlib
import threading
from urllib.parse import urlparse, quote
from urllib.error import URLError

class URLShortener:
    def __init__(self):
        self.url_dict = {}
        self.lock = threading.Semaphore()
        self.counter = 0

    def shorten_url(self, long_url):
        """
        Shorten a long URL and return the shortened version.

        Args:
            long_url (str): The long URL to shorten.

        Returns:
            str: The shortened URL.

        Raises:
            ValueError: If the input URL is empty or invalid.
        """
        if not long_url:
            raise ValueError("Input URL cannot be empty")

        if not self.validate_url(long_url):
            raise ValueError("Invalid URL")

        if len(long_url) > 1000:
            raise ValueError("Input URL is too long")

        with self.lock:
            hash_value = self.generate_hash(long_url)
            short_url = f"http://short.url/{hash_value}"

            if short_url in self.url_dict:
                # Duplicate shortened URL, resolve it
                short_url = self.resolve_duplicate(short_url)

            self.url_dict[short_url] = long_url

        return short_url

    def reverse_lookup(self, short_url):
        """
        Perform a reverse lookup from a shortened URL to the original long URL.

        Args:
            short_url (str): The shortened URL to lookup.

        Returns:
            str: The original long URL.
        """
        with self.lock:
            return self.url_dict.get(short_url)

    def validate_url(self, long_url):
        """
        Validate the input URL to ensure it is in a valid format.

        Args:
            long_url (str): The input URL to validate.

        Returns:
            bool: True if the URL is valid, False otherwise.
        """
        try:
            result = urlparse(long_url)
            return all([result.scheme, result.netloc])
        except ValueError:
            return False

    def generate_hash(self, long_url):
        """
        Generate a unique hash for the shortened URL.

        Args:
            long_url (str): The long URL to generate a hash for.

        Returns:
            str: The generated hash.
        """
        return hashlib.blake2b(long_url.encode()).hexdigest()[:6]

    def resolve_duplicate(self, short_url):
        """
        Resolve a duplicate shortened URL by generating a new hash and checking for duplicates again.

        Args:
            short_url (str): The shortened URL that duplicated.

        Returns:
            str: The new shortened URL.
        """
        self.counter += 1
        new_hash = f"{hashlib.blake2b(short_url.encode()).hexdigest()[:6]}{self.counter}"
        new_short_url = f"http://short.url/{new_hash}"

        if new_short_url in self.url_dict:
            # Recursively resolve the duplicate
            return self.resolve_duplicate(new_short_url)

        return new_short_url
