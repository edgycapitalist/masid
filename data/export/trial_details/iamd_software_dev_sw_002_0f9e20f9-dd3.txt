**Architecture Document: Rate Limiter Module**

## Overview
The rate limiter module is designed to provide a thread-safe and configurable rate limiting system for Python applications. It utilizes the token bucket algorithm to manage the rate of requests. The module supports multiple named rate limiters, making it suitable for various use cases such as per-user or per-endpoint rate limiting.

## Module Components
1. **Token Bucket Algorithm**: This is the core component of the rate limiter. It uses a token bucket to track the available tokens (requests) and refills the bucket at a configurable rate.
2. **Rate Limiter Class**: This class represents a single rate limiter instance. It encapsulates the token bucket and provides methods for checking and consuming tokens.
3. **Rate Limiter Manager**: This class manages multiple named rate limiters. It allows creating, retrieving, and removing rate limiters by name.
4. **Decorator**: A convenience decorator is provided to easily apply rate limiting to functions.

## Data Flow
1. **Initialization**: The rate limiter manager is initialized with a dictionary of named rate limiters. Each rate limiter has a configurable rate (requests per second) and burst size (initial tokens).
2. **Request**: When a request is made, the decorator or the application code checks the rate limiter manager for the corresponding rate limiter by name.
3. **Token Check**: The rate limiter checks if there are sufficient tokens available in the bucket. If not, the request is blocked or an exception is raised.
4. **Token Consumption**: If tokens are available, the rate limiter consumes the required tokens and allows the request to proceed.
5. **Refill**: The token bucket is refilled at the configurable rate.

import threading
import time
from functools import wraps

class TokenBucket:
    def __init__(self, rate, burst):
        self.rate = rate
        self.burst = burst
        self.tokens = burst
        self.last_refill = time.time()

    def consume(self, amount=1):
        if self.tokens < amount:
            return False
        self.tokens -= amount
        return True

    def refill(self):
        now = time.time()
        elapsed = now - self.last_refill
        self.last_refill = now
        self.tokens = min(self.burst, self.tokens + elapsed * self.rate)

class RateLimiter:
    def __init__(self, rate, burst):
        self.token_bucket = TokenBucket(rate, burst)
        self.lock = threading.Lock()

    def allow(self):
        with self.lock:
            self.token_bucket.refill()
            return self.token_bucket.consume()

class RateLimiterManager:
    def __init__(self):
        self.rate_limiters = {}
        self.lock = threading.Lock()

    def get_rate_limiter(self, name, rate, burst):
        with self.lock:
            if name not in self.rate_limiters:
                self.rate_limiters[name] = RateLimiter(rate, burst)
            return self.rate_limiters[name]

def rate_limited(name, rate, burst):
    def decorator(func):
        rate_limiter_manager = RateLimiterManager()
        rate_limiter = rate_limiter_manager.get_rate_limiter(name, rate, burst)

        @wraps(func)
        def wrapper(*args, **kwargs):
            if not rate_limiter.allow():
                raise Exception("Rate limit exceeded")
            return func(*args, **kwargs)

        return wrapper

    return decorator

# Example usage:

import threading
import time
from functools import wraps

class TokenBucket:
    def __init__(self, rate, burst):
        self.rate = rate
        self.burst = burst
        self.tokens = burst
        self.last_refill = time.time()

        if self.tokens < amount:

        self.tokens -= amount
        return True

import threading
import time
from functools import wraps

class TokenBucket:
    def __init__(self, rate, burst):
        self.rate = rate
        self.burst = burst
        self.tokens = burst
        self.last_refill = time.time()

    def consume(self, amount=1):
        if self.tokens < amount:
            return False
        self.tokens -= amount
        return True

    def refill(self):
        now = time.time()
        elapsed = now - self.last_refill
        self.last_refill = now
        self.tokens = min(self.burst, self.tokens + elapsed * self.rate)

class RateLimiter:
    def __init__(self, rate, burst):
        self.token_bucket = TokenBucket(rate, burst)
        self.lock = threading.Lock()

    def allow(self):
        with self.lock:
            self.token_bucket.refill()
            return self.token_bucket.consume()

    def get_tokens(self):
        with self.lock:
            self.token_bucket.refill()
            return self.token_bucket.tokens

    def get_rate(self):
        return self.token_bucket.rate

class RateLimiterManager:
    def __init__(self):
        self.rate_limiters = {}
        self.lock = threading.Lock()

    def get_rate_limiter(self, name, rate, burst):
        with self.lock:
            if name not in self.rate_limiters:
                self.rate_limiters[name] = RateLimiter(rate, burst)
            return self.rate_limiters[name]

    def remove_rate_limiter(self, name):
        with self.lock:
            if name in self.rate_limiters:
                del self.rate_limiters[name]

def rate_limited(name, rate, burst):
    rate_limiter_manager = RateLimiterManager()
    rate_limiter = rate_limiter_manager.get_rate_limiter(name, rate, burst)

    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            if not rate_limiter.allow():
                raise Exception("Rate limit exceeded")
            return func(*args, **kwargs)

        return wrapper

    return decorator

# Example usage:
@rate_limited("my_limiter", 5, 10)  # 5 requests per second, burst size 10
