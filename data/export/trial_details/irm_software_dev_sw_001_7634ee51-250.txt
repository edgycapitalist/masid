class ThreadSafeDict:
    def __init__(self):
        self._dict = {}
        self._lock = threading.Lock()

    def put(self, key, value):
        with self._lock:
            self._dict[key] = value

    def get(self, key):
        with self._lock:
            return self._dict.get(key)

    def delete(self, key):
        with self._lock:
            if key in self._dict:
                del self._dict[key]

class UrlValidator:
    def __init__(self):
        pass

    def validate(self, url):
        try:
            result = urlparse(url)
            return all([result.scheme, result.netloc])
        except ValueError:
            return False

class HashUtil:
    def __init__(self):
        pass

    def generate_hash(self, url):

        self._url_validator = UrlValidator()

        self._lock = threading.Lock()

        with self._lock:
