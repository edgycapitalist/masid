import time
import functools

class RateLimiter:
    def __init__(self, name, rate, burst_size):
        self.name = name
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()

    def allow(self):
        now = time.time()
        elapsed = now - self.last_update
        self.tokens = min(self.burst_size, self.tokens + elapsed * self.rate)

        if self.tokens < 1:

        self.lock = threading.Lock()

class RateLimiter:
    """
    A rate limiter that uses the token bucket algorithm to manage the rate of requests.
    
    Attributes:
        name (str): A unique identifier for the rate limiter.
        rate (float): The maximum number of requests per second.
        burst_size (int): The maximum number of requests that can be made in a single burst.
        tokens (float): The current number of tokens in the bucket.
        last_update (float): The time of the last update to the token bucket.
    """

        """
        Initializes a new rate limiter with the specified name, rate, and burst size.
        
        Args:
            name (str): A unique identifier for the rate limiter.
            rate (float): The maximum number of requests per second.
            burst_size (int): The maximum number of requests that can be made in a single burst.
        
        Raises:
            ValueError: If the rate or burst size is invalid.
        """
        if rate <= 0:
            raise ValueError("Rate must be greater than zero")
        if burst_size <= 0:
            raise ValueError("Burst size must be greater than zero")
        
        self.name = name
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()

    def allow(self):
        """
        Checks if a request is allowed based on the current token bucket state.
        
        Returns:
            bool: True if the request is allowed, False otherwise.
        """
        now = time.time()
        elapsed = now - self.last_update
        self.tokens = min(self.burst_size, self.tokens + elapsed * self.rate)
        self.last_update = now
        if self.tokens < 1:
            return False
        self.tokens -= 1
        return True

    """
    A rate limiter that uses the token bucket algorithm to manage the rate of requests.
    
    Attributes:
        name (str): A unique identifier for the rate limiter.
        rate (float): The maximum number of requests per second.
        burst_size (int): The maximum number of requests that can be made in a single burst.
        tokens (float): The current number of tokens in the bucket.
        last_update (float): The time of the last update to the token bucket.

    def __init__(self, name, rate, burst_size):
        """
        Initializes a new rate limiter with the specified name, rate, and burst size.
        
        Args:
            name (str): A unique identifier for the rate limiter.
            rate (float): The maximum number of requests per second.
            burst_size (int): The maximum number of requests that can be made in a single burst.

        Initializes a new rate limiter with the specified name, rate, and burst size.
        
        Args:
            name (str): A unique identifier for the rate limiter.
            rate (float): The maximum number of requests per second.
            burst_size (int): The maximum number of requests that can be made in a single burst.
        
        Raises:
            ValueError: If the rate or burst size is invalid.
        """
        if rate <= 0:
            raise ValueError("Rate must be greater than zero")
        if burst_size <= 0:
            raise ValueError("Burst size must be greater than zero")
        
        self.name = name
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()

    def allow(self):
        """
        Checks if a request is allowed based on the current token bucket state.
        
        Returns:
            bool: True if the request is allowed, False otherwise.
        """
        now = time.time()
        elapsed = now - self.last_update
        self.tokens = min(self.burst_size, self.tokens + elapsed * self.rate)
        self.last_update = now
        if self.tokens < 1:
            return False
        self.tokens -= 1
        return True

    def reset(self):
        """
        Resets the rate limiter to its initial state.
        """
        self.tokens = self.burst_size
        self.last_update = time.time()

    def __repr__(self):
        return f"RateLimiter(name={self.name}, rate={self.rate}, burst_size={self.burst_size}, tokens={self.tokens}, last_update={self.last_update})"

class RateLimiterManager:
    """
    A manager for multiple named rate limiters.
    
    Attributes:
        rate_limiters (dict): A dictionary of named rate limiters.
        lock (threading.Lock): A lock for thread-safe access to the rate limiters.
    """

    def __init__(self):
        """
        Initializes a new rate limiter manager.
        """
        self.rate_limiters = {}
        self.lock = threading.Lock()

    def create_rate_limiter(self, name, rate, burst_size):
        """
        Creates a new rate limiter with the specified name, rate, and burst size.
        
        Args:
            name (str): A unique identifier for the rate limiter.
            rate (float): The maximum number of requests per second.
            burst_size (int): The maximum number of requests that can be made in a single burst.
        
        Returns:
            RateLimiter: The created rate limiter.
        """
        with self.lock:
            if name not in self.rate_limiters:
                self.rate_limiters[name] = RateLimiter(name, rate, burst_size)
            return self.rate_limiters[name]

    def get_rate_limiter(self, name):
        """
        Gets a rate limiter by its name.
        
        Args:
            name (str): The name of the rate limiter.
        
        Returns:
            RateLimiter: The rate limiter, or None if it does not exist.
        """
        with self.lock:
            return self.rate_limiters.get(name)

    def delete_rate_limiter(self, name):
        """
        Deletes a rate limiter by its name.
        
        Args:
            name (str): The name of the rate limiter.
        
        Returns:
            bool: True if the rate limiter was deleted, False otherwise.
        """
        with self.lock:
            if name in self.rate_limiters:
                del self.rate_limiters[name]
                return True
            return False

    def get_rate_limiter_status(self, name):
        """
        Gets the current status of a rate limiter by its name.
        
        Args:
            name (str): The name of the rate limiter.
        
        Returns:
            dict: A dictionary containing the current status of the rate limiter.
        """
        with self.lock:
            rate_limiter = self.rate_limiters.get(name)
            if rate_limiter:
                return {
                    "name": rate_limiter.name,
                    "rate": rate_limiter.rate,
                    "burst_size": rate_limiter.burst_size,
                    "tokens": rate_limiter.tokens,
                    "last_update": rate_limiter.last_update
                }
            return None

class RateLimiterDecorator:
    """
    A decorator that applies rate limiting to functions.
    
    Attributes:
        rate_limiter (RateLimiter): The rate limiter to apply to the function.
        max_retries (int): The maximum number of retries if the rate limit is exceeded.
    """

    def __init__(self, rate_limiter, max_retries=0):
        """
        Initializes a new rate limiter decorator with the specified rate limiter and maximum retries.
        
        Args:
            rate_limiter (RateLimiter): The rate limiter to apply to the function.
            max_retries (int): The maximum number of retries if the rate limit is exceeded. Defaults to 0.
        """
        self.rate_limiter = rate_limiter
        self.max_retries = max_retries

    def __call__(self, func):
        """
        Applies the rate limiter to the specified function.
        
        Args:
            func (callable): The function to apply the rate limiter to.
        
        Returns:
            callable: The decorated function.
        """
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            retries = 0
            while retries <= self.max_retries:
                if self.rate_limiter.allow():
                    return func(*args, **kwargs)
                retries += 1
                time.sleep(1)  # wait for 1 second before retrying
            raise Exception(f"Rate limit exceeded for {self.rate_limiter.name} after {self.max_retries} retries")
        return wrapper
