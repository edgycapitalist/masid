1. The user provides a long URL to the `shorten_url` function.
2. The `shorten_url` function calls the `url_validator` module to validate the input URL.

6. When the user provides a shortened URL to the `get_original_url` function, the function calls the `storage` module to retrieve the original URL.
7. The `get_original_url` function returns the original URL to the user.

# url_shortener.py
import hashlib
import threading
import sqlite3
from url_validator import is_valid_url
from storage import Storage

class URLShortener:
    def __init__(self):
        self.storage = Storage()
        self.lock = threading.Lock()

    def shorten_url(self, long_url: str) -> str:
        if not is_valid_url(long_url):
            raise ValueError("Invalid URL")

        with self.lock:
            hash_value = hashlib.sha256(long_url.encode()).hexdigest()[:6]
            short_url = f"http://short.url/{hash_value}"

            if self.storage.exists(short_url):
                # Handle collision detection
                short_url = self._generate_alternate_short_url(hash_value)

            self.storage.store(long_url, short_url)
            return short_url

    def get_original_url(self, short_url: str) -> str:
        with self.lock:
            return self.storage.retrieve(short_url)

    def _generate_alternate_short_url(self, hash_value: str) -> str:
        # Generate an alternate short URL in case of collision
        return f"http://short.url/{hash_value}_{int(hash_value, 16) % 1000}"

class URLShortener:
    def __init__(self):
        self.storage = Storage()
        self.lock = threading.Lock()

    def shorten_url(self, long_url: str) -> str:
        """
        Shortens a given long URL and returns the shortened version.

        Args:
            long_url (str): The long URL to be shortened.

        Returns:
            str: The shortened URL.

        Raises:
            URLShortenerException: If the input URL is not valid.

        Args:

class URLShortener:
    def __init__(self):
        self.storage = Storage()
        self.lock = threading.Lock()

    def shorten_url(self, long_url: str) -> str:
        """
        Shortens a given long URL and returns the shortened version.

        Args:
            long_url (str): The long URL to be shortened.

        Returns:
            str: The shortened URL.

        Raises:
            URLShortenerException: If the input URL is not valid.
        """
        if not is_valid_url(long_url):
            raise URLShortenerException("Invalid URL")

        with self.lock:
            hash_value = hashlib.sha256(long_url.encode()).hexdigest()[:6]
            short_url = f"http://short.url/{hash_value}"

            if self.storage.exists(short_url):
                # Handle collision detection
                short_url = self._generate_alternate_short_url(hash_value)

            self.storage.store(long_url, short_url)
            return short_url

        """
        Retrieves the original URL for a given shortened URL.

        Args:
            short_url (str): The shortened URL.

        Returns:
            str: The original URL.

        Raises:
            URLShortenerException: If the shortened URL is not found.

        Returns:

import threading

from url_validator import is_valid_url
from storage import Storage
from exceptions import URLShortenerException

class URLShortener:
    def __init__(self):
        self.storage = Storage()
        self.lock = threading.Lock()

    def shorten_url(self, long_url: str) -> str:
        """
        Shortens a given long URL and returns the shortened version.

        Args:
            long_url (str): The long URL to be shortened.

        Returns:
            str: The shortened URL.

        Raises:
            URLShortenerException: If the input URL is not valid.
        """
        if not is_valid_url(long_url):
            raise URLShortenerException("Invalid URL")

        with self.lock:
            hash_value = hashlib.sha256(long_url.encode()).hexdigest()[:6]
            short_url = f"http://short.url/{hash_value}"

            if self.storage.exists(short_url):
                # Handle collision detection
                short_url = self._generate_alternate_short_url(hash_value)

            self.storage.store(long_url, short_url)
            return short_url

    def get_original_url(self, short_url: str) -> str:
        """
        Retrieves the original URL for a given shortened URL.

        Args:
            short_url (str): The shortened URL.

        Returns:
            str: The original URL.

        Raises:
            URLShortenerException: If the shortened URL is not found.
        """
        with self.lock:
            return self.storage.retrieve(short_url)

    def _generate_alternate_short_url(self, hash_value: str) -> str:
        """
        Generates an alternate short URL in case of collision.

        Args:
            hash_value (str): The hash value of the original URL.

        Returns:
            str: The alternate short URL.
        """
        import uuid
        return f"http://short.url/{hash_value}_{uuid.uuid4().hex[:6]}"
