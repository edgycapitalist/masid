import threading
import time
from functools import wraps

class TokenBucket:
    def __init__(self, rate, burst_size):
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()

    def consume_token(self):
        current_time = time.time()
        elapsed_time = current_time - self.last_update
        self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)
        self.last_update = current_time

        if self.tokens < 1:
            return False
        else:
            self.tokens -= 1
            return True

    def add_tokens(self, tokens):
        self.tokens = min(self.burst_size, self.tokens + tokens)

class RateLimiter:
    def __init__(self, name, rate, burst_size):
        self.name = name
        self.rate = rate
        self.burst_size = burst_size
        self.token_bucket = TokenBucket(rate, burst_size)
        self.lock = threading.Lock()

    def allow_request(self):
        with self.lock:
            return self.token_bucket.consume_token()

    def add_tokens(self, tokens):
        with self.lock:
            self.token_bucket.add_tokens(tokens)

def rate_limiter_decorator(rate_limiter):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            if rate_limiter.allow_request():
                return func(*args, **kwargs)
            else:
                raise Exception("Rate limit exceeded")
        return wrapper
    return decorator
