**Architecture Document: URL Shortening Library**

### Introduction

The URL Shortening Library is a Python module that provides a hash-based approach to shorten URLs, supports reverse lookup, handles collision detection, and includes input validation for valid URL formats. The library is designed to be thread-safe for concurrent usage, ensuring reliability and performance in a multi-threaded environment.

### High-Level Design

The library consists of the following components:

1. **URLValidator**: Responsible for validating the input URL format.
2. **URLShortener**: Uses a hash-based approach to shorten the URL.
3. **URLMapper**: Maps the shortened URL to the original URL and vice versa.
4. **CollisionDetector**: Detects and handles collisions that may occur during the shortening process.

### Module Decomposition

The library is decomposed into the following modules:

* `url_validator.py`: Contains the `URLValidator` class.
* `url_shortener.py`: Contains the `URLShortener` class.
* `url_mapper.py`: Contains the `URLMapper` class.
* `collision_detector.py`: Contains the `CollisionDetector` class.
* `url_shortening_library.py`: The main library module that integrates the above components.

### Interfaces

The library provides the following interfaces:

* `shorten_url(long_url)`: Returns a shortened version of the input URL.
* `get_original_url(short_url)`: Returns the original URL corresponding to the input short URL.

### Data Flow

The data flow is as follows:

1. The user calls the `shorten_url` method with a long URL as input.
2. The `URLValidator` validates the input URL format.
3. If valid, the `URLShortener` generates a shortened URL using a hash-based approach.
4. The `CollisionDetector` checks for collisions and handles them if necessary.
5. The `URLMapper` maps the shortened URL to the original URL.
6. The user can retrieve the original URL by calling the `get_original_url` method with the short URL as input.

### Technology Choices

* Python 3.9 or later as the programming language.
* `hashlib` library for generating hashes.
* `threading` library for ensuring thread safety.

# url_validator.py
import re

class URLValidator:
    def __init__(self):
        self.url_pattern = re.compile(r'^https?://[^\s/$.?#].[^\s]*$')

    def validate(self, url):
        return bool(self.url_pattern.match(url))

# url_shortener.py
import hashlib

class URLShortener:
    def __init__(self):
        self.hash_function = hashlib.sha256

    def shorten(self, url):
        hash_object = self.hash_function(url.encode())
        return hash_object.hexdigest()[:6]

# url_mapper.py
import threading

class URLMapper:
    def __init__(self):
        self.url_map = {}
        self.lock = threading.Lock()

    def add_mapping(self, short_url, long_url):
        with self.lock:
            self.url_map[short_url] = long_url

    def get_original_url(self, short_url):
        with self.lock:
            return self.url_map.get(short_url)

# collision_detector.py
import random

class CollisionDetector:
    def __init__(self):
        self.collision_count = 0

    def detect_collision(self, short_url, url_map):
        if short_url in url_map:
            self.collision_count += 1
            return True
        return False

    def handle_collision(self, short_url, long_url, url_map):
        new_short_url = self.generate_new_short_url(short_url)
        url_map[new_short_url] = long_url
        return new_short_url

    def generate_new_short_url(self, short_url):
        return short_url + str(random.randint(0, 1000))

# url_shortening_library.py
from url_validator import URLValidator
from url_shortener import URLShortener
from url_mapper import URLMapper
from collision_detector import CollisionDetector

class URLShorteningLibrary:
    def __init__(self):
        self.url_validator = URLValidator()
        self.url_shortener = URLShortener()
        self.url_mapper = URLMapper()
        self.collision_detector = CollisionDetector()

    def shorten_url(self, long_url):
        if not self.url_validator.validate(long_url):
            raise ValueError("Invalid URL format")

        short_url = self.url_shortener.shorten(long_url)
        if self.collision_detector.detect_collision(short_url, self.url_mapper.url_map):
            short_url = self.collision_detector.handle_collision(short_url, long_url, self.url_mapper.url_map)
        else:
            self.url_mapper.add_mapping(short_url, long_url)

        return short_url

    def get_original_url(self, short_url):
        return self.url_mapper.get_original_url(short_url)

# Example usage
library = URLShorteningLibrary()
long_url = "https://www.example.com/very/long/url"
short_url = library.shorten_url(long_url)
print(f"Short URL: {short_url}")
print(f"Original URL: {library.get_original_url(short_url)}")

import random

class CollisionDetector:
    def __init__(self):
        self.collision_count = 0

    def detect_collision(self, short_url, url_map):
        if short_url in url_map:
            self.collision_count += 1
            return True
        return False

    def handle_collision(self, short_url, long_url, url_map):
        new_short_url = self.generate_new_short_url(short_url)
        url_map[new_short_url] = long_url
        return new_short_url

    def generate_new_short_url(self, short_url):
        return short_url + str(random.randint(0, 1000))

from url_validator import URLValidator, InvalidURLException
from url_shortener import URLShortener
from url_mapper import URLMapper
from collision_detector import CollisionDetector

class URLShorteningLibrary:
    def __init__(self):
        self.url_validator = URLValidator()
        self.url_shortener = URLShortener()
        self.url_mapper = URLMapper()
        self.collision_detector = CollisionDetector()

    def shorten_url(self, long_url):
        try:
            self.url_validator.validate(long_url)
        except InvalidURLException as e:
            raise e

        short_url = self.url_shortener.shorten(long_url)
        if self.collision_detector.detect_collision(short_url, self.url_mapper.url_map):
            short_url = self.collision_detector.handle_collision(short_url, long_url, self.url_mapper.url_map)
        else:
            self.url_mapper.add_mapping(short_url, long_url)

        return short_url

    def get_original_url(self, short_url):
        return self.url_mapper.get_original_url(short_url)
