import threading
import functools
import logging
import time

logging.basicConfig(level=logging.INFO)

class RateLimiter:
    def __init__(self, rate, burst_size):
        """
        Initializes the rate limiter with the given rate and burst size.

        :param rate: The rate at which tokens are added to the bucket (tokens per second).
        :param burst_size: The maximum number of tokens in the bucket.
        """
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()
        self.lock = threading.Lock()

    def consume_token(self):
        """
        Consumes a token from the bucket, blocking if necessary.
        """
        with self.lock:
            while self.tokens <= 0:
                time.sleep(0.01)
                self._update_tokens()
            self.tokens -= 1

    def check_token(self):
        """
        Checks if a token is available in the bucket.
        """
        with self.lock:
            self._update_tokens()
            return self.tokens > 0

    def _update_tokens(self):
        """
        Updates the number of tokens in the bucket based on the elapsed time.
        """
        now = time.time()
        elapsed = now - self.last_update
        self.last_update = now
        self.tokens = min(self.burst_size, self.tokens + elapsed * self.rate)

class RateLimiterFactory:
    def __init__(self):
        """
        Initializes the rate limiter factory.
        """
        self.rate_limiters = {}

    def create_rate_limiter(self, name, rate, burst_size):
        """
        Creates a new named rate limiter.

        :param name: The name of the rate limiter.
        :param rate: The rate at which tokens are added to the bucket (tokens per second).
        :param burst_size: The maximum number of tokens in the bucket.
        """
        self.rate_limiters[name] = RateLimiter(rate, burst_size)

    def get_rate_limiter(self, name):
        """
        Returns the named rate limiter.

        :param name: The name of the rate limiter.
        :return: The named rate limiter.
        """
        return self.rate_limiters[name]

def rate_limiter_decorator(rate_limiter):
    """
    Applies the rate limiter to a function.

    :param rate_limiter: The rate limiter to apply.
    :return: The decorated function.
    """
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            if not rate_limiter.check_token():
                logging.warning("Rate limit exceeded")
                rate_limiter.consume_token()
            return func(*args, **kwargs)
        return wrapper
    return decorator
