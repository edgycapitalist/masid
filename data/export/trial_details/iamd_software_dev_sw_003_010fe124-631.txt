## Introduction

import pandas as pd
from pydantic import BaseModel
from typing import List

# Data Ingestion
class DataIngestion:
    def __init__(self, schema: BaseModel):
        self.schema = schema

    def read_csv(self, file_path: str) -> pd.DataFrame:
        try:
            data = pd.read_csv(file_path)
            self.schema.validate(data)
            return data
        except Exception as e:
            raise ValueError(f"Error reading CSV file: {e}")

# Transformation Engine
class TransformationEngine:
    def __init__(self):
        self.transformations = []

    def apply_transformations(self, data: pd.DataFrame) -> pd.DataFrame:
        for transformation in self.transformations:
            data = transformation(data)
        return data

    def add_transformation(self, transformation):
        self.transformations.append(transformation)

# Plugin System
class PluginSystem:
    def __init__(self):
        self.plugins = []

    def register_plugin(self, plugin):
        self.plugins.append(plugin)

# Data Output
class DataOutput:
    def __init__(self):
        pass

    def output_to_csv(self, data: pd.DataFrame, file_path: str):
        data.to_csv(file_path, index=False)

    def output_to_json(self, data: pd.DataFrame, file_path: str):
        data.to_json(file_path, orient="records")

# Error Reporting
class ErrorReporting:
    def __init__(self):
        pass

    def report_error(self, error: str):
        print(f"Error: {error}")

# Example usage
if __name__ == "__main__":
    # Define schema
    class Schema(BaseModel):
        name: str
        age: int

    # Create data ingestion module
