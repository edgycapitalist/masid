import threading
import time
from functools import wraps

class TokenBucket:
    """Implements the token bucket algorithm."""

    def __init__(self, rate, burst):
        """
        Initializes the token bucket.

        :param rate: The rate at which tokens are added (requests per second).
        :param burst: The maximum number of tokens in the bucket.
        """
        self.rate = rate
        self.burst = burst
        self.tokens = burst
        self.last_update = time.time()

    def add_tokens(self):
        """Adds tokens to the bucket based on the elapsed time."""
        now = time.time()
        elapsed = now - self.last_update
        self.last_update = now
        self.tokens = min(self.burst, self.tokens + elapsed * self.rate)

    def consume_token(self):
        """Consumes a token from the bucket."""
        self.add_tokens()
        if self.tokens < 1:
            return False
        self.tokens -= 1
        return True


class RateLimiter:
    """Manages the rate limiting logic."""

    def __init__(self, rate, burst):
        """
        Initializes the rate limiter.

        :param rate: The rate at which tokens are added (requests per second).
        :param burst: The maximum number of tokens in the bucket.
        """
        self.bucket = TokenBucket(rate, burst)
        self.lock = threading.Lock()

    def allow(self):
        """Checks if a request is allowed."""
        with self.lock:
            return self.bucket.consume_token()


class RateLimiterDecorator:
    """A decorator that applies rate limiting to functions."""

    def __init__(self, limiter):
        """
        Initializes the decorator.

        :param limiter: The rate limiter instance.
        """
        self.limiter = limiter

    def __call__(self, func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            if not self.limiter.allow():
                raise Exception("Rate limit exceeded")
            return func(*args, **kwargs)
        return wrapper


class RateLimiterFactory:
    """A factory class that creates and manages named rate limiters."""

    def __init__(self):
        self.limiters = {}
        self.lock = threading.Lock()

    def get_limiter(self, name):
        """
        Retrieves a named rate limiter.

        :param name: The name of the rate limiter.
        :return: The rate limiter instance.
        """
        with self.lock:
            return self.limiters.get(name)

    def create_limiter(self, name, rate, burst):
        """
        Creates a new named rate limiter.

        :param name: The name of the rate limiter.
        :param rate: The rate at which tokens are added (requests per second).
        :param burst: The maximum number of tokens in the bucket.
        :return: The rate limiter instance.
        """
        with self.lock:
            if name in self.limiters:
                raise Exception("Rate limiter already exists")
            limiter = RateLimiter(rate, burst)
            self.limiters[name] = limiter
            return limiter


# Example usage:
factory = RateLimiterFactory()
limiter = factory.create_limiter("my_limiter", 10, 20)

@decorator
def my_function():

import threading
import time
from functools import wraps

class TokenBucket:
    """Implements the token bucket algorithm."""

    def __init__(self, rate, burst):
        """
        Initializes the token bucket.

        :param rate: The rate at which tokens are added (requests per second).
        :param burst: The maximum number of tokens in the bucket.
        """
        self.rate = rate
        self.burst = burst
        self.tokens = burst
        self.last_update = time.time()

    def add_tokens(self):
        """Adds tokens to the bucket based on the elapsed time."""
        now = time.time()
        elapsed = now - self.last_update
        self.last_update = now
        self.tokens = min(self.burst, self.tokens + elapsed * self.rate)

    def consume_token(self):
        """Consumes a token from the bucket."""
        self.add_tokens()
        if self.tokens < 1:
            return False
        self.tokens -= 1
        return True

    def get_tokens(self):
        """Returns the current number of tokens in the bucket."""
        self.add_tokens()
        return self.tokens


class RateLimiter:
    """Manages the rate limiting logic."""

    def __init__(self, rate, burst):
        """
        Initializes the rate limiter.

        :param rate: The rate at which tokens are added (requests per second).
        :param burst: The maximum number of tokens in the bucket.
        """
        self.bucket = TokenBucket(rate, burst)
        self.lock = threading.Lock()

    def allow(self):
        """Checks if a request is allowed."""
        with self.lock:
            return self.bucket.consume_token()

    def get_rate(self):
        """Returns the rate at which tokens are added."""
        return self.bucket.rate

    def get_burst(self):
        """Returns the maximum number of tokens in the bucket."""
        return self.bucket.burst

    def get_tokens(self):
        """Returns the current number of tokens in the bucket."""
        with self.lock:
            return self.bucket.get_tokens()


class RateLimiterDecorator:
    """A decorator that applies rate limiting to functions."""

    def __init__(self, limiter):
        """
        Initializes the decorator.

        :param limiter: The rate limiter instance.
        """
        self.limiter = limiter

    def __call__(self, func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            if not self.limiter.allow():
                raise Exception("Rate limit exceeded")
            return func(*args, **kwargs)
        return wrapper

    def __get__(self, instance, owner):
        if instance is None:
            return self
        return RateLimiterDecorator(self.limiter)


class RateLimiterFactory:
    """A factory class that creates and manages named rate limiters."""

    def __init__(self):
        self.limiters = {}
        self.lock = threading.Lock()

    def get_limiter(self, name):
        """
        Retrieves a named rate limiter.

        :param name: The name of the rate limiter.
        :return: The rate limiter instance.
        """
        with self.lock:
            return self.limiters.get(name)

    def create_limiter(self, name, rate, burst):
        """
        Creates a new named rate limiter.

        :param name: The name of the rate limiter.
        :param rate: The rate at which tokens are added (requests per second).
        :param burst: The maximum number of tokens in the bucket.
        :return: The rate limiter instance.
        """
        with self.lock:
            if name in self.limiters:
                raise Exception("Rate limiter already exists")
            limiter = RateLimiter(rate, burst)
            self.limiters[name] = limiter
            return limiter

    def delete_limiter(self, name):
        """
        Deletes a named rate limiter.

        :param name: The name of the rate limiter.
        """
        with self.lock:
            if name not in self.limiters:
                raise Exception("Rate limiter does not exist")
            del self.limiters[name]

    def update_limiter(self, name, rate, burst):
        """
        Updates a named rate limiter.

        :param name: The name of the rate limiter.
        :param rate: The new rate at which tokens are added (requests per second).
        :param burst: The new maximum number of tokens in the bucket.
        """
        with self.lock:
            if name not in self.limiters:
                raise Exception("Rate limiter does not exist")
            self.limiters[name].bucket.rate = rate
            self.limiters[name].bucket.burst = burst
