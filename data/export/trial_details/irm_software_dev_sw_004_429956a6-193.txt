## Introduction

This document outlines the design and implementation of an in-memory task queue in Python. The task queue is designed to support adding tasks with priorities, allow worker threads to consume tasks, provide task status tracking, support task retry with configurable max retries, and include graceful shutdown.

## System Components

The task queue system consists of the following components:

1.  **Task**: Represents a single task with a unique ID, priority, and a callable function to execute.
2.  **Task Queue**: Manages the tasks and provides methods for adding tasks, consuming tasks, and tracking task status.
3.  **Worker Thread**: Consumes tasks from the task queue and executes them.
4.  **Task Status Tracker**: Tracks the status of each task (pending, running, completed, failed).

## System Design

The task queue system uses the following design patterns:

1.  **Producer-Consumer Pattern**: The task queue acts as a buffer between the task producers (add_task method) and task consumers (worker threads).
2.  **Observer Pattern**: The task status tracker observes the task queue and updates the task status accordingly.

## System Flow

The system flow is as follows:

1.  **Add Task**: A task is added to the task queue with a unique ID, priority, and a callable function to execute.
2.  **Consume Task**: A worker thread consumes a task from the task queue based on the task priority.
3.  **Execute Task**: The worker thread executes the task and updates the task status to "running".
4.  **Task Completion**: If the task completes successfully, the task status is updated to "completed". If the task fails, the task status is updated to "failed" and the task is retried if the max retries are not exceeded.
5.  **Task Retry**: If a task fails, it is retried with a configurable number of max retries.
6.  **Shutdown**: The system is shut down gracefully, ensuring that all tasks are completed or failed.

## Implementation Details

*   The task queue is implemented using a priority queue (heapq) to ensure that tasks are consumed based on their priority.
*   The task status tracker uses a dictionary to store the task status for each task ID.
*   The worker threads use a thread pool (concurrent.futures) to execute tasks concurrently.
*   The system uses a lock (threading.Lock) to ensure thread safety when accessing shared resources.

import heapq
import threading
import concurrent.futures
from enum import Enum

class TaskStatus(Enum):
    PENDING = 1
    RUNNING = 2
    COMPLETED = 3
    FAILED = 4

class Task:
    def __init__(self, task_id, priority, func, max_retries=3):
        self.task_id = task_id
        self.priority = priority
        self.func = func
        self.max_retries = max_retries
        self.retries = 0
        self.status = TaskStatus.PENDING

class TaskQueue:
    def __init__(self):
        self.task_queue = []
        self.task_status = {}
        self.lock = threading.Lock()

    def add_task(self, task):
        with self.lock:
            heapq.heappush(self.task_queue, (task.priority, task))
            self.task_status[task.task_id] = task.status

    def consume_task(self):
        with self.lock:
            if not self.task_queue:
                return None
            _, task = heapq.heappop(self.task_queue)
            self.task_status[task.task_id] = TaskStatus.RUNNING
            return task

    def update_task_status(self, task_id, status):
        with self.lock:
            self.task_status[task_id] = status

class WorkerThread:
    def __init__(self, task_queue):
        self.task_queue = task_queue

    def run(self):
        while True:
            task = self.task_queue.consume_task()
            if task is None:
                break
            try:
                task.func()
                self.task_queue.update_task_status(task.task_id, TaskStatus.COMPLETED)
            except Exception as e:
                if task.retries < task.max_retries:
                    task.retries += 1
                    self.task_queue.add_task(task)
                else:
                    self.task_queue.update_task_status(task.task_id, TaskStatus.FAILED)

def main():
    task_queue = TaskQueue()

    # Add tasks to the task queue
    task_queue.add_task(Task(1, 1, lambda: print("Task 1 executed")))
    task_queue.add_task(Task(2, 2, lambda: print("Task 2 executed")))

    # Create and start worker threads
    with concurrent.futures.ThreadPoolExecutor(max_workers=2) as executor:
        futures = []
        for _ in range(2):
            worker_thread = WorkerThread(task_queue)
            futures.append(executor.submit(worker_thread.run))

        # Wait for all tasks to complete
        for future in concurrent.futures.as_completed(futures):
            future.result()

    # Print task status
    for task_id, status in task_queue.task_status.items():
        print(f"Task {task_id}: {status.name}")

if __name__ == "__main__":
    main()

class TaskStatus(Enum):
    PENDING = 1
    RUNNING = 2
    COMPLETED = 3
    FAILED = 4

class Task:
    def __init__(self, task_id, priority, func, max_retries=3):
        self.task_id = task_id
        self.priority = priority
        self.func = func
        self.max_retries = max_retries
        self.retries = 0
        self.status = TaskStatus.PENDING

    def execute(self):
        try:
            self.func()
            self.status = TaskStatus.COMPLETED
            logger.info(f"Task {self.task_id} completed successfully")
        except Exception as e:
            self.retries += 1
            if self.retries < self.max_retries:
                self.status = TaskStatus.PENDING
                logger.info(f"Task {self.task_id} failed, retrying...")
            else:
                self.status = TaskStatus.FAILED
                logger.error(f"Task {self.task_id} failed after {self.max_retries} retries")

class TaskQueue:
    def __init__(self, max_size=100):
        self.task_queue = []
        self.task_status = {}
        self.lock = threading.Lock()
        self.max_size = max_size

    def add_task(self, task):
        with self.lock:
            if len(self.task_queue) < self.max_size:
                heapq.heappush(self.task_queue, (task.priority, task))
                self.task_status[task.task_id] = task.status
                logger.info(f"Task {task.task_id} added to queue")
            else:
                logger.error("Task queue is full")

    def consume_task(self):
        with self.lock:
            if not self.task_queue:
                return None
            _, task = heapq.heappop(self.task_queue)
            self.task_status[task.task_id] = TaskStatus.RUNNING
            logger.info(f"Task {task.task_id} consumed from queue")
            return task

    def update_task_status(self, task_id, status):
        with self.lock:
            self.task_status[task_id] = status

class WorkerThread:
    def __init__(self, task_queue):
        self.task_queue = task_queue

    def run(self):
        while True:
            task = self.task_queue.consume_task()
            if task is None:
                break
            task.execute()
            self.task_queue.update_task_status(task.task_id, task.status)

def main():
    task_queue = TaskQueue()

    # Add tasks to the task queue
    task_queue.add_task(Task(1, 1, lambda: print("Task 1 executed")))

            futures.append(executor.submit(worker_thread.run))

        for future in concurrent.futures.as_completed(futures):
            future.result()

    # Print task status
    for task_id, status in task_queue.task_status.items():

class TaskStatus(Enum):

class Task:
    def __init__(self, task_id, priority, func, max_retries=3, timeout=60):

    def execute(self):
        try:
            self.func()
            self.status = TaskStatus.COMPLETED
            logger.info(f"Task {self.task_id} completed successfully")
        except Exception as e:
            self.retries += 1
            if self.retries < self.max_retries:
                self.status = TaskStatus.PENDING
                logger.info(f"Task {self.task_id} failed, retrying...")
            else:
                self.status = TaskStatus.FAILED
                logger.error(f"Task {self.task_id} failed after {self.max_retries} retries")

class TaskQueue:
    def __init__(self, max_size=100):
        self.task_queue = []
        self.task_status = {}
        self.lock = threading.Lock()
        self.max_size = max_size

    def add_task(self, task):
        with self.lock:
            if len(self.task_queue) < self.max_size:
                heapq.heappush(self.task_queue, (task.priority, task))
                self.task_status[task.task_id] = task.status
                logger.info(f"Task {task.task_id} added to queue")
            else:
                logger.error("Task queue is full")

    def consume_task(self):
        with self.lock:
            if not self.task_queue:
                return None
            _, task = heapq.heappop(self.task_queue)
            self.task_status[task.task_id] = TaskStatus.RUNNING
            logger.info(f"Task {task.task_id} consumed from queue")
            return task

    def update_task_status(self, task_id, status):
        with self.lock:
            self.task_status[task_id] = status

class WorkerThread:
    def __init__(self, task_queue):
        self.task_queue = task_queue

    def run(self):
        while True:
            task = self.task_queue.consume_task()
            if task is None:
                break
            start_time = time.time()
            task.execute()
            end_time = time.time()
            if end_time - start_time > task.timeout:
                task.status = TaskStatus.FAILED
                logger.error(f"Task {task.task_id} timed out")
            self.task_queue.update_task_status(task.task_id, task.status)
