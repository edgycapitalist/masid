import pandas as pd
import csv
import json
from pluginlib import Plugin

class DataPipeline:
    def __init__(self, schema_file, input_file, output_file):
        self.schema_file = schema_file
        self.input_file = input_file
        self.output_file = output_file

    def ingest_data(self):
        # Read CSV file and perform schema validation
        schema = json.load(open(self.schema_file))
        data = pd.read_csv(self.input_file)
        if not self.validate_data(data, schema):
            raise ValueError("Invalid data")

        return data

    def validate_data(self, data, schema):
        # Validate data against schema
        for column in schema:
            if column not in data.columns:
                return False
            if schema[column]['type'] == 'int' and not pd.api.types.is_integer_dtype(data[column]):
                return False
            if schema[column]['type'] == 'float' and not pd.api.types.is_float_dtype(data[column]):
                return False
        return True

    def apply_transformations(self, data):
        # Apply transformation chain
        transformations = [
            self.filter_data,
            self.map_data,
            self.aggregate_data
        ]
        for transformation in transformations:
            data = transformation(data)
        return data

    def filter_data(self, data):
        # Filter data
        return data[data['column'] > 0]

    def map_data(self, data):
        # Map data
        data['column'] = data['column'].apply(lambda x: x * 2)
        return data

    def aggregate_data(self, data):
        # Aggregate data
        return data.groupby('column').sum()

    def output_data(self, data):
        # Write data to CSV or JSON file
        if self.output_file.endswith('.csv'):
            data.to_csv(self.output_file, index=False)
        elif self.output_file.endswith('.json'):
            data.to_json(self.output_file, orient='records')

    def run(self):
        data = self.ingest_data()
        data = self.apply_transformations(data)
        self.output_data(data)

        """
        Initializes the data pipeline with the schema file, input file, and output file.
        
        Args:
        schema_file (str): The path to the schema file.
        input_file (str): The path to the input file.
        output_file (str): The path to the output file.
        """
        self.schema_file = schema_file
        self.input_file = input_file
        self.output_file = output_file

    def ingest_data(self) -> pd.DataFrame:
        """
        Reads the input file and performs schema validation.
        
        Returns:
        pd.DataFrame: The ingested data.
        """
        # Read CSV file and perform schema validation
        schema = self.load_schema()
        data = pd.read_csv(self.input_file)
        if not self.validate_data(data, schema):
            raise ValueError("Invalid data")

        return data

    def load_schema(self) -> Dict:
        """
        Loads the schema from the schema file.
        
        Returns:
        Dict: The schema.
        """
        with open(self.schema_file, 'r') as f:
            schema = json.load(f)
        return schema

    def validate_data(self, data: pd.DataFrame, schema: Dict) -> bool:
        """
        Validates the data against the schema.
        
        Args:
        data (pd.DataFrame): The data to validate.
        schema (Dict): The schema to validate against.
        
        Returns:
        bool: True if the data is valid, False otherwise.
        """
        for column in schema:
            if column not in data.columns:
                return False
            if schema[column]['type'] == 'int' and not pd.api.types.is_integer_dtype(data[column]):
                return False
            if schema[column]['type'] == 'float' and not pd.api.types.is_float_dtype(data[column]):
                return False
        return True

    def apply_transformations(self, data: pd.DataFrame) -> pd.DataFrame:
        """
        Applies the transformation chain to the data.
        
        Args:
        data (pd.DataFrame): The data to transform.
        
        Returns:
        pd.DataFrame: The transformed data.
        """
        transformations = [
            self.filter_data,
            self.map_data,
            self.aggregate_data
        ]
        for transformation in transformations:
            data = transformation(data)
        return data

    def filter_data(self, data: pd.DataFrame) -> pd.DataFrame:
        """
        Filters the data.
        
        Args:
        data (pd.DataFrame): The data to filter.
        
        Returns:
        pd.DataFrame: The filtered data.
        """
        return data[data['column'] > 0]

    def map_data(self, data: pd.DataFrame) -> pd.DataFrame:
        """
        Maps the data.
        
        Args:
        data (pd.DataFrame): The data to map.
        
        Returns:
        pd.DataFrame: The mapped data.
        """
        data['column'] = data['column'].apply(lambda x: x * 2)
        return data

    def aggregate_data(self, data: pd.DataFrame) -> pd.DataFrame:
        """
        Aggregates the data.
        
        Args:
        data (pd.DataFrame): The data to aggregate.
        
        Returns:
        pd.DataFrame: The aggregated data.
        """
        return data.groupby('column').sum()

    def output_data(self, data: pd.DataFrame) -> None:
        """
        Writes the transformed data to the output file.
        
        Args:

    def run(self) -> None:
        """
        Runs the data pipeline.
        """
        data = self.ingest_data()
        data = self.apply_transformations(data)
        self.output_data(data)

## System Components

2. **Transformation Chain**: Applies a series of transformations (filtering, mapping, aggregation) to the ingested data.
3. **Plugin System**: Allows for custom transformations to be added to the pipeline.
4. **Data Output**: Writes the transformed data to CSV or JSON files.

## Data Flow
The data flow through the pipeline is as follows:

1. **Data Ingestion**: Reads CSV files and performs schema validation using a configurable schema.
2. **Data Validation**: If the data is valid, it is passed to the transformation chain. If the data is invalid, an error is reported.
3. **Transformation Chain**: Applies a series of transformations to the data, including filtering, mapping, and aggregation.
4. **Plugin System**: Custom transformations can be added to the pipeline using the plugin system.
5. **Data Output**: The transformed data is written to a CSV or JSON file.
