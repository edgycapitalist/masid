# url_shortener.py
import hashlib

class URLShortener:
    def __init__(self):
        self.salt = "salt"

    def shorten_url(self, long_url):
        hash_object = hashlib.sha256((long_url + self.salt).encode())
        return hash_object.hexdigest()[:6]

# url_mapper.py
class URLMapper:
    def __init__(self):
        self.url_map = {}

    def add_mapping(self, short_url, original_url):
        self.url_map[short_url] = original_url

    def get_original_url(self, short_url):
        return self.url_map.get(short_url)

# collision_detector.py
class CollisionDetector:
    def __init__(self):
        self.url_map = {}

    def detect_collision(self, short_url, original_url):
        if short_url in self.url_map:
            return True
        self.url_map[short_url] = original_url
        return False

# input_validator.py
import re

class InputValidator:
    def __init__(self):
        self.url_pattern = re.compile(r"^https?://[^\s]+$")

    def validate_url(self, url):
        return bool(self.url_pattern.match(url))

# thread_safe_url_shortener.py
import threading

class ThreadSafeURLShortener:
    def __init__(self):
        self.url_shortener = URLShortener()
        self.url_mapper = URLMapper()
        self.collision_detector = CollisionDetector()
        self.input_validator = InputValidator()
        self.lock = threading.Lock()

    def shorten_url(self, long_url):
        with self.lock:
            if not self.input_validator.validate_url(long_url):
                raise ValueError("Invalid URL")
            short_url = self.url_shortener.shorten_url(long_url)
            if self.collision_detector.detect_collision(short_url, long_url):
                raise ValueError("Collision detected")
            self.url_mapper.add_mapping(short_url, long_url)
            return short_url

    def get_original_url(self, short_url):
        with self.lock:
            return self.url_mapper.get_original_url(short_url)

        self.url_shortener = URLShortener(salt)
