from functools import wraps
from time import time

class RateLimiter:
    def __init__(self, rate, burst_size):
        """
        Initialize the rate limiter.

        Args:
        rate (float): The rate at which tokens are refilled per second.
        burst_size (int): The maximum number of tokens in the bucket.
        """
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_refill = time()
        self.lock = threading.Lock()

    def allow_request(self):
        """
        Check if a request is allowed.

        Returns:
        bool: True if the request is allowed, False otherwise.
        """
        with self.lock:
            self._refill_bucket()
            if self.tokens > 0:
                self.tokens -= 1
                return True
            return False

    def _refill_bucket(self):
        """
        Refill the bucket with tokens.
        """
        current_time = time()
        elapsed_time = current_time - self.last_refill
        self.last_refill = current_time
        self.tokens = min(self.burst_size, self.tokens + elapsed_time * self.rate)

class RateLimiterDecorator:
    def __init__(self, rate_limiter):

        Initialize the rate limiter decorator.

        Args:
        rate_limiter (RateLimiter): The rate limiter instance.
