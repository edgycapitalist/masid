# task_queue.py
import threading
from queue import Queue
from enum import Enum

class TaskStatus(Enum):
    PENDING = 1
    RUNNING = 2
    COMPLETED = 3
    FAILED = 4

class TaskQueue:
    def __init__(self):
        self.queue = Queue()
        self.lock = threading.Lock()
        self.task_status = {}

    def add_task(self, task, priority):
        with self.lock:
            self.queue.put((task, priority))
            self.task_status[task] = TaskStatus.PENDING

    def consume_task(self):
        with self.lock:
            if not self.queue.empty():
                task, priority = self.queue.get()
                self.task_status[task] = TaskStatus.RUNNING
                return task, priority
            return None, None

    def update_task_status(self, task, status):
        with self.lock:
            self.task_status[task] = status

# worker_thread.py
import threading
from task_queue import TaskQueue, TaskStatus

class WorkerThread(threading.Thread):
    def __init__(self, task_queue):
        super().__init__()
        self.task_queue = task_queue

    def run(self):
        while True:
            task, priority = self.task_queue.consume_task()
            if task is None:
                break
            try:
                # Execute the task
                print(f"Executing task {task} with priority {priority}")
                self.task_queue.update_task_status(task, TaskStatus.COMPLETED)
            except Exception as e:
                self.task_queue.update_task_status(task, TaskStatus.FAILED)

# task_queue.py
import threading
from queue import Queue
from enum import Enum

class TaskStatus(Enum):
    PENDING = 1
    RUNNING = 2
    COMPLETED = 3
    FAILED = 4

class TaskQueue:
    def __init__(self, max_size):
        self.queue = Queue(maxsize=max_size)
        self.lock = threading.Lock()
        self.task_status = {}
        self.max_retries = 3

    def add_task(self, task, priority):
        with self.lock:
            if not self.queue.full():
                self.queue.put((task, priority))
                self.task_status[task] = TaskStatus.PENDING
                return True
            return False

    def consume_task(self):
        with self.lock:
            if not self.queue.empty():
                task, priority = self.queue.get()
                self.task_status[task] = TaskStatus.RUNNING
                return task, priority
            return None, None

    def update_task_status(self, task, status):
        with self.lock:
            self.task_status[task] = status

    def retry_task(self, task):
        with self.lock:
            if self.task_status[task] == TaskStatus.FAILED:
                # Retry the task
                print(f"Retrying task {task}")
                self.task_status[task] = TaskStatus.RUNNING
                self.queue.put((task, 1))  # Add the task back to the queue
                return True
            return False

import threading
from task_queue import TaskQueue, TaskStatus

class WorkerThread(threading.Thread):
    def __init__(self, task_queue):
        super().__init__()
        self.task_queue = task_queue
        self.running = True

    def run(self):
        while self.running:
            task, priority = self.task_queue.consume_task()
            if task is None:
                break
            try:
                # Execute the task
                print(f"Executing task {task} with priority {priority}")
                self.task_queue.update_task_status(task, TaskStatus.COMPLETED)
            except Exception as e:
                self.task_queue.update_task_status(task, TaskStatus.FAILED)
                print(f"Task {task} failed with error {e}")
                if self.task_queue.retry_task(task):
                    print(f"Task {task} will be retried")
                else:
                    print(f"Task {task} failed after max retries")

    def stop(self):
        self.running = False

class TaskStatus(Enum):
    PENDING = 1
    RUNNING = 2
    COMPLETED = 3
    FAILED = 4

        self.max_retries = 3

                self.queue.put((task, priority))

            if not self.queue.empty():
                task, priority = self.queue.get()
                self.task_status[task] = TaskStatus.RUNNING

                self.task_status[task] = TaskStatus.RUNNING
                self.queue.put((task, 1))  # Add the task back to the queue
