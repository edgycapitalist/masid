import pandas as pd
import json
from abc import ABC, abstractmethod
from typing import Callable

# Data Ingestion
class CSVReader:
    def __init__(self, file_path: str, schema_definition: dict):
        self.file_path = file_path
        self.schema_definition = schema_definition

    def read(self) -> pd.DataFrame:
        try:
            data = pd.read_csv(self.file_path)
            self.validate_schema(data)
            return data
        except Exception as e:
            raise DataIngestionError(f"Error reading CSV file: {e}")

    def validate_schema(self, data: pd.DataFrame) -> None:
        # Validate schema using schema definition
        for column, dtype in self.schema_definition.items():
            if column not in data.columns:
                raise DataIngestionError(f"Column '{column}' not found in CSV file")
            if not all(isinstance(x, dtype) for x in data[column]):
                raise DataIngestionError(f"Column '{column}' has incorrect data type")

# Transformation Chain
class Transformation(ABC):
    @abstractmethod
    def apply(self, data: pd.DataFrame) -> pd.DataFrame:
        pass

class FilterTransformation(Transformation):
    def __init__(self, condition: Callable[[pd.DataFrame], bool]):
        self.condition = condition

    def apply(self, data: pd.DataFrame) -> pd.DataFrame:
        return data[self.condition(data)]

class MapTransformation(Transformation):
    def __init__(self, mapping: Callable[[pd.DataFrame], pd.DataFrame]):
        self.mapping = mapping

    def apply(self, data: pd.DataFrame) -> pd.DataFrame:
        return self.mapping(data)

class AggregateTransformation(Transformation):
    def __init__(self, aggregation: Callable[[pd.DataFrame], pd.DataFrame]):
        self.aggregation = aggregation

    def apply(self, data: pd.DataFrame) -> pd.DataFrame:
        return self.aggregation(data)

# Plugin System
class Plugin(Transformation):
    def __init__(self, apply: Callable[[pd.DataFrame], pd.DataFrame]):
        self.apply = apply

class PluginRegistry:
    def __init__(self):
        self.plugins = []

    def register(self, plugin: Plugin) -> None:
        self.plugins.append(plugin)

    def apply(self, data: pd.DataFrame) -> pd.DataFrame:
        for plugin in self.plugins:
            data = plugin.apply(data)
        return data

# Data Output
class CSVWriter:
    def __init__(self, file_path: str):
        self.file_path = file_path

    def write(self, data: pd.DataFrame) -> None:
        data.to_csv(self.file_path, index=False)

        self.file_path = file_path

    def write(self, data: pd.DataFrame) -> None:
        with open(self.file_path, 'w') as f:

class JSONWriter:
    def __init__(self, file_path: str):
        self.file_path = file_path

    def write(self, data: pd.DataFrame) -> None:

# Error Handling
class DataIngestionError(Exception):
    pass

class TransformationError(Exception):
    pass

class OutputError(Exception):
    pass

# Pipeline
class Pipeline:
    def __init__(self, csv_reader: CSVReader, transformations: list, output_writer: CSVWriter or JSONWriter):
        self.csv_reader = csv_reader
        self.transformations = transformations
        self.output_writer = output_writer

    def run(self) -> None:
        data = self.csv_reader.read()
        for transformation in self.transformations:
            try:
                data = transformation.apply(data)
            except Exception as e:
                raise TransformationError(f"Error applying transformation: {e}")
        try:
            self.output_writer.write(data)
        except Exception as e:
            raise OutputError(f"Error writing output: {e}")
