## Introduction
The URL Shortener Library is a Python library that provides a simple and efficient way to shorten URLs using a hash-based approach. The library supports reverse lookup, collision detection, and input validation, making it suitable for concurrent usage.

## System Components
The library consists of the following components:

1. **URLShortener**: The main class responsible for shortening URLs and performing reverse lookups.
2. **URLValidator**: A helper class that validates the input URL format.
3. **HashGenerator**: A helper class that generates a unique hash for each URL.
4. **CollisionDetector**: A helper class that detects collisions and handles them accordingly.

## System Architecture
The library uses a modular architecture, with each component interacting with the others through well-defined interfaces.

1. **URLShortener**:
	* Accepts a long URL as input and validates it using **URLValidator**.
	* Generates a unique hash for the URL using **HashGenerator**.
	* Checks for collisions using **CollisionDetector**.
	* If no collision is detected, stores the mapping between the short URL and the original URL.
	* Returns the shortened URL.
2. **URLValidator**:
	* Checks if the input URL is in a valid format (e.g., starts with http:// or https://).
	* Returns a boolean indicating whether the URL is valid.
3. **HashGenerator**:
	* Generates a unique hash for the input URL using a hash function (e.g., SHA-256).
	* Returns the generated hash.
4. **CollisionDetector**:
	* Checks if a collision has occurred by verifying if the generated hash is already in use.
	* If a collision is detected, generates a new hash and repeats the process until a unique hash is found.

## Data Flow
The data flow between components is as follows:

1. **URLShortener** receives a long URL as input.
2. **URLShortener** calls **URLValidator** to validate the input URL.
3. If the URL is valid, **URLShortener** calls **HashGenerator** to generate a unique hash.
4. **URLShortener** calls **CollisionDetector** to check for collisions.
5. If no collision is detected, **URLShortener** stores the mapping between the short URL and the original URL.
6. **URLShortener** returns the shortened URL.

import hashlib
import threading
from urllib.parse import urlparse

class URLValidator:
    def __init__(self):
        pass

    def is_valid(self, url):
        try:
            result = urlparse(url)
            return all([result.scheme, result.netloc])
        except ValueError:
            return False

class HashGenerator:
    def __init__(self):
        pass

    def generate_hash(self, url):
        return hashlib.sha256(url.encode()).hexdigest()[:6]

class CollisionDetector:
    def __init__(self):
        self.lock = threading.Lock()
        self.hash_map = {}

    def is_collision(self, hash_value):
        with self.lock:
            return hash_value in self.hash_map

    def add_hash(self, hash_value, url):
        with self.lock:
            self.hash_map[hash_value] = url

class URLShortener:
    def __init__(self):
        self.validator = URLValidator()
        self.hash_generator = HashGenerator()
        self.collision_detector = CollisionDetector()

    def shorten_url(self, url):
        if not self.validator.is_valid(url):
            raise ValueError("Invalid URL")

        hash_value = self.hash_generator.generate_hash(url)
        while self.collision_detector.is_collision(hash_value):
            hash_value = self.hash_generator.generate_hash(url + str(hash_value))

        self.collision_detector.add_hash(hash_value, url)
        return f"http://short.url/{hash_value}"

    def get_original_url(self, short_url):
        hash_value = short_url.split("/")[-1]
        return self.collision_detector.hash_map.get(hash_value)

# Example usage:
url_shortener = URLShortener()
short_url = url_shortener.shorten_url("https://www.example.com")
print(short_url)
print(url_shortener.get_original_url(short_url))
```

    def __init__(self):

    def __init__(self):
        pass

class CollisionDetector:

        self.hash_map = {}

        while self.collision_detector.is_collision(hash_value):

        return self.collision_detector.hash_map.get(hash_value)

import threading
from urllib.parse import urlparse

class URLValidator:
    def __init__(self):
        pass

    def is_valid(self, url):
        try:
            result = urlparse(url)
            return all([result.scheme, result.netloc])
        except ValueError:
            return False

class HashGenerator:
    def __init__(self):
        pass

    def generate_hash(self, url):
        return hashlib.sha256(url.encode()).hexdigest()[:6]

class CollisionDetector:
    def __init__(self):
        self.lock = threading.Lock()
        self.hash_map = {}

    def is_collision(self, hash_value):
        with self.lock:
            return hash_value in self.hash_map

    def add_hash(self, hash_value, url):
        with self.lock:
            self.hash_map[hash_value] = url

class URLShortener:
    def __init__(self):
        self.validator = URLValidator()
        self.hash_generator = HashGenerator()
        self.collision_detector = CollisionDetector()

    def shorten_url(self, url):
        if not self.validator.is_valid(url):
            raise ValueError("Invalid URL")

        hash_value = self.hash_generator.generate_hash(url)
        while self.collision_detector.is_collision(hash_value):
            hash_value = self.hash_generator.generate_hash(url + str(hash_value))

        self.collision_detector.add_hash(hash_value, url)
        return f"http://short.url/{hash_value}"

        hash_value = short_url.split("/")[-1]
        return self.collision_detector.hash_map.get(hash_value)

    def get_original_url_or_none(self, short_url):
        original_url = self.get_original_url(short_url)
        if original_url is None:
            return None
        return original_url

    def is_short_url_valid(self, short_url):
        try:
            result = urlparse(short_url)
            return all([result.scheme, result.netloc, result.path])
        except ValueError:
            return False

    def is_valid(self, url):
        if not url:  # Check for empty string
            return False
        try:
            result = urlparse(url)
            return all([result.scheme, result.netloc])
        except ValueError:
            return False

class HashGenerator:
    def generate_hash(self, url):

class CollisionDetector:
    def __init__(self):
        self.lock = threading.Lock()
        self.hash_map = {}

    def is_collision(self, hash_value):
        with self.lock:
            return hash_value in self.hash_map

    def add_hash(self, hash_value, url):
        with self.lock:
            self.hash_map[hash_value] = url

class URLShortener:
    def __init__(self):
        self.validator = URLValidator()
        self.hash_generator = HashGenerator()
        self.collision_detector = CollisionDetector()

    def shorten_url(self, url):
        if not self.validator.is_valid(url):
            raise ValueError("Invalid URL")

        hash_value = self.hash_generator.generate_hash(url)
        while self.collision_detector.is_collision(hash_value):
            hash_value = self.hash_generator.generate_hash(url + str(hash_value))

        self.collision_detector.add_hash(hash_value, url)
        return f"http://short.url/{hash_value}"

    def get_original_url(self, short_url):
        hash_value = short_url.split("/")[-1]
        return self.collision_detector.hash_map.get(hash_value)

    def get_original_url_or_none(self, short_url):
        original_url = self.get_original_url(short_url)
        if original_url is None:
            return None
        return original_url

    def is_short_url_valid(self, short_url):
        try:
            result = urlparse(short_url)
            return all([result.scheme, result.netloc, result.path])
        except ValueError:
            return False
