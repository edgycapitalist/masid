import time
import threading
from functools import wraps

class RateLimitExceeded(Exception):
    pass

class TokenBucket:
    def __init__(self, rate, burst_size):
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()

    def update(self):
        now = time.time()
        elapsed = now - self.last_update
        self.tokens = min(self.burst_size, self.tokens + elapsed * self.rate)
        self.last_update = now

    def consume(self):
        if self.tokens < 1:
            return False
        self.tokens -= 1
        return True

class RateLimiter:
    def __init__(self, name, rate, burst_size):
        self.name = name
        self.rate = rate
        self.burst_size = burst_size
        self.token_bucket = TokenBucket(rate, burst_size)
        self.lock = threading.Lock()

    def allow_request(self):
        with self.lock:
            self.token_bucket.update()
            return self.token_bucket.consume()

    def update_token_bucket(self):
        with self.lock:
            self.token_bucket.update()

def rate_limiter_decorator(rate_limiter):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            if not rate_limiter.allow_request():
                raise RateLimitExceeded(f"Rate limit exceeded for {rate_limiter.name}")
            return func(*args, **kwargs)
        return wrapper
    return decorator

# Example usage
rate_limiter = RateLimiter("example", 5, 10)

@rate_limiter_decorator(rate_limiter)
def example_function():
    print("Hello, world!")

example_function()
