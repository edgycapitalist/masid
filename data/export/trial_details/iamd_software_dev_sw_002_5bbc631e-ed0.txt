```python
import threading
import time
from functools import wraps

class RateLimiter:
    """The core class responsible for managing the token bucket algorithm."""

    def __init__(self, rate, burst_size):
        """
        Initialize the rate limiter.

        Args:
        rate (float): The rate at which tokens are added to the bucket (requests per second).
        burst_size (int): The maximum number of tokens in the bucket.
        """
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()
        self.lock = threading.Lock()

    def allow(self):
        """
        Check if a request is allowed based on the current token count.

        Returns:
        bool: True if the request is allowed, False otherwise.
        """
        with self.lock:
            self.update()
            if self.tokens > 0:
                self.tokens -= 1
                return True
            return False

    def update(self):
        """
        Update the token count based on the elapsed time since the last update.
        """
        now = time.time()
        elapsed = now - self.last_update
        self.tokens = min(self.burst_size, self.tokens + elapsed * self.rate)
        self.last_update = now


class NamedRateLimiter:
    """A wrapper class that provides multiple named rate limiters."""

    def __init__(self):
        """
        Initialize the named rate limiter.
        """
        self.rate_limiters = {}
        self.lock = threading.Lock()

    def get_limiter(self, name, rate, burst_size):
        """
        Get or create a rate limiter with the specified name.

        Args:
        name (str): The name of the rate limiter.
        rate (float): The rate at which tokens are added to the bucket (requests per second).
        burst_size (int): The maximum number of tokens in the bucket.

        Returns:
        RateLimiter: The rate limiter instance.
        """
        with self.lock:
            if name not in self.rate_limiters:
                self.rate_limiters[name] = RateLimiter(rate, burst_size)
            return self.rate_limiters[name]

    def rate_limited(self, name, rate, burst_size):
        """
        A decorator that applies rate limiting to functions.

        Args:
        name (str): The name of the rate limiter.
        rate (float): The rate at which tokens are added to the bucket (requests per second).
        burst_size (int): The maximum number of tokens in the bucket.

        Returns:
        function: The decorated function.
        """
        limiter = self.get_limiter(name, rate, burst_size)

        def decorator(func):
            @wraps(func)
            def wrapper(*args, **kwargs):
                if limiter.allow():
                    return func(*args, **kwargs)
                else:
                    raise Exception("Rate limit exceeded")

            return wrapper

        return decorator

@named_limiter.rate_limited("example", 5, 10)
def example_function():

        example_function()
    except Exception as e:
        print(e)
```

```python
import threading
import time
from functools import wraps

class RateLimitExceeded(Exception):
```
```python
    """A custom exception for rate limit exceeded errors."""
    pass
```

```python
class RateLimiter:
    """The core class responsible for managing the token bucket algorithm."""

    def __init__(self, rate, burst_size):
        """
        Initialize the rate limiter.

        Args:
        rate (float): The rate at which tokens are added to the bucket (requests per second).
        burst_size (int): The maximum number of tokens in the bucket.
        """
        self.rate = rate
        self.burst_size = burst_size
        self.tokens = burst_size
        self.last_update = time.time()
        self.lock = threading.Lock()

    def allow(self):
        """
        Check if a request is allowed based on the current token count.

        Returns:
        bool: True if the request is allowed, False otherwise.
        """
        with self.lock:
            self.update()
            if self.tokens > 0:
                self.tokens -= 1
                return True
            return False

    def update(self):
        """
        Update the token count based on the elapsed time since the last update.
        """
        now = time.time()
        elapsed = now - self.last_update
        self.tokens = min(self.burst_size, self.tokens + elapsed * self.rate)
        self.last_update = now
```
